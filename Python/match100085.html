1286     server/client implementation.

1288     :param str/int socket_path: A path on the filesystem where a socket

1294     """
1295     def __init__(self, socket_path, io_loop=None):
1296         """

1315         """

1317         """
1318         if self._connecting_future is not None and not self._connecting_future.done():
1319             future = self._connecting_future
1320         else:

1322                 self._connecting_future.exception()  # pylint: disable=E0203

1324             self._connecting_future = future
1325             self._connect(timeout)
1326         if callback is not None:
1327             def handle_future(future):
1328                 response = future.result()
1329                 self.io_loop.add_callback(callback, response)
1330             future.add_done_callback(handle_future)
1331         return future

1351                     self.stream = IOStream(socket.socket(sock_type, socket.SOCK_STREAM))

1361                     if self.stream is not None:
