35      default=" ",

37  )

40  for line in args.infile:
41    if args.downcase:
42      line = line.lower()

50  if args.min_frequency > 0:

53    cnt = collections.Counter(dict(filtered_tokens))
54  logging.info("Found %d unique tokens with frequency > %d.",
55               len(cnt), args.min_frequency)
56  word_with_counts = cnt.most_common()

59  if args.max_vocab_size is not None:
