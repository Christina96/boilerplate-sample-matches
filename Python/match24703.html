            "6666/tcp": [{"HostIp": "0.0.0.0", "HostPort": "32769"}],
        },
        "bar": {
            "4444/udp": [{"HostIp": "0.0.0.0", "HostPort": "32767"}],
            "5555/tcp": [{"HostIp": "0.0.0.0", "HostPort": "32768"}],
            "6666/tcp": [{"HostIp": "0.0.0.0", "HostPort": "32769"}],
        },

            "6666/udp": [{"HostIp": "0.0.0.0", "HostPort": "32769"}],
        },

        docker_mod, "inspect_container", inspect_mock
