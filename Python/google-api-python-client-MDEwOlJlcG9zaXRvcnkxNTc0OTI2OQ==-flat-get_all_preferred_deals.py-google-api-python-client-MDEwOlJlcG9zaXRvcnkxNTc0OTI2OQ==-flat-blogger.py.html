16      try:

18          thisuser = users.get(userId="self").execute()
19          print("This user's display name is: %s" % thisuser["displayName"])
20          blogs = service.blogs()
21          thisusersblogs = blogs.listByUser(userId="self").execute()
22          for blog in thisusersblogs["items"]:
23              print("The blog named '%s' is at: %s" % (blog["name"], blog["url"]))
24          posts = service.posts()
25          for blog in thisusersblogs["items"]:
26              print("The posts for %s:" % blog["name"])
27              request = posts.list(blogId=blog["id"])
28              while request != None:
29                  posts_doc = request.execute()
30                  if "items" in posts_doc and not (posts_doc["items"] is None):
31                      for post in posts_doc["items"]:
32                          print("  %s (%s)" % (post["title"], post["url"]))
33                  request = posts.list_next(request, posts_doc)
34      except client.AccessTokenRefreshError:
35          print(
36              "The credentials have been revoked or expired, please re-run"
