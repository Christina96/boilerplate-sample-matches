688     def _handle_aes(self, data):

690             log.error("Received malformed command %s", data)
691             return {}
692         cmd = data["cmd"]

694         method = self.aes_funcs.get_method(cmd)
695         if not method:
696             return {}, {"fun": "send"}

703             functools.partial(RequestContext, {"data": data, "opts": self.opts})
704         ):
