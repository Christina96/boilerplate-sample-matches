324  def espefuse(esp, efuses, args, command):
325      parser = argparse.ArgumentParser()

327      add_commands(subparsers, efuses)
328      try:
329          cmd_line_args = parser.parse_args(command.split())
330      except SystemExit:
331          traceback.print_stack()
332          raise esptool.FatalError('"{}" - incorrect command'.format(command))
333      if cmd_line_args.operation == "execute_scripts":
334          configfiles = cmd_line_args.configfiles
335          index = cmd_line_args.index
336      vars(cmd_line_args).update(vars(args))
337      if cmd_line_args.operation == "execute_scripts":
338          cmd_line_args.configfiles = configfiles
339          cmd_line_args.index = index
340      if cmd_line_args.operation is None:
341          parser.print_help()
342          parser.exit(1)
343      operation_func = globals()[cmd_line_args.operation]
344      operation_func(esp, efuses, cmd_line_args)
345  def execute_scripts(esp, efuses, args):
346      efuses.batch_mode_cnt += 1
347      del args.operation
348      scripts = args.scripts
349      del args.scripts
350      for file in scripts:
351          with open(file.name, "r") as file:
352              exec(compile(file.read(), file.name, "exec"))
353      if args.debug:
354          for block in efuses.blocks:
355              data = block.get_bitstring(from_read=False)
356              block.print_block(data, "regs_for_burn", args.debug)
357      efuses.batch_mode_cnt -= 1
358      if not efuses.burn_all(check_batch_mode=True):
359          return
360      print("Successful")
