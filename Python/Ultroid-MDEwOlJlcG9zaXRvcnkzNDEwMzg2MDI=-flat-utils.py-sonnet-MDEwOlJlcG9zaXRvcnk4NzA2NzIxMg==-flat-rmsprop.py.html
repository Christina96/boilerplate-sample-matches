10    ms = tf.square(update) * (1. - decay) + ms * decay
11    if mg is not None:  # centered
12      mg = update * (1. - decay) + mg * decay
13      denominator = ms - tf.square(mg) + epsilon

15      denominator = ms + epsilon

17    return mom, ms, mg
