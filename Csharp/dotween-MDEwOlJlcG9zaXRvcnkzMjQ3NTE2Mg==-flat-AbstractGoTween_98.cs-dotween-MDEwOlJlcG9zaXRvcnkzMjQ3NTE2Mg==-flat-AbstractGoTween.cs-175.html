203  		reset( skipDelay );
204  		pause();
205  	}

207  	{

209  		play();
210  	}
211  	public virtual void reverse()
212  	{
213  		isReversed = !isReversed;
214  		_completedIterations = isReversed ? Mathf.CeilToInt( _totalElapsedTime / duration ) : Mathf.FloorToInt( _totalElapsedTime / duration );
215  		if( ( isReversed && _totalElapsedTime == totalDuration ) || ( !isReversed && _totalElapsedTime == 0f ) )
216  		{
217  			_didBegin = false;
218  			_fireIterationStart = true;
219  		}
220  	}
221  	public virtual void complete()
222  	{
223  		if( iterations < 0 )
224  			return;
225  		goTo( isReversed ? 0 : totalDuration, true );
226  	}

228  	public void goToAndPlay( float time, bool skipDelay = true )

232  	}
