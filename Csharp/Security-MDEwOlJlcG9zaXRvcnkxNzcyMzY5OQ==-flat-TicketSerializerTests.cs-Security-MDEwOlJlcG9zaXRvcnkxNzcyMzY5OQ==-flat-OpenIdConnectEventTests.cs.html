14          {
15              var serializer = new TicketSerializer();
16              var properties = new AuthenticationProperties();
17              properties.RedirectUri = "bye";
18              var ticket = new AuthenticationTicket(new ClaimsPrincipal(), properties, "Hello");
19              using (var stream = new MemoryStream())
20              using (var writer = new BinaryWriter(stream))
21              using (var reader = new BinaryReader(stream))
22              {
23                  serializer.Write(writer, ticket);
24                  stream.Position = 0;
25                  var readTicket = serializer.Read(reader);
26                  Assert.Empty(readTicket.Principal.Identities);
27                  Assert.Equal("bye", readTicket.Properties.RedirectUri);
28                  Assert.Equal("Hello", readTicket.AuthenticationScheme);
29              }
30          }

37              ticket.Principal.AddIdentity(new ClaimsIdentity("misc") { BootstrapContext = "bootstrap" });
38              using (var stream = new MemoryStream())
39              using (var writer = new BinaryWriter(stream))
