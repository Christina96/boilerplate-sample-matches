271          private void TryReceive()
272          {
273              if (_receiving)
274                  return;
275              if (!IsConnected)
276                  return;
277              bool process = true;
278              while (process)
279              {
280                  process = false;
281                  try
282                  {
283                      _receiving = true;
284                      _receiveEventArg.SetBuffer(_receiveBuffer.Data, 0, (int)_receiveBuffer.Capacity);
285                      if (!Socket.ReceiveAsync(_receiveEventArg))
286                          process = ProcessReceive(_receiveEventArg);
287                  }
288                  catch (ObjectDisposedException) {}
289              }
290          }
291          private void TrySend()
292          {
293              if (!IsConnected)
294                  return;
295              bool empty = false;
296              bool process = true;
297              while (process)
298              {
299                  process = false;
300                  lock (_sendLock)
301                  {
302                      if (_sendBufferFlush.IsEmpty)
303                      {
304                          _sendBufferFlush = Interlocked.Exchange(ref _sendBufferMain, _sendBufferFlush);
305                          _sendBufferFlushOffset = 0;
306                          BytesPending = 0;
307                          BytesSending += _sendBufferFlush.Size;
308                          if (_sendBufferFlush.IsEmpty)
309                          {
310                              empty = true;
311                              _sending = false;
312                          }
313                      }
314                      else
315                          return;
316                  }
317                  if (empty)
318                  {
319                      OnEmpty();
320                      return;
321                  }
322                  try
323                  {
324                      _sendEventArg.SetBuffer(_sendBufferFlush.Data, (int)_sendBufferFlushOffset, (int)(_sendBufferFlush.Size - _sendBufferFlushOffset));
325                      if (!Socket.SendAsync(_sendEventArg))
326                          process = ProcessSend(_sendEventArg);
327                  }
328                  catch (ObjectDisposedException) {}
329              }
330          }
331          private void ClearBuffers()
332          {
333              lock (_sendLock)
334              {
335                  _sendBufferMain.Clear();
336                  _sendBufferFlush.Clear();
337                  _sendBufferFlushOffset= 0;
338                  BytesPending = 0;
339                  BytesSending = 0;
340              }
341          }

344          private void OnAsyncCompleted(object sender, SocketAsyncEventArgs e)
345          {
346              if (IsSocketDisposed)
347                  return;
348              switch (e.LastOperation)
349              {
350                  case SocketAsyncOperation.Connect:
351                      ProcessConnect(e);
352                      break;
353                  case SocketAsyncOperation.Receive:
354                      if (ProcessReceive(e))
355                          TryReceive();
356                      break;
357                  case SocketAsyncOperation.Send:
358                      if (ProcessSend(e))
359                          TrySend();
360                      break;
361                  default:
362                      throw new ArgumentException("The last operation completed on the socket was not a receive or send");
363              }
364          }
365          private void ProcessConnect(SocketAsyncEventArgs e)
366          {
367              IsConnecting = false;
368              if (e.SocketError == SocketError.Success)
369              {
370                  if (OptionKeepAlive)
371                      Socket.SetSocketOption(SocketOptionLevel.Socket, SocketOptionName.KeepAlive, true);
372                  if (OptionTcpKeepAliveTime >= 0)
373                      Socket.SetSocketOption(SocketOptionLevel.Tcp, SocketOptionName.TcpKeepAliveTime, OptionTcpKeepAliveTime);
374                  if (OptionTcpKeepAliveInterval >= 0)
375                      Socket.SetSocketOption(SocketOptionLevel.Tcp, SocketOptionName.TcpKeepAliveInterval, OptionTcpKeepAliveInterval);
376                  if (OptionTcpKeepAliveRetryCount >= 0)
377                      Socket.SetSocketOption(SocketOptionLevel.Tcp, SocketOptionName.TcpKeepAliveRetryCount, OptionTcpKeepAliveRetryCount);
378                  if (OptionNoDelay)
379                      Socket.SetSocketOption(SocketOptionLevel.Tcp, SocketOptionName.NoDelay, true);
380                  _receiveBuffer.Reserve(OptionReceiveBufferSize);
381                  _sendBufferMain.Reserve(OptionSendBufferSize);
382                  _sendBufferFlush.Reserve(OptionSendBufferSize);
383                  BytesPending = 0;
384                  BytesSending = 0;
385                  BytesSent = 0;
386                  BytesReceived = 0;
387                  IsConnected = true;
388                  TryReceive();
389                  if (IsSocketDisposed)
390                      return;
391                  OnConnected();
392                  if (_sendBufferMain.IsEmpty)
393                      OnEmpty();
394              }
395              else
396              {
397                  SendError(e.SocketError);
398                  OnDisconnected();
399              }
400          }
401          private bool ProcessReceive(SocketAsyncEventArgs e)
402          {
403              if (!IsConnected)
404                  return false;
405              long size = e.BytesTransferred;
406              if (size > 0)
407              {
408                  BytesReceived += size;
409                  OnReceived(_receiveBuffer.Data, 0, size);
410                  if (_receiveBuffer.Capacity == size)
411                  {
412                      if (((2 * size) > OptionReceiveBufferLimit) && (OptionReceiveBufferLimit > 0))
413                      {
414                          SendError(SocketError.NoBufferSpaceAvailable);
415                          DisconnectAsync();
416                          return false;
417                      }
418                      _receiveBuffer.Reserve(2 * size);
419                  }
420              }
421              _receiving = false;
422              if (e.SocketError == SocketError.Success)
423              {
424                  if (size > 0)
425                      return true;
426                  else
427                      DisconnectAsync();
428              }
429              else
430              {
431                  SendError(e.SocketError);
432                  DisconnectAsync();
433              }
434              return false;
435          }
436          private bool ProcessSend(SocketAsyncEventArgs e)
437          {
438              if (!IsConnected)
439                  return false;
440              long size = e.BytesTransferred;
441              if (size > 0)
442              {
443                  BytesSending -= size;
444                  BytesSent += size;
445                  _sendBufferFlushOffset += size;
446                  if (_sendBufferFlushOffset == _sendBufferFlush.Size)
447                  {
448                      _sendBufferFlush.Clear();
449                      _sendBufferFlushOffset = 0;
450                  }
451                  OnSent(size, BytesPending + BytesSending);
452              }
453              if (e.SocketError == SocketError.Success)
454                  return true;
455              else
456              {
457                  SendError(e.SocketError);
458                  DisconnectAsync();
459                  return false;
460              }
461          }

467          protected virtual void OnDisconnected() {}
468          protected virtual void OnReceived(byte[] buffer, long offset, long size) {}
469          protected virtual void OnSent(long sent, long pending) {}
470          protected virtual void OnEmpty() {}
471          protected virtual void OnError(SocketError error) {}
472          #endregion

475          {
476              if ((error == SocketError.ConnectionAborted) ||
477                  (error == SocketError.ConnectionRefused) ||
478                  (error == SocketError.ConnectionReset) ||
479                  (error == SocketError.OperationAborted) ||
480                  (error == SocketError.Shutdown))
481                  return;
482              OnError(error);
483          }
