337              if (expressionFunctions?.Count > 0)
338              {
339                  foreach (var function in expressionFunctions)

341                      result.ExpressionFunctions.Add(function);
342                      functionNames.Add(function.Name);

360              {
361                  foreach (var pair in expressionState)
