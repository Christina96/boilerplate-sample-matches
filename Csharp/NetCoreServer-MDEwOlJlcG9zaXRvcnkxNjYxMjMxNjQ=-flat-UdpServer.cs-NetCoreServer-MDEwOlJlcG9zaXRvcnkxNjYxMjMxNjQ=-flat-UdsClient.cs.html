335                      throw new ArgumentException("The last operation completed on the socket was not a receive or send");
336              }

338          private void ProcessConnect(SocketAsyncEventArgs e)
339          {
340              IsConnecting = false;
341              if (e.SocketError == SocketError.Success)
342              {
343                  _receiveBuffer.Reserve(OptionReceiveBufferSize);
344                  _sendBufferMain.Reserve(OptionSendBufferSize);
345                  _sendBufferFlush.Reserve(OptionSendBufferSize);
346                  BytesPending = 0;
347                  BytesSending = 0;
348                  BytesSent = 0;
349                  BytesReceived = 0;
350                  IsConnected = true;
351                  TryReceive();
352                  if (IsSocketDisposed)
353                      return;
354                  OnConnected();
355                  if (_sendBufferMain.IsEmpty)
356                      OnEmpty();
357              }
358              else
359              {
360                  SendError(e.SocketError);
361                  OnDisconnected();
362              }
363          }
364          private bool ProcessReceive(SocketAsyncEventArgs e)
365          {
366              if (!IsConnected)
367                  return false;
368              long size = e.BytesTransferred;
369              if (size > 0)
370              {
371                  BytesReceived += size;
372                  OnReceived(_receiveBuffer.Data, 0, size);
373                  if (_receiveBuffer.Capacity == size)
374                  {
375                      if (((2 * size) > OptionReceiveBufferLimit) && (OptionReceiveBufferLimit > 0))
376                      {
377                          SendError(SocketError.NoBufferSpaceAvailable);
378                          DisconnectAsync();
379                          return false;
380                      }
381                      _receiveBuffer.Reserve(2 * size);
382                  }
383              }
384              _receiving = false;
385              if (e.SocketError == SocketError.Success)
386              {
387                  if (size > 0)
388                      return true;
389                  else
390                      DisconnectAsync();
391              }
392              else
393              {
394                  SendError(e.SocketError);
395                  DisconnectAsync();
396              }
397              return false;
398          }
399          private bool ProcessSend(SocketAsyncEventArgs e)
400          {
401              if (!IsConnected)
402                  return false;
403              long size = e.BytesTransferred;
404              if (size > 0)
405              {
406                  BytesSending -= size;
407                  BytesSent += size;
408                  _sendBufferFlushOffset += size;
409                  if (_sendBufferFlushOffset == _sendBufferFlush.Size)
410                  {
411                      _sendBufferFlush.Clear();
412                      _sendBufferFlushOffset = 0;
413                  }
414                  OnSent(size, BytesPending + BytesSending);
415              }
416              if (e.SocketError == SocketError.Success)
417                  return true;
418              else
419              {
420                  SendError(e.SocketError);
421                  DisconnectAsync();
422                  return false;
423              }
