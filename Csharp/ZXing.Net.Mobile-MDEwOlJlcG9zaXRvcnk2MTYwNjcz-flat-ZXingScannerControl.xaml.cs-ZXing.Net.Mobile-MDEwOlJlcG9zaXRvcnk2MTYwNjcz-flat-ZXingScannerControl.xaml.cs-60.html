507  		Rect ConvertUiTapToPreviewRect(Point tap, Size size, Rect previewRect)
508  		{
509  			double left = tap.X - size.Width / 2, top = tap.Y - size.Height / 2;
510  			double previewWidth = previewRect.Width, previewHeight = previewRect.Height;
511  			double previewLeft = previewRect.Left, previewTop = previewRect.Top;
512  			switch (displayOrientation)
513  			{
514  				case DisplayOrientations.Portrait:
515  					var tempLeft = left;
516  					left = top;
517  					top = previewRect.Width - tempLeft;
518  					break;
519  				case DisplayOrientations.LandscapeFlipped:
520  					left = previewRect.Width - left;
521  					top = previewRect.Height - top;
522  					break;
523  				case DisplayOrientations.PortraitFlipped:
524  					var tempTop = top;
525  					top = left;
526  					left = previewRect.Width - tempTop;
527  					break;
528  			}
529  			if (displayOrientation == DisplayOrientations.Portrait || displayOrientation == DisplayOrientations.PortraitFlipped)
530  			{
531  				previewWidth = previewRect.Height;
532  				previewHeight = previewRect.Width;
533  				previewLeft = previewRect.Top;
534  				previewTop = previewRect.Left;
535  			}
536  			var width = size.Width / previewWidth;
537  			var height = size.Height / previewHeight;
538  			left -= previewLeft;
539  			top -= previewTop;
540  			left /= previewWidth;
541  			top /= previewHeight;
542  			left = Math.Max(left, 0);
543  			left = Math.Min(1 - width, left);
544  			top = Math.Max(top, 0);
545  			top = Math.Min(1 - height, top);
546  			return new Rect(left, top, width, height);
547  		}

549  		{
550  			var result = new Rect();
551  			if (previewControl == null || previewControl.ActualHeight < 1 || previewControl.ActualWidth < 1 ||
552  				previewResolution.Height < 1 || previewResolution.Width < 1)
553  			{
554  				return result;
555  			}
556  			var streamWidth = previewResolution.Width;
557  			var streamHeight = previewResolution.Height;
558  			result.Width = previewControl.ActualWidth;
559  			result.Height = previewControl.ActualHeight;
560  			if ((previewControl.ActualWidth / previewControl.ActualHeight > streamWidth / (double)streamHeight))
561  			{
562  				var scale = previewControl.ActualHeight / streamHeight;
563  				var scaledWidth = streamWidth * scale;
564  				result.X = (previewControl.ActualWidth - scaledWidth) / 2.0;
565  				result.Width = scaledWidth;
566  			}
567  			else 
568  			{
569  				var scale = previewControl.ActualWidth / streamWidth;
570  				var scaledHeight = streamHeight * scale;
571  				result.Y = (previewControl.ActualHeight - scaledHeight) / 2.0;
572  				result.Height = scaledHeight;
573  			}
574  			return result;

507  		Rect ConvertUiTapToPreviewRect(Point tap, Size size, Rect previewRect)
508  		{
509  			double left = tap.X - size.Width / 2, top = tap.Y - size.Height / 2;
510  			double previewWidth = previewRect.Width, previewHeight = previewRect.Height;
511  			double previewLeft = previewRect.Left, previewTop = previewRect.Top;
512  			switch (displayOrientation)
513  			{
514  				case DisplayOrientations.Portrait:
515  					var tempLeft = left;
516  					left = top;
517  					top = previewRect.Width - tempLeft;
518  					break;
519  				case DisplayOrientations.LandscapeFlipped:
520  					left = previewRect.Width - left;
521  					top = previewRect.Height - top;
522  					break;
523  				case DisplayOrientations.PortraitFlipped:
524  					var tempTop = top;
525  					top = left;
526  					left = previewRect.Width - tempTop;
527  					break;
528  			}
529  			if (displayOrientation == DisplayOrientations.Portrait || displayOrientation == DisplayOrientations.PortraitFlipped)
530  			{
531  				previewWidth = previewRect.Height;
532  				previewHeight = previewRect.Width;
533  				previewLeft = previewRect.Top;
534  				previewTop = previewRect.Left;
535  			}
536  			var width = size.Width / previewWidth;
537  			var height = size.Height / previewHeight;
538  			left -= previewLeft;
539  			top -= previewTop;
540  			left /= previewWidth;
541  			top /= previewHeight;
542  			left = Math.Max(left, 0);
543  			left = Math.Min(1 - width, left);
544  			top = Math.Max(top, 0);
545  			top = Math.Min(1 - height, top);
546  			return new Rect(left, top, width, height);
547  		}

549  		{

557  			var streamHeight = previewResolution.Height;
558  			result.Width = previewControl.ActualWidth;
559  			result.Height = previewControl.ActualHeight;
560  			if ((previewControl.ActualWidth / previewControl.ActualHeight > streamWidth / (double)streamHeight))
561  			{

563  				var scaledWidth = streamWidth * scale;
564  				result.X = (previewControl.ActualWidth - scaledWidth) / 2.0;
565  				result.Width = scaledWidth;
566  			}
567  			else 
568  			{
569  				var scale = previewControl.ActualWidth / streamWidth;
570  				var scaledHeight = streamHeight * scale;
571  				result.Y = (previewControl.ActualHeight - scaledHeight) / 2.0;
572  				result.Height = scaledHeight;
573  			}
574  			return result;
