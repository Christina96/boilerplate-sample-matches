183          internal void SignOut(
184              Uri serverUrl,
185              Uri serviceLocation,
186              string identityProvider)
187          {
188              if ((m_currentProvider != null) && (m_currentProvider.CurrentToken != null))
189              {
190                  if (m_currentProvider.Credential.Storage != null && m_currentProvider.TokenStorageUrl != null)
191                  {
192                      m_currentProvider.Credential.Storage.RemoveToken(m_currentProvider.TokenStorageUrl, m_currentProvider.CurrentToken);
193                  }
194                  m_currentProvider.CurrentToken = null;
195              }
196              ISupportSignOut tokenProviderWithSignOut = m_currentProvider as ISupportSignOut;
197              if (tokenProviderWithSignOut == null)
198              {
199                  return;
200              }
201              if (serviceLocation != null)
202              {
203                  string serviceLocationUri = serviceLocation.AbsoluteUri;
204                  serviceLocationUri = serviceLocationUri.Replace("{mode}", "SignOut");
205                  serviceLocationUri = serviceLocationUri.Replace("{redirectUrl}", serverUrl.AbsoluteUri);
206                  serviceLocation = new Uri(serviceLocationUri);
207              }
208              tokenProviderWithSignOut.SignOut(serviceLocation, serverUrl, identityProvider);
209          }
