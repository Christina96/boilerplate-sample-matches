303          {
304              if (_receiving)
305                  return;
306              if (!IsHandshaked)
307                  return;
308              try

310                  IAsyncResult result;
311                  do

313                      if (!IsHandshaked)
314                          return;
315                      _receiving = true;
316                      result = _sslStream.BeginRead(_receiveBuffer.Data, 0, (int)_receiveBuffer.Capacity, ProcessReceive, _sslStreamId);
317                  } while (result.CompletedSynchronously);

321          private void TrySend()
322          {
323              if (!IsHandshaked)
324                  return;
325              bool empty = false;
326              lock (_sendLock)
327              {
328                  if (_sendBufferFlush.IsEmpty)
329                  {
330                      _sendBufferFlush = Interlocked.Exchange(ref _sendBufferMain, _sendBufferFlush);
331                      _sendBufferFlushOffset = 0;
332                      BytesPending = 0;
333                      BytesSending += _sendBufferFlush.Size;
334                      if (_sendBufferFlush.IsEmpty)
335                      {
336                          empty = true;
337                          _sending = false;
338                      }
339                  }
340                  else
341                      return;
342              }
343              if (empty)
344              {
345                  OnEmpty();
346                  return;
347              }
348              try
349              {
350                  _sslStream.BeginWrite(_sendBufferFlush.Data, (int)_sendBufferFlushOffset, (int)(_sendBufferFlush.Size - _sendBufferFlushOffset), ProcessSend, _sslStreamId);
351              }
352              catch (ObjectDisposedException) {}
353          }

355          {

357              {
358                  _sendBufferMain.Clear();
359                  _sendBufferFlush.Clear();
360                  _sendBufferFlushOffset= 0;
361                  BytesPending = 0;
362                  BytesSending = 0;

381          {
382              IsConnecting = false;
383              if (e.SocketError == SocketError.Success)
384              {
385                  if (OptionKeepAlive)
386                      Socket.SetSocketOption(SocketOptionLevel.Socket, SocketOptionName.KeepAlive, true);
387                  if (OptionTcpKeepAliveTime >= 0)
388                      Socket.SetSocketOption(SocketOptionLevel.Tcp, SocketOptionName.TcpKeepAliveTime, OptionTcpKeepAliveTime);
389                  if (OptionTcpKeepAliveInterval >= 0)
390                      Socket.SetSocketOption(SocketOptionLevel.Tcp, SocketOptionName.TcpKeepAliveInterval, OptionTcpKeepAliveInterval);
391                  if (OptionTcpKeepAliveRetryCount >= 0)
392                      Socket.SetSocketOption(SocketOptionLevel.Tcp, SocketOptionName.TcpKeepAliveRetryCount, OptionTcpKeepAliveRetryCount);
393                  if (OptionNoDelay)
394                      Socket.SetSocketOption(SocketOptionLevel.Tcp, SocketOptionName.NoDelay, true);
395                  _receiveBuffer.Reserve(OptionReceiveBufferSize);
396                  _sendBufferMain.Reserve(OptionSendBufferSize);
397                  _sendBufferFlush.Reserve(OptionSendBufferSize);
398                  BytesPending = 0;
399                  BytesSending = 0;
400                  BytesSent = 0;
401                  BytesReceived = 0;
402                  IsConnected = true;
403                  OnConnected();
404                  try
405                  {
406                      _sslStreamId = Guid.NewGuid();

408                      OnHandshaking();
409                      IsHandshaking = true;
410                      if (Context.Certificates != null)
411                          _sslStream.BeginAuthenticateAsClient(Address, Context.Certificates, Context.Protocols, true, ProcessHandshake, _sslStreamId);
412                      else if (Context.Certificate != null)
413                          _sslStream.BeginAuthenticateAsClient(Address, new X509CertificateCollection(new[] { Context.Certificate }), Context.Protocols, true, ProcessHandshake, _sslStreamId);
414                      else
415                          _sslStream.BeginAuthenticateAsClient(Address, ProcessHandshake, _sslStreamId);
416                  }
417                  catch (Exception)

422              }

429          private void ProcessHandshake(IAsyncResult result)
430          {
431              try
432              {
433                  IsHandshaking = false;
434                  if (IsHandshaked)
435                      return;
436                  var sslStreamId = result.AsyncState as Guid?;
437                  if (_sslStreamId != sslStreamId)
438                      return;
439                  _sslStream.EndAuthenticateAsClient(result);
440                  IsHandshaked = true;
441                  TryReceive();
442                  if (IsSocketDisposed)
443                      return;
444                  OnHandshaked();
445                  if (_sendBufferMain.IsEmpty)
446                      OnEmpty();
447              }
448              catch (Exception)
449              {

451                  DisconnectAsync();
452              }

454          private void ProcessReceive(IAsyncResult result)
455          {
456              try
457              {
458                  if (!IsHandshaked)
459                      return;
460                  var sslStreamId = result.AsyncState as Guid?;
461                  if (_sslStreamId != sslStreamId)
462                      return;
463                  long size = _sslStream.EndRead(result);
464                  if (size > 0)
465                  {
466                      BytesReceived += size;
467                      OnReceived(_receiveBuffer.Data, 0, size);
468                      if (_receiveBuffer.Capacity == size)
469                      {
470                          if (((2 * size) > OptionReceiveBufferLimit) && (OptionReceiveBufferLimit > 0))
471                          {
472                              SendError(SocketError.NoBufferSpaceAvailable);
473                              DisconnectAsync();
474                              return;
475                          }
476                          _receiveBuffer.Reserve(2 * size);
477                      }
478                  }
479                  _receiving = false;
480                  if (size > 0)
481                  {
482                      if (!result.CompletedSynchronously)
483                          TryReceive();
484                  }
485                  else
486                      DisconnectAsync();
487              }
488              catch (Exception)
489              {
490                  SendError(SocketError.OperationAborted);
491                  DisconnectAsync();
492              }
493          }
494          private void ProcessSend(IAsyncResult result)
495          {
496              try
497              {
498                  if (!IsHandshaked)
499                      return;
500                  var sslStreamId = result.AsyncState as Guid?;
501                  if (_sslStreamId != sslStreamId)
502                      return;
503                  _sslStream.EndWrite(result);
504                  long size = _sendBufferFlush.Size;
505                  if (size > 0)
506                  {
507                      BytesSending -= size;
508                      BytesSent += size;
509                      _sendBufferFlushOffset += size;
510                      if (_sendBufferFlushOffset == _sendBufferFlush.Size)
511                      {
512                          _sendBufferFlush.Clear();
513                          _sendBufferFlushOffset = 0;
514                      }
515                      OnSent(size, BytesPending + BytesSending);
516                  }
517                  TrySend();
518              }
519              catch (Exception)
520              {
521                  SendError(SocketError.OperationAborted);
522                  DisconnectAsync();
523              }
