347 static int
348 vips_morph_gen_vector( VipsRegion *or, 
349 	void *vseq, void *a, void *b, gboolean *stop )
350 {
351 	VipsMorphSequence *seq = (VipsMorphSequence *) vseq;
352 	VipsMorph *morph = (VipsMorph *) b;
353 	VipsImage *M = morph->M;
354 	VipsRegion *ir = seq->ir;
355 	VipsRect *r = &or->valid;
356 	int sz = VIPS_REGION_N_ELEMENTS( or );
357 	VipsRect s;
358 	int y, j;
359 	VipsExecutor executor[MAX_PASS];
360 	s = *r;
361 	s.width += M->Xsize - 1;
362 	s.height += M->Ysize - 1;
363 	if( vips_region_prepare( ir, &s ) )
364 		return( -1 );
365 #ifdef DEBUG_VERBOSE
366 	printf( "vips_morph_gen_vector: preparing %dx%d@%dx%d pixels\n", 
367 		s.width, s.height, s.left, s.top );
368 	for( j = 0; j < morph->n_pass; j++ ) 
369 		vips_executor_set_program( &executor[j], 
370 			morph->pass[j].vector, sz );
371 	VIPS_GATE_START( "vips_morph_gen_vector: work" ); 
372 	for( y = 0; y < r->height; y++ ) { 
373 		for( j = 0; j < morph->n_pass; j++ ) {
374 			void *d;
375 			if( j == morph->n_pass - 1 )
376 				d = VIPS_REGION_ADDR( or, r->left, r->top + y );
377 			else 
378 				d = seq->t2;
379 			vips_executor_set_scanline( &executor[j], 
380 				ir, r->left, r->top + y );
381 			vips_executor_set_array( &executor[j],
382 				morph->pass[j].r, seq->t1 );
383 			vips_executor_set_destination( &executor[j], d );
384 			vips_executor_run( &executor[j] );
385 			VIPS_SWAP( void *, seq->t1, seq->t2 );
386 		}
387 	}
388 	VIPS_GATE_STOP( "vips_morph_gen_vector: work" ); 
389 	VIPS_COUNT_PIXELS( or, "vips_morph_gen_vector" ); 
390 	return( 0 );

392 static int
393 vips_morph_build( VipsObject *object )
394 {
395 	VipsObjectClass *class = VIPS_OBJECT_GET_CLASS( object );
396 	VipsMorphology *morphology = (VipsMorphology *) object;
397 	VipsMorph *morph = (VipsMorph *) object;
398 	VipsImage **t = (VipsImage **) vips_object_local_array( object, 5 );
399 	VipsImage *in;
400 	VipsImage *M;
401 	VipsGenerateFn generate;
402 	double *coeff;
403 	int i;
404 	if( VIPS_OBJECT_CLASS( vips_morph_parent_class )->build( object ) )
405 		return( -1 );
406 	in = morphology->in;
407 	if( vips_image_decode( in, &t[0] ) )
408 		return( -1 );
409 	in = t[0];
410 	if( vips_check_matrix( class->nickname, morph->mask, &t[1] ) )
411 		return( -1 ); 
412 	morph->M = M = t[1];
413 	morph->n_point = M->Xsize * M->Ysize;
414 	if( vips_embed( in, &t[2], 
415 		M->Xsize / 2, M->Ysize / 2, 
416 		in->Xsize + M->Xsize - 1, in->Ysize + M->Ysize - 1,
417 		"extend", VIPS_EXTEND_COPY,
418 		NULL ) )
419 		return( -1 );
420 	in = t[2]; 
421 	if( vips_cast( in, &t[3], VIPS_FORMAT_UCHAR, NULL ) )
422 		return( -1 );
423 	in = t[3];
424 	if( vips__image_intize( M, &t[4] ) )
425 		return( -1 ); 
426 	M = t[4];
427 	coeff = VIPS_MATRIX( M, 0, 0 ); 
428 	if( !(morph->coeff = VIPS_ARRAY( object, morph->n_point, int )) )
429 		return( -1 );
430 	for( i = 0; i < morph->n_point; i++ ) {
431 		if( coeff[i] != 0 && 
432 			coeff[i] != 128 &&
433 			coeff[i] != 255 ) {
434 			vips_error( class->nickname, 
435 				_( "bad mask element (%f "
436 				"should be 0, 128 or 255)" ),
437 				coeff[i] );
438 			return( -1 );
439 		}
440 		morph->coeff[i] = coeff[i];
441 	}
442 	generate = morph->morph == VIPS_OPERATION_MORPHOLOGY_DILATE
443 		? vips_dilate_gen : vips_erode_gen;
444 	if( vips_vector_isenabled() ) {
445 		if( !vips_morph_compile( morph ) ) {
446 			generate = vips_morph_gen_vector;
447 			g_info( "morph: using vector path" ); 
448 		}
449 		else
450 			vips_morph_compile_free( morph );
451 	}
452 	g_object_set( morph, "out", vips_image_new(), NULL ); 
453 	if( vips_image_pipelinev( morph->out, 
454 		VIPS_DEMAND_STYLE_SMALLTILE, in, NULL ) )
455 		return( -1 );
456 	morph->out->Xsize -= M->Xsize - 1;
457 	morph->out->Ysize -= M->Ysize - 1;
458 	if( vips_image_generate( morph->out, 
459 		vips_morph_start, generate, vips_morph_stop, in, morph ) )
460 		return( -1 );
461 	morph->out->Xoffset = -M->Xsize / 2;
462 	morph->out->Yoffset = -M->Ysize / 2;
463 	vips_reorder_margin_hint( morph->out, morph->n_point );
464 	return( 0 );
465 }
