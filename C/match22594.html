	VipsDrawPoint draw_point, void *client )
{

	int x, y, err;
	dx = x2 - x1;
	dy = y2 - y1;
	if( abs( dx ) >= abs( dy ) && 
		dx < 0 ) {
		VIPS_SWAP( int, x1, x2 );
		VIPS_SWAP( int, y1, y2 );
	}
	else if( abs( dx ) < abs( dy ) && 
		dy < 0 ) {
		VIPS_SWAP( int, x1, x2 );
		VIPS_SWAP( int, y1, y2 );
	}

	dy = y2 - y1;
	x = x1; 
	y = y1;
	if( dx == 0 && 
		dy == 0 ) 
		draw_point( image, x, y, client );
	else if( dx == 0 ) {
		for( ; y <= y2; y++ ) 
			draw_point( image, x, y, client );
	}
	else if( dy == 0 ) {
		for( ; x <= x2; x++ )
			draw_point( image, x, y, client );
	}

		dy > 0 ) {

	}

		dy < 0 ) {
		for( ; x <= x2; x++, y-- )

	}
	else if( abs( dy ) < abs( dx ) && 
		dy > 0 ) {
		for( err = 0; x <= x2; x++ ) {
			draw_point( image, x, y, client );
			err += dy;

			}
		}
	}
