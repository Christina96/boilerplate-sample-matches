420 static int
421 vips_gravity_build( VipsObject *object )
422 {
423 	VipsEmbedBase *base = (VipsEmbedBase *) object;
424 	VipsGravity *gravity = (VipsGravity *) object;
425 	if( vips_object_argument_isset( object, "in" ) &&
426 		vips_object_argument_isset( object, "width" ) &&
427 		vips_object_argument_isset( object, "height" ) &&
428 		vips_object_argument_isset( object, "direction" ) ) {
429 		switch( gravity->direction ) {
430 		case VIPS_COMPASS_DIRECTION_CENTRE:
431 			base->x = (base->width - base->in->Xsize) / 2;
432 			base->y = (base->height - base->in->Ysize) / 2;
433 			break;
434 		case VIPS_COMPASS_DIRECTION_NORTH:
435 			base->x = (base->width - base->in->Xsize) / 2;
436 			base->y = 0;
437 			break;
438 		case VIPS_COMPASS_DIRECTION_EAST:
439 			base->x = base->width - base->in->Xsize;
440 			base->y = (base->height - base->in->Ysize) / 2;
441 			break;
442 		case VIPS_COMPASS_DIRECTION_SOUTH:
443 			base->x = (base->width - base->in->Xsize) / 2;
444 			base->y = base->height - base->in->Ysize;
445 			break;
446 		case VIPS_COMPASS_DIRECTION_WEST:
447 			base->x = 0;
448 			base->y = (base->height - base->in->Ysize) / 2;
449 			break;
450 		case VIPS_COMPASS_DIRECTION_NORTH_EAST:
451 			base->x = base->width - base->in->Xsize;
452 			base->y = 0;
453 			break;
454 		case VIPS_COMPASS_DIRECTION_SOUTH_EAST:
455 			base->x = base->width - base->in->Xsize;
456 			base->y = base->height - base->in->Ysize;
457 			break;
458 		case VIPS_COMPASS_DIRECTION_SOUTH_WEST:
459 			base->x = 0;
460 			base->y = base->height - base->in->Ysize;
461 			break;
462 		case VIPS_COMPASS_DIRECTION_NORTH_WEST:
463 			base->x = 0;
464 			base->y = 0;
465 			break;
466 		default:
467 			g_assert_not_reached();
468 		}
469 	}
470 	if( VIPS_OBJECT_CLASS( vips_gravity_parent_class )->build( object ) )
471 		return( -1 );
472 	return( 0 );
473 }
