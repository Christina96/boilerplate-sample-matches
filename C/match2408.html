524 vips_convasep_pass( VipsConvasep *convasep, 
525 	VipsImage *in, VipsImage **out, VipsDirection direction )
526 {
527 	VipsObjectClass *class = VIPS_OBJECT_GET_CLASS( convasep );
528 	VipsGenerateFn gen;
529 	*out = vips_image_new(); 
530 	if( vips_image_pipelinev( *out, 
531 		VIPS_DEMAND_STYLE_SMALLTILE, in, NULL ) )
532 		return( -1 );
533 	if( direction == VIPS_DIRECTION_HORIZONTAL ) { 
534 		(*out)->Xsize -= convasep->width - 1;
535 		gen = vips_convasep_generate_horizontal;
536 	}
537 	else {

546 	}

548 		vips_convasep_start, gen, vips_convasep_stop, in, convasep ) )
549 		return( -1 );

552 static int 
553 vips_convasep_build( VipsObject *object )
554 {
555 	VipsObjectClass *class = VIPS_OBJECT_GET_CLASS( object );
556 	VipsConvolution *convolution = (VipsConvolution *) object;
557 	VipsConvasep *convasep = (VipsConvasep *) object;
558 	VipsImage **t = (VipsImage **) vips_object_local_array( object, 4 );
559 	VipsImage *in;
560 	if( VIPS_OBJECT_CLASS( vips_convasep_parent_class )->build( object ) )
561 		return( -1 );
562 	if( vips_check_separable( class->nickname, convolution->M ) ) 
563                 return( -1 );
564 	if( vips__image_intize( convolution->M, &t[3] ) )
565 		return( -1 );
566 	convasep->iM = t[3]; 
567 	convasep->width = convasep->iM->Xsize * convasep->iM->Ysize;
568 	in = convolution->in;
569 	if( vips_convasep_decompose( convasep ) )
570 		return( -1 ); 
571 	g_object_set( convasep, "out", vips_image_new(), NULL ); 
572 	if( 
573 		vips_embed( in, &t[0], 
574 			convasep->width / 2, 
575 			convasep->width / 2, 
576 			in->Xsize + convasep->width - 1, 
577 			in->Ysize + convasep->width - 1,
578 			"extend", VIPS_EXTEND_COPY,
579 			NULL ) ||
580 		vips_convasep_pass( convasep, 
581 			t[0], &t[1], VIPS_DIRECTION_HORIZONTAL ) ||
582 		vips_convasep_pass( convasep, 
583 			t[1], &t[2], VIPS_DIRECTION_VERTICAL ) ||
584 		vips_image_write( t[2], convolution->out ) )
585 		return( -1 );
586 	convolution->out->Xoffset = 0;
587 	convolution->out->Yoffset = 0;
588 	vips_reorder_margin_hint( convolution->out,
589 		convolution->M->Xsize * convolution->M->Ysize );
590 
	return( 0 );
