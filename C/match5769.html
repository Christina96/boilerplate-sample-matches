48 	int *width, int *height, double *scale, double *offset )   
49 {
50 	double header[4];
51 	char *p, *q;
52 	int i;
53 	for( i = 0, p = line; 
54 		(q = vips_break_token( p, " \t" )) &&
55 			i < 4; 
56 		i++, p = q )
57 		if( vips_strtod( p, &header[i] ) ) {
58 			vips_error( "matload",

60 			return( -1 );
61 		}

68 		return( -1 );
69 	}

71 		VIPS_FLOOR( header[1] ) != header[1] ) {
72 		vips_error( "mask2vips", "%s", _( "width / height not int" ) );
73 		return( -1 );
74 	}
75 	*width = header[0];
76 	*height = header[1];
77 	if( *width <= 0 || 
78 		*width > 100000 ||
79 		*height <= 0 || 
80 		*height > 100000 ) { 
81 		vips_error( "mask2vips", 
82 			"%s", _( "width / height out of range" ) );
