116 	missing = 0;
117 	for( y = oreg->top; y < VIPS_RECT_BOTTOM( oreg ); y++ ) {
118 		const int j = y - ovlap->overlap.top;
119 		const int first = ovlap->first[j];
120 		if( first < 0 ) {
121 			missing = 1;
122 			break;
123 		}
124 	}
125 	if( !missing ) {
126 		g_mutex_unlock( ovlap->fl_lock );
127 		return( 0 );
128 	}
129 	rr.left = ovlap->overlap.left;
130 	rr.top = oreg->top;
131 	rr.width = ovlap->overlap.width;
132 	rr.height = oreg->height;
133 	rr.left -= ovlap->rarea.left;
134 	rr.top -= ovlap->rarea.top;
135 	sr.left = ovlap->overlap.left;
136 	sr.top = oreg->top;
137 	sr.width = ovlap->overlap.width;
138 	sr.height = oreg->height;
139 	sr.left -= ovlap->sarea.left;
140 	sr.top -= ovlap->sarea.top;
141 #ifdef DEBUG
142 	printf( "lrmerge: making first/last for areas:\n" );
143 	printf( "ref: left = %d, top = %d, width = %d, height = %d\n",
144 		rr.left, rr.top, rr.width, rr.height );
145 	printf( "sec: left = %d, top = %d, width = %d, height = %d\n",
146 		sr.left, sr.top, sr.width, sr.height );
147 #endif
148 	if( vips_region_prepare( rir, &rr ) ||
149 		vips_region_prepare( sir, &sr ) ) {
150 		g_mutex_unlock( ovlap->fl_lock );
151 		return( -1 );
152 	}
153 	for( y = oreg->top, yr = rr.top, ys = sr.top; 
154 		y < VIPS_RECT_BOTTOM( oreg ); y++, yr++, ys++ ) {
155 		const int j = y - ovlap->overlap.top;
156 		int *first = &ovlap->first[j];
157 		int *last = &ovlap->last[j];
158 		if( *first < 0 ) {
159 			if( find_first( sir, first, 
160 				sr.left, ys, sr.width ) ||
161 				find_last( rir, last, 
162 					rr.left, yr, rr.width ) ) {
163 				g_mutex_unlock( ovlap->fl_lock );
164 				return( -1 );
165 			}
166 			*first += ovlap->sarea.left;
167 			*last += ovlap->rarea.left;
168 			if( ovlap->mwidth >= 0 && 
169 				*last - *first > ovlap->mwidth ) {
170 				int shrinkby = (*last - *first) - ovlap->mwidth;
171 				*first += shrinkby / 2;
172 				*last -= shrinkby / 2;
173 			}
174 		}
175 	}
176 	g_mutex_unlock( ovlap->fl_lock );
177 	return( 0 );
178 }
179 #define TEST_ZERO( TYPE, T, RESULT ) { \
180 	TYPE *tt = (T); \
181 	int ii; \
182 	\
183 	for( ii = 0; ii < cb; ii++ ) \
184 		if( tt[i + ii] ) \
185 			break; \
186 	if( ii == cb )  \
187 		(RESULT) = 1; \
188 }
189 #define iblend( TYPE, B, IN1, IN2, OUT ) { \
190 	TYPE *tr = (TYPE *) (IN1); \
191 	TYPE *ts = (TYPE *) (IN2); \
192 	TYPE *tq = (TYPE *) (OUT); \
193 	const int cb = (B); \
194 	const int left = VIPS_CLIP( 0, first - oreg->left, oreg->width ); \
195 	const int right = VIPS_CLIP( left, last - oreg->left, oreg->width ); \
196 	int ref_zero; \
197 	int sec_zero; \
198 	int x, b; \
199 	int i; \
200 	\
201 	for( i = 0, x = 0; x < left; x++ ) { \
202 		ref_zero = 0; \
203 		TEST_ZERO( TYPE, tr, ref_zero ); \
204 		if( !ref_zero ) \
205 			for( b = 0; b < cb; b++, i++ ) \
206 				tq[i] = tr[i]; \
207 		else \
208 			for( b = 0; b < cb; b++, i++ ) \
209 				tq[i] = ts[i]; \
210 	} \
211 	\
212 	for( x = left; x < right; x++ ) { \
213 		ref_zero = 0; \
214 		sec_zero = 0; \
215 		TEST_ZERO( TYPE, tr, ref_zero ); \
216 		TEST_ZERO( TYPE, ts, sec_zero ); \
217 		\
218 		if( !ref_zero && !sec_zero ) { \
219 			int inx = ((x + oreg->left - first) <<  \
220 				BLEND_SHIFT) / bwidth; \
221 			int c1 = vips__icoef1[inx]; \
222 			int c2 = vips__icoef2[inx]; \
223 			\
224 			for( b = 0; b < cb; b++, i++ ) \
225 				tq[i] = c1 * tr[i] / BLEND_SCALE + \
226 					c2 * ts[i] / BLEND_SCALE; \
227 		} \
228 		else if( !ref_zero ) \
229 			for( b = 0; b < cb; b++, i++ ) \
230 				tq[i] = tr[i]; \
231 		else \
232 			for( b = 0; b < cb; b++, i++ ) \
233 				tq[i] = ts[i]; \
234 	} \
235 	\
236 	for( x = right; x < oreg->width; x++ ) { \
237 		sec_zero = 0; \
238 		TEST_ZERO( TYPE, ts, sec_zero ); \
239 		if( !sec_zero ) \
240 			for( b = 0; b < cb; b++, i++ )  \
241 				tq[i] = ts[i]; \
242 		else \
243 			for( b = 0; b < cb; b++, i++ )  \
244 				tq[i] = tr[i]; \
245 	} \
246 }

256 	int x, b; \

258 	\

263 			for( b = 0; b < cb; b++, i++ ) \

265 		else \

268 	} \
269 	\
270 	for( x = left; x < right; x++ ) { \
271 		ref_zero = 0; \

273 		TEST_ZERO( TYPE, tr, ref_zero ); \
274 		TEST_ZERO( TYPE, ts, sec_zero ); \
275 		\
276 		if( !ref_zero && !sec_zero ) { \
277 			int inx = ((x + oreg->left - first) <<  \
278 				BLEND_SHIFT) / bwidth; \
279 			double c1 = vips__coef1[inx];  \
280 			double c2 = vips__coef2[inx];  \
281 			\
282 			for( b = 0; b < cb; b++, i++ ) \
283 				tq[i] = c1 * tr[i] + c2 * ts[i]; \
284 		} \
285 		else if( !ref_zero ) \
286 			for( b = 0; b < cb; b++, i++ ) \
287 				tq[i] = tr[i]; \
288 		else \
289 			for( b = 0; b < cb; b++, i++ ) \
290 				tq[i] = ts[i]; \
291 	} \
292 	\
293 	for( x = right; x < oreg->width; x++ ) { \
294 		sec_zero = 0; \
295 		TEST_ZERO( TYPE, ts, sec_zero ); \
296 		if( !sec_zero ) \
297 			for( b = 0; b < cb; b++, i++ )  \
298 				tq[i] = ts[i]; \
299 		else \
300 			for( b = 0; b < cb; b++, i++ )  \
301 				tq[i] = tr[i]; \
302 	} \

304 static int
305 lr_blend( VipsRegion *or, MergeInfo *inf, Overlapping *ovlap, VipsRect *oreg )
306 {
307 	VipsRegion *rir = inf->rir;
308 	VipsRegion *sir = inf->sir;
309 	VipsImage *im = or->im;
310 	VipsRect prr, psr;
311 	int y, yr, ys;
312 	if( make_firstlast( inf, ovlap, oreg ) )
313 		return( -1 );
314 	prr = *oreg;
315 	prr.left -= ovlap->rarea.left;
316 	prr.top -= ovlap->rarea.top;
317 	psr = *oreg;
318 	psr.left -= ovlap->sarea.left;
319 	psr.top -= ovlap->sarea.top;
320 	if( vips_region_prepare( rir, &prr ) || 
321 		vips_region_prepare( sir, &psr ) )
322 		return( -1 );
323 	for( y = oreg->top, yr = prr.top, ys = psr.top; 
324 		y < VIPS_RECT_BOTTOM( oreg ); y++, yr++, ys++ ) { 
325 		VipsPel *pr = VIPS_REGION_ADDR( rir, prr.left, yr );
326 		VipsPel *ps = VIPS_REGION_ADDR( sir, psr.left, ys );
327 		VipsPel *q = VIPS_REGION_ADDR( or, oreg->left, y );
328 		const int j = y - ovlap->overlap.top;
329 		const int first = ovlap->first[j];
330 		const int last = ovlap->last[j];
331 		const int bwidth = last - first;
332 		switch( im->BandFmt ) {
333 		case VIPS_FORMAT_UCHAR: 	
334 			iblend( unsigned char, im->Bands, pr, ps, q ); break; 
335 		case VIPS_FORMAT_CHAR: 	
336 			iblend( signed char, im->Bands, pr, ps, q ); break; 
337 		case VIPS_FORMAT_USHORT: 
338 			iblend( unsigned short, im->Bands, pr, ps, q ); break; 
339 		case VIPS_FORMAT_SHORT: 	
340 			iblend( signed short, im->Bands, pr, ps, q ); break; 
341 		case VIPS_FORMAT_UINT: 	
342 			iblend( unsigned int, im->Bands, pr, ps, q ); break; 
343 		case VIPS_FORMAT_INT: 	
344 			iblend( signed int, im->Bands, pr, ps, q );  break; 
345 		case VIPS_FORMAT_FLOAT: 	
346 			fblend( float, im->Bands, pr, ps, q ); break; 
347 		case VIPS_FORMAT_DOUBLE:	
348 			fblend( double, im->Bands, pr, ps, q ); break; 
349 		case VIPS_FORMAT_COMPLEX:
350 			fblend( float, im->Bands * 2, pr, ps, q ); break; 
351 		case VIPS_FORMAT_DPCOMPLEX:
352 			fblend( double, im->Bands * 2, pr, ps, q ); break;
353 		default:
354 			g_assert_not_reached();
355 			return( -1 );
356 		}
357 	}
358 	return( 0 );
359 }

403 Overlapping *
404 vips__build_mergestate( const char *domain,
405 	VipsImage *ref, VipsImage *sec, VipsImage *out, 
406 	int dx, int dy, int mwidth )
407 {
408 	VipsImage **t = (VipsImage **)
409 		vips_object_local_array( VIPS_OBJECT( out ), 4 );
410 	VipsImage **arry;
411    	Overlapping *ovlap;
412 	int x;
413 	if( vips_image_pio_input( ref ) ||
414 		vips_image_pio_input( sec ) ||
415 		vips_check_bands_1orn( domain, ref, sec ) ||
416 		vips_check_coding_known( domain, ref ) ||
417 		vips_check_coding_same( domain, ref, sec ) )
418 		return( NULL );
419 	if( vips__formatalike( ref, sec, &t[0], &t[1] ) ||
420 		vips__bandalike( domain, t[0], t[1], &t[2], &t[3] ) )
421 		return( NULL );
422 	if( !(arry = vips_allocate_input_array( out,
423 		t[2], t[3], NULL )) )
424 		return( NULL );
425 	if( vips_image_pipeline_array( out,
426 		VIPS_DEMAND_STYLE_SMALLTILE, arry ) )
427 		return( NULL );
428 	if( mwidth < -1 ) {
429 		vips_error( domain, "%s", _( "mwidth must be -1 or >= 0" ) );
430 		return( NULL );
431 	}
432 	if( !(ovlap = VIPS_NEW( out, Overlapping )) )
433 		return( NULL );
434 
	ovlap->ref = arry[0];
435 	ovlap->sec = arry[1];
436 	ovlap->out = out;
437 	ovlap->dx = dx;
438 	ovlap->dy = dy;
439 	ovlap->mwidth = mwidth;
440    	ovlap->rarea.left = 0;
441    	ovlap->rarea.top = 0;
442    	ovlap->rarea.width = ovlap->ref->Xsize;
443    	ovlap->rarea.height = ovlap->ref->Ysize;
444    	ovlap->sarea.left = -dx;
445    	ovlap->sarea.top = -dy;
446    	ovlap->sarea.width = ovlap->sec->Xsize;
447    	ovlap->sarea.height = ovlap->sec->Ysize;
448 	vips_rect_intersectrect( &ovlap->rarea, &ovlap->sarea, &ovlap->overlap );
449 	if( vips_rect_isempty( &ovlap->overlap ) ) {
450 		vips_error( domain, "%s", _( "no overlap" ) );
451 		return( NULL );
452 	}
453 	vips_rect_unionrect( &ovlap->rarea, &ovlap->sarea, &ovlap->oarea );
454 	ovlap->rarea.left -= ovlap->oarea.left;
455 	ovlap->rarea.top -= ovlap->oarea.top;
456 	ovlap->sarea.left -= ovlap->oarea.left;
457 	ovlap->sarea.top -= ovlap->oarea.top;
458 	ovlap->overlap.left -= ovlap->oarea.left;
459 	ovlap->overlap.top -= ovlap->oarea.top;
460 	ovlap->oarea.left = 0;
461 	ovlap->oarea.top = 0;
462 	vips__make_blend_luts();
463 	ovlap->flsize = VIPS_MAX( ovlap->overlap.width, ovlap->overlap.height );
464 	ovlap->first = VIPS_ARRAY( out, ovlap->flsize, int );
465 	ovlap->last = VIPS_ARRAY( out, ovlap->flsize, int );
466 	if( !ovlap->first || !ovlap->last ) 
467 		return( NULL ); 
468 	for( x = 0; x < ovlap->flsize; x++ )
469 		ovlap->first[x] = -1;
470 	ovlap->fl_lock = vips_g_mutex_new();
471 	g_signal_connect( out, "close",
472 		G_CALLBACK( lock_free ), ovlap->fl_lock );
473 	return( ovlap );
474 }
475 static Overlapping *
476 build_lrstate( VipsImage *ref, VipsImage *sec, VipsImage *out, 
477 	int dx, int dy, int mwidth )
478 {
479    	Overlapping *ovlap;
480 	if( !(ovlap = vips__build_mergestate( "lrmerge", 
481 		ref, sec, out, dx, dy, mwidth )) )
482 		return( NULL );
483 	switch( ovlap->ref->Coding ) {
484 	case VIPS_CODING_LABQ:
485 		ovlap->blend = lr_blend_labpack;
486 		break;
487 	case VIPS_CODING_NONE:
488 		ovlap->blend = lr_blend;
489 		break;
490 	default:
491 		vips_error( "lrmerge", "%s", _( "unknown coding type" ) );
492 		return( NULL );
493 	}
494 	ovlap->rpart = ovlap->rarea;
495 	ovlap->spart = ovlap->sarea;
496 	ovlap->rpart.width -= ovlap->overlap.width;
497 	ovlap->spart.left += ovlap->overlap.width;
498 	ovlap->spart.width -= ovlap->overlap.width;
499 	if( VIPS_RECT_RIGHT( &ovlap->rarea ) > 
500 		VIPS_RECT_RIGHT( &ovlap->sarea ) ||
501 		ovlap->rarea.left > ovlap->sarea.left ) {
502 		vips_error( "lrmerge", "%s", _( "too much overlap" ) );
503 		return( NULL );
504 	}
505 	ovlap->blsize = ovlap->overlap.width;
506 	return( ovlap );
507 }

531 int
532 vips__merge_gen( VipsRegion *or, void *seq, void *a, void *b, 
533 	gboolean *stop )
534 {
535 	MergeInfo *inf = (MergeInfo *) seq;
536 	Overlapping *ovlap = (Overlapping *) a;
537 	VipsRect *r = &or->valid;
538 	VipsRect rreg, sreg, oreg;
539 	vips_rect_intersectrect( r, &ovlap->rpart, &rreg );
540 	vips_rect_intersectrect( r, &ovlap->spart, &sreg );
541 	if( vips_rect_equalsrect( r, &rreg ) ) {
542 		if( vips__attach_input( or, inf->rir, &ovlap->rarea ) )
543 			return( -1 );
544 	}
545 	else if( vips_rect_equalsrect( r, &sreg ) ) {
546 		if( vips__attach_input( or, inf->sir, &ovlap->sarea ) )
547 			return( -1 );
548 	}
549 	else {
550 		vips_rect_intersectrect( r, &ovlap->rarea, &rreg );
551 		vips_rect_intersectrect( r, &ovlap->sarea, &sreg );
552 		vips_rect_intersectrect( r, &ovlap->overlap, &oreg );
553 		vips_region_black( or );
554 		if( !vips_rect_isempty( &rreg ) ) 
555 			if( vips__copy_input( or, 
556 				inf->rir, &ovlap->rarea, &rreg ) )
557 				return( -1 );
558 		if( !vips_rect_isempty( &sreg ) )
559 			if( vips__copy_input( or, 
560 				inf->sir, &ovlap->sarea, &sreg ) )
561 				return( -1 );
562 		inf->rir->valid.width = inf->sir->valid.width = 0;
563 		if( !vips_rect_isempty( &oreg ) )
564 			if( ovlap->blend( or, inf, ovlap, &oreg ) )
565 				return( -1 );
566 	}
567 	return( 0 );
568 }

610 int
611 vips__lrmerge( VipsImage *ref, VipsImage *sec, VipsImage *out, 
612 	int dx, int dy, int mwidth )
613 {
614 	Overlapping *ovlap;
615 #ifdef DEBUG
616 	printf( "lrmerge %s %s %s %d %d %d\n", 
617 		ref->filename, sec->filename, out->filename, 
618 		dx, dy, mwidth );
619 	printf( "ref is %d x %d pixels\n", ref->Xsize, ref->Ysize );
620 	printf( "sec is %d x %d pixels\n", sec->Xsize, sec->Ysize );
621 #endif
622 	if( dx > 0 || dx < 1 - ref->Xsize ) {
623 		VipsImage *x;
624 #ifdef DEBUG
625 		printf( "lrmerge: no overlap, using insert\n" ); 
626 #endif
