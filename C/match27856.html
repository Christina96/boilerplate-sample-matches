703 static void 
704 encode_lab( float *lab, guint16 *fixed, int n )
705 {
706 	int i;
707 	for( i = 0; i < n; i++ ) {
708 		float L = lab[0];
709 		float a = lab[1];
710 		float b = lab[2];
711 		if( L < 0 ) 
712 			L = 0;
713 		if( L > 100. ) 
714 			L = 100.;
715 		if( a < -128. ) 
716 			a = -128;
717 		if( a > 127.9961 ) 
718 			a = 127.9961;
719 		if( b < -128. ) 
720 			b = -128;
721 		if( b > 127.9961 ) 
722 			b = 127.9961;
723 		fixed[0] = L *  652.800 + 0.5;
724 		fixed[1] = (a + 128.0) * 256.0 + 0.5;
725 		fixed[2] = (b + 128.0) * 256.0 + 0.5;
726 		lab += 3;
727 		fixed += 3;
728 	}
729 }
730 #define MAX_ENCODEABLE_XYZ  (100 * (1.0 + 32767.0 / 32768.0))
731 static void 
732 encode_xyz( float *xyz, guint16 *fixed, int n )
733 {
734 	int i;
735 	for( i = 0; i < n; i++ ) {
736 		float X = xyz[0];
737 		float Y = xyz[1];
738 		float Z = xyz[2];
739 		if( X < 0 ) 
740 			X = 0;
741 		if( X > MAX_ENCODEABLE_XYZ ) 
742 			X = MAX_ENCODEABLE_XYZ;
743 		if( Y < 0 ) 
744 			Y = 0;
745 		if( Y > MAX_ENCODEABLE_XYZ ) 
746 			Y = MAX_ENCODEABLE_XYZ;
747 		if( Z < 0 ) 
748 			Z = 0;
749 		if( Z > MAX_ENCODEABLE_XYZ ) 
750 			Z = MAX_ENCODEABLE_XYZ;
751 		fixed[0] = X * X_FAC + 0.5;
752 		fixed[1] = Y * Y_FAC + 0.5;
753 		fixed[2] = Z * Z_FAC + 0.5;
754 		xyz += 3;
755 		fixed += 3;
756 	}
