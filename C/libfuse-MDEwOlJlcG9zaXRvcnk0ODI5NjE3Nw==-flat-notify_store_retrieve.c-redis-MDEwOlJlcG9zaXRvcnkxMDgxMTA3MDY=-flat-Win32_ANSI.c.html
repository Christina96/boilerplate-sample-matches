519      {
520          hConOut = hDev;

522          shifted = FALSE;
523      }
524      for (i = nNumberOfBytesToWrite, s = (LPCSTR)lpBuffer; i > 0; i--, s++) {
525          if (state == 1) {
526              if (*s == ESC) state = 2;
527              else if (*s == SO) shifted = TRUE;
528              else if (*s == SI) shifted = FALSE;

530          } else if (state == 2) {
531              if (*s == ESC);

536                  state = 3;

539              } else if (*s == ')' || *s == '(') state = 6;
540              else state = 1;
541          } else if (state == 3) {
542              if (is_digit(*s)) {
543                  es_argc = 0;
544                  es_argv[0] = *s - '0';
545                  state = 4;
546              } else if (*s == ';') {
547                  es_argc = 1;
548                  es_argv[0] = 0;
549                  es_argv[1] = 0;
550                  state = 4;
551              } else if (*s == '?' || *s == '>') {
552                  prefix2 = *s;
553              } else {
554                  es_argc = 0;
555                  suffix = *s;
556                  InterpretEscSeq();
557                  state = 1;
558              }
559          } else if (state == 4) {
560              if (is_digit(*s)) {
561                  es_argv[es_argc] = 10 * es_argv[es_argc] + (*s - '0');
562              } else if (*s == ';') {
563                  if (es_argc < MAX_ARG - 1) es_argc++;
564                  es_argv[es_argc] = 0;
565                  if (prefix == ']')
566                      state = 5;
567              } else {

569                  suffix = *s;

571                  state = 1;
572              }
573          } else if (state == 5) {

575                  Pt_arg[Pt_len] = '\0';
