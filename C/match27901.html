552 #define NBIT_LINE( N, EXPAND ) \
553 static void \
554 rtiff_ ## N ## bit_line( Rtiff *rtiff, \
555 	VipsPel *q, VipsPel *p, int n, void *flg ) \
556 { \
557 	int photometric = rtiff->header.photometric_interpretation; \
558 	int mask = photometric == PHOTOMETRIC_MINISBLACK ?  0 : 0xff; \
559 	int bps = rtiff->header.bits_per_sample; \
560 	int load = 8 / bps - 1; \
561 	\
562 	int x; \
563 	VipsPel bits; \
564 	\
565 	bits = 0; \
566 	\
567 	for( x = 0; x < n; x++ ) { \
568 		if( (x & load) == 0 ) \
569 			bits = *p++ ^ mask; \
570 	 	\
571 		EXPAND( q[x], bits ); \
572 		\
573 		bits <<= bps; \
574 	} \
575 }
576 #define EXPAND1( Q, BITS ) G_STMT_START { \
577 	(Q) = (((signed char) (BITS & 128)) >> 7); \
578 } G_STMT_END
579 #define EXPAND2( Q, BITS ) G_STMT_START { \
580 	VipsPel twobits = BITS >> 6; \
581 	VipsPel fourbits = twobits | (twobits << 2); \
582 	Q = fourbits | (fourbits << 4); \
583 } G_STMT_END
584 #define EXPAND4( Q, BITS ) G_STMT_START { \
585 	Q = (BITS & 0xf0) | (BITS >> 4); \
586 } G_STMT_END
587 NBIT_LINE( 1, EXPAND1 )
588 NBIT_LINE( 2, EXPAND2 )
589 NBIT_LINE( 4, EXPAND4 )

718 static void
719 rtiff_palette_line_bit( Rtiff *rtiff, 
720 	VipsPel *q, VipsPel *p, int n, void *client )
721 {
722 	PaletteRead *read = (PaletteRead *) client;
723 	int samples_per_pixel = rtiff->header.samples_per_pixel;
724 	int bits_per_sample = rtiff->header.bits_per_sample;
725 	int bit;
726 	VipsPel data;
727 	int x;
728 	bit = 0;
729 	data = 0;
730 	for( x = 0; x < n * samples_per_pixel; x++ ) {
731 		int i;
732 		if( bit <= 0 ) {
733 			data = *p++;
734 			bit = 8;
735 		}
736 		i = data >> (8 - bits_per_sample);
737 		data <<= bits_per_sample;
738 		bit -= bits_per_sample;
739 		if( x % samples_per_pixel == 0 ) { 
740 			if( read->mono ) 
741 				*q++ = read->red8[i];
742 			else {
743 				q[0] = read->red8[i];
744 				q[1] = read->green8[i];
745 				q[2] = read->blue8[i];
746 				q += 3;
747 			}
748 		}
749 		else 
750 			*q++ = VIPS_LSHIFT_INT( i, 8 - bits_per_sample );
751 	}
752 }

804 static int
805 rtiff_parse_palette( Rtiff *rtiff, VipsImage *out )
806 {
807 	int samples_per_pixel = rtiff->header.samples_per_pixel;
808 	int bits_per_sample = rtiff->header.bits_per_sample;
809 	int len;
810 	PaletteRead *read;
811 	int i;
812 	if( rtiff_check_bits_palette( rtiff ) ||
813 		rtiff_check_min_samples( rtiff, 1 ) )
814 		return( -1 ); 
815 	len = 1 << bits_per_sample;
816 	if( !(read = VIPS_NEW( out, PaletteRead )) ||
817 		!(read->red8 = VIPS_ARRAY( out, len, VipsPel )) ||
818 		!(read->green8 = VIPS_ARRAY( out, len, VipsPel )) ||
819 		!(read->blue8 = VIPS_ARRAY( out, len, VipsPel )) )
820 		return( -1 );
821 	if( !TIFFGetField( rtiff->tiff, 
822 		TIFFTAG_COLORMAP, 
823 		&read->red16, &read->green16, &read->blue16 ) ) {
824 		vips_error( "tiff2vips", "%s", _( "bad colormap" ) );
825 		return( -1 );
826 	}
827 	for( i = 0; i < len; i++ ) 
828 		if( (read->red16[i] >> 8) | 
829 			(read->green16[i] >> 8) | 
830 			(read->blue16[i] >> 8) )
831 			break;
832 	if( i < len ) 
833 		for( i = 0; i < len; i++ ) {
834 			read->red8[i] = read->red16[i] >> 8;
835 			read->green8[i] = read->green16[i] >> 8;
836 			read->blue8[i] = read->blue16[i] >> 8;
837 		}
838 	else {
839 		g_warning( "%s", _( "assuming 8-bit palette" ) );
840 		for( i = 0; i < len; i++ ) {
841 			read->red8[i] = read->red16[i] & 0xff;
842 			read->green8[i] = read->green16[i] & 0xff;
843 			read->blue8[i] = read->blue16[i] & 0xff;
844 		}
845 	}
846 	read->mono = TRUE;
847 	for( i = 0; i < len; i++ ) 
848 		if( read->red16[i] != read->green16[i] ||
849 			read->green16[i] != read->blue16[i] ) {
850 			read->mono = FALSE;
851 			break;
852 		}
853 	if( bits_per_sample <= 8 )
854 		out->BandFmt = VIPS_FORMAT_UCHAR; 
855 	else
856 		out->BandFmt = VIPS_FORMAT_USHORT; 
857 	out->Coding = VIPS_CODING_NONE; 
858 	if( read->mono ) {
859 		out->Bands = samples_per_pixel; 
860 		if( bits_per_sample <= 8 )
861 			out->Type = VIPS_INTERPRETATION_B_W; 
862 		else
863 			out->Type = VIPS_INTERPRETATION_GREY16; 
864 	}
865 	else {
866 		out->Bands = samples_per_pixel + 2; 
867 		if( bits_per_sample <= 8 )
868 			out->Type = VIPS_INTERPRETATION_sRGB; 
869 		else
870 			out->Type = VIPS_INTERPRETATION_RGB16; 
871 	}
872 	rtiff->client = read;
873 	if( bits_per_sample < 8 )
874 		rtiff->sfn = rtiff_palette_line_bit;
875 	else if( bits_per_sample == 8 )
876 		rtiff->sfn = rtiff_palette_line8;
877 	else if( bits_per_sample == 16 )
878 		rtiff->sfn = rtiff_palette_line16;
879 	else
880 		g_assert_not_reached(); 
881 	return( 0 );
882 }

958 	}
959 	if( photometric_interpretation == PHOTOMETRIC_PALETTE ) 
960 		return( rtiff_parse_palette );

1373 			for( z = 0; z < hit.height; z++ ) {

1376 				p += scanline_size;
