554 gint64
555 vips_source_seek( VipsSource *source, gint64 offset, int whence )
556 {
557 	const char *nick = vips_connection_nick( VIPS_CONNECTION( source ) );
558 	VipsSourceClass *class = VIPS_SOURCE_GET_CLASS( source );
559 	gint64 new_pos;
560 	VIPS_DEBUG_MSG( "vips_source_seek: offset = %" G_GINT64_FORMAT 
561 		", whence = %d\n", offset, whence );
562 	if( vips_source_unminimise( source ) ||
563 		vips_source_test_features( source ) )
564 		return( -1 );
565 	if( source->data ) {
566 		switch( whence ) {
567 		case SEEK_SET:
568 			new_pos = offset;
569 			break;
570 		case SEEK_CUR:
571 			new_pos = source->read_position + offset;
572 			break;
573 		case SEEK_END:
574 			new_pos = source->length + offset;
575 			break;
576 		default:
577 			vips_error( nick, "%s", _( "bad 'whence'" ) );
578 			return( -1 );
579 		}
580 	}
581 	else if( source->is_pipe ) {
582 		switch( whence ) {
583 		case SEEK_SET:
584 			new_pos = offset;
585 			break;
586 		case SEEK_CUR:
587 			new_pos = source->read_position + offset;
588 			break;
589 		case SEEK_END:
590 			if( vips_source_pipe_read_to_position( source, -1 ) )
591 				return( -1 );
592 			new_pos = source->length + offset;
593 			break;
594 		default:
595 			vips_error( nick, "%s", _( "bad 'whence'" ) );
596 			return( -1 );
597 		}
598 	}
599 	else {
600 		if( (new_pos = class->seek( source, offset, whence )) == -1 )
601 			return( -1 );
602 	}
603 	if( source->is_pipe &&
604 		vips_source_pipe_read_to_position( source, new_pos ) )
605 		return( -1 );
606 	if( new_pos < 0 ||
607 		(source->length != -1 && 
608 		 new_pos > source->length) ) {
609 		vips_error( nick, 
610 			_( "bad seek to %" G_GINT64_FORMAT ), new_pos );
611                 return( -1 );
612 	}
613 	source->read_position = new_pos;
614 	VIPS_DEBUG_MSG( "    new_pos = %" G_GINT64_FORMAT "\n", new_pos );
615 	return( new_pos );
616 }

644 gint64
645 vips_source_sniff_at_most( VipsSource *source, 
646 	unsigned char **data, size_t length )
647 {
648 	unsigned char *q;
649 	gint64 read_position;
650 	VIPS_DEBUG_MSG( "vips_source_sniff_at_most: %zd bytes\n", length );
651 	SANITY( source );
652 	if( vips_source_test_features( source ) ||
653 		vips_source_rewind( source ) )
654 		return( -1 );
655 	g_byte_array_set_size( source->sniff, length );
656 	read_position = 0; 
657 	q = source->sniff->data;
658 	while( read_position < length ) {
659 		gint64 bytes_read;
660 		bytes_read = vips_source_read( source, q, 
661 			length - read_position );
662 		if( bytes_read == -1 )
663 			return( -1 );
664 		if( bytes_read == 0 )
665 			break;
666 		read_position += bytes_read;
667 		q += bytes_read;
668 	}
669 	SANITY( source );
670 	*data = source->sniff->data;
671 	return( read_position );
672 }
673 unsigned char *
674 vips_source_sniff( VipsSource *source, size_t length )
675 {
676 	unsigned char *data;
677 	gint64 bytes_read;
678 	if( vips_source_test_features( source ) )
679 		return( NULL );
680 	bytes_read = vips_source_sniff_at_most( source, &data, length );
681 	if( bytes_read == -1 )
682 		return( NULL );
683 	if( bytes_read < length )
684 		return( NULL );
685 	return( data );
