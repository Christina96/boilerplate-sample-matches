275 			rslide->bounds.height /= rslide->downsample;
276 			whole.left = 0;
277 			whole.top = 0;
278 			whole.width = w;
279 			whole.height = h;
280 			vips_rect_intersectrect( &rslide->bounds, &whole, 
281 				&rslide->bounds );
282 			if( vips_rect_isempty( &rslide->bounds ) )
283 				rslide->autocrop = FALSE;
284 		}
285 		if( rslide->autocrop ) {
286 			w = rslide->bounds.width;
287 			h = rslide->bounds.height;
288 		}
289 		if( rslide->attach_associated &&
290 			readslide_attach_associated( rslide, image ) )
291 			return( -1 ); 
292 	}
293 	rslide->bg = 0xffffff;
294 	if( (background = openslide_get_property_value( rslide->osr,
295 		OPENSLIDE_PROPERTY_NAME_BACKGROUND_COLOR )) )
296 		rslide->bg = strtoul( background, NULL, 16 );
297 	if( w <= 0 || 
298 		h <= 0 || 
299 		rslide->downsample < 0 ) {
300 		vips_error( "openslide2vips", _( "getting dimensions: %s" ),
301 			openslide_get_error( rslide->osr ) );
302 		return( -1 );
303 	}
304 	if( w > INT_MAX || 
305 		h > INT_MAX ) {
306 		vips_error( "openslide2vips",
307 			"%s", _( "image dimensions overflow int" ) );
308 		return( -1 );
309 	}
310 	if( !rslide->autocrop ) {
311 		rslide->bounds.left = 0;
312 		rslide->bounds.top = 0;
313 		rslide->bounds.width = w;
314 		rslide->bounds.height = h;
315 	}
316 	xres = 1.0;
317 	yres = 1.0;
318 	for( properties = openslide_get_property_names( rslide->osr );
319 		*properties != NULL; properties++ ) {
320 		const char *name = *properties;
321 		const char *value = 
322 			openslide_get_property_value( rslide->osr, name );
323 		if( value ) { 
324 			vips_image_set_string( image, name, value ); 
325 			if( strcmp( *properties, "openslide.mpp-x" ) == 0 ) 
326 				xres = readslice_parse_res( rslide, name );
327 			if( strcmp( *properties, "openslide.mpp-y" ) == 0 ) 
328 				yres = readslice_parse_res( rslide, name );
329 		}
330 	}
331 	associated_names = g_strjoinv( ", ", (char **)
332 		openslide_get_associated_image_names( rslide->osr ) );
333 	vips_image_set_string( image, 
334 		"slide-associated-images", associated_names );
335 	VIPS_FREE( associated_names );
336 	vips_image_init_fields( image, w, h, 4, VIPS_FORMAT_UCHAR,
337 		VIPS_CODING_NONE, VIPS_INTERPRETATION_sRGB, xres, yres );
338 	return( 0 );
339 }

352 static int
353 vips__openslide_generate( VipsRegion *out, 
354 	void *_seq, void *_rslide, void *unused, gboolean *stop )
355 {
356 	ReadSlide *rslide = _rslide;
357 	uint32_t bg = rslide->bg;
358 	VipsRect *r = &out->valid;
359 	int n = r->width * r->height;
360 	uint32_t *buf = (uint32_t *) VIPS_REGION_ADDR( out, r->left, r->top );
361 	const char *error;
362 	VIPS_DEBUG_MSG( "vips__openslide_generate: %dx%d @ %dx%d\n",
363 		r->width, r->height, r->left, r->top );
364 	g_assert( (r->left % rslide->tile_width) == 0 );
365 	g_assert( (r->top % rslide->tile_height) == 0 );
366 	g_assert( r->width <= rslide->tile_width );
367 	g_assert( r->height <= rslide->tile_height );
368 	g_assert( VIPS_REGION_LSKIP( out ) == r->width * 4 );
369 	openslide_read_region( rslide->osr, 
370 		buf,
371 		(r->left + rslide->bounds.left) * rslide->downsample, 
372 		(r->top + rslide->bounds.top) * rslide->downsample, 
373 		rslide->level,
374 		r->width, r->height ); 
375 	error = openslide_get_error( rslide->osr );
376 	if( error ) {
377 		vips_error( "openslide2vips", 
378 			_( "reading region: %s" ), error );
379 		return( -1 );
380 	}
381 	argb2rgba( buf, n, bg );
382 	return( 0 );
383 }
384 static int
385 vips__openslide_read( const char *filename, VipsImage *out, 
386 	int level, gboolean autocrop, gboolean attach_associated )
387 {
388 	ReadSlide *rslide;
389 	VipsImage *raw;
390 	VipsImage *t;
391 	VIPS_DEBUG_MSG( "vips__openslide_read: %s %d\n", 
392 		filename, level );
393 	if( !(rslide = readslide_new( filename, out, level, autocrop, 
394 		NULL, attach_associated )) )
395 		return( -1 );
396 	raw = vips_image_new();
397 	vips_object_local( out, raw );
398 	if( readslide_parse( rslide, raw ) ||
399 		vips_image_generate( raw, 
400 			NULL, vips__openslide_generate, NULL, rslide, NULL ) )
401 		return( -1 );
402 	if( vips_tilecache( raw, &t, 
403 		"tile_width", rslide->tile_width, 
404 		"tile_height", rslide->tile_height,
405 		"max_tiles", 
406 			(int) (2.5 * (1 + raw->Xsize / rslide->tile_width)),
407 		"threaded", TRUE,
408 		NULL ) ) 
409 		return( -1 );
410 	if( vips_image_write( t, out ) ) {
411 		g_object_unref( t );
412 		return( -1 );
413 	}
414 	g_object_unref( t );
415 	return( 0 );
416 }
