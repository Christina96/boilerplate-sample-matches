494 static char *
495 extract_prefix( const char *dir, const char *name )
496 {
497 	char edir[VIPS_PATH_MAX];
498 	char vname[VIPS_PATH_MAX];
499 	int i;
500 	g_info( "trying for dir = \"%s\", name = \"%s\"", dir, name );
501 	if( !g_path_is_absolute( dir ) ) {
502 		char *cwd; 
503 		cwd = g_get_current_dir();
504 		vips_snprintf( edir, VIPS_PATH_MAX, 
505 			"%s" G_DIR_SEPARATOR_S "%s", cwd, dir );
506 		g_free( cwd );
507 	}
508 	else {
509 		vips_strncpy( edir, dir, VIPS_PATH_MAX );
510 	}
511 	if( !vips_ispostfix( edir, name ) ) 
512 		return( NULL );
513 	vips_strncpy( vname, edir, VIPS_PATH_MAX );
514 	vname[strlen( edir ) - strlen( name ) - 1] = '\0';
515 	for( i = 0; i < (int) strlen( vname ); i++ ) 
516 		if( vips_isprefix( G_DIR_SEPARATOR_S "." G_DIR_SEPARATOR_S, 
517 			vname + i ) )
518 			memmove( vname + i, vname + i + 2, 
519 				strlen( vname + i + 2 ) + 1 );
520 	if( vips_ispostfix( vname, G_DIR_SEPARATOR_S "." ) )
521 		vname[strlen( vname ) - 2] = '\0';
522 	if( vips_ispostfix( vname, G_DIR_SEPARATOR_S ) )
523 		vname[strlen( vname ) - 1] = '\0';
524 	g_info( "canonicalised path = \"%s\"", vname );
525 	if( !vips_ispostfix( vname, G_DIR_SEPARATOR_S "bin" ) ) 
526 		return( NULL );
527 	vname[strlen( vname ) - strlen( G_DIR_SEPARATOR_S "bin" )] = '\0';
528 	g_info( "found \"%s\"", vname );
529 	return( vips_strdup( NULL, vname ) );
530 }
531 static char *
532 scan_path( char *path, const char *name )
533 {
534 	char *p, *q;
535 	char *prefix;
536 	for( p = path; 
537 		(q = vips_break_token( p, G_SEARCHPATH_SEPARATOR_S )); p = q ) {
538 		char str[VIPS_PATH_MAX];
539 		vips_snprintf( str, VIPS_PATH_MAX, 
540 			"%s" G_DIR_SEPARATOR_S "%s", p, name );
541 		g_info( "looking in \"%s\" for \"%s\"", 
542 			p, name );
543 		if( vips_existsf( "%s", str ) && 
544 			(prefix = extract_prefix( str, name )) ) {
545 			return( prefix );
546 		}
547 	}
548 	return( NULL );
549 }

572 static const char *
573 guess_prefix( const char *argv0, const char *name )
574 {
575 	char *prefix;
576 	if( vips_existsf( "%s/vips-modules-%d.%d", 
577 		VIPS_LIBDIR, VIPS_MAJOR_VERSION, VIPS_MINOR_VERSION ) ) {
578 		g_info( "found %s/vips-modules-%d.%d", 
579 			VIPS_LIBDIR, VIPS_MAJOR_VERSION, VIPS_MINOR_VERSION );
580 		g_info( "using configure-time prefix" );
581 		return( VIPS_PREFIX );
582 	}
583 	if( argv0 ) {
584 		if( g_path_is_absolute( argv0 ) ) {
585 			if( (prefix = extract_prefix( argv0, name )) ) {
586 				g_info( "found \"%s\" from argv0", prefix );
587 				return( prefix );
588 			} 
589 		}
590 		if( (prefix = find_file( name )) ) {
591 			g_info( "found \"%s\" from PATH", prefix );
592 			return( prefix );
593 		}
594         }
595 	if( argv0 &&
596 		!g_path_is_absolute( argv0 ) ) {
597 		char *dir;
598 		char full_path[VIPS_PATH_MAX];
599 		char *resolved;
600 		dir = g_get_current_dir(); 
601 		vips_snprintf( full_path, VIPS_PATH_MAX, 
602 			"%s" G_DIR_SEPARATOR_S "%s", dir, argv0 );
603 		g_free( dir ); 
604 		if( (resolved = vips_realpath( full_path )) ) {
605 			prefix = extract_prefix( resolved, name );
606 			g_free( resolved );
607 			if( prefix ) { 
608 				g_info( "found \"%s\" from cwd", prefix );
609 				return( prefix );
610 			}
611 		}
612 	}
613 	return( VIPS_PREFIX );
614 }
615 const char *
616 vips_guess_prefix( const char *argv0, const char *env_name )
617 {
618         const char *prefix;
619 	if( (prefix = g_getenv( env_name )) )
620                 return( prefix );
621 #ifdef G_OS_WIN32

623 {
624         char *basename;

626 	prefix = guess_prefix( argv0, basename );

629 	g_setenv( env_name, prefix, TRUE );
630 	return( prefix );

633 vips_guess_libdir( const char *argv0, const char *env_name )
634 {
635 	const char *prefix = vips_guess_prefix( argv0, env_name );

644 	else

646 	return( libdir );
