322 			this->y0 != first->y0 ||
323 			this->w * this->dx != first->w * first->dx ||
324 			this->h * this->dy != first->h * first->dy ||
325 			this->resno_decoded != first->resno_decoded ||
326 			this->factor != first->factor ) {
327 			vips_error( class->nickname, 
328 				"%s", _( "components differ in geometry" ) );
329 			return( -1 );
330 		}
331 		if( this->prec != first->prec ||
332 			this->bpp != first->bpp ||
333 			this->sgnd != first->sgnd ) {
334 			vips_error( class->nickname, 
335 				"%s", _( "components differ in precision" ) );
336 			return( -1 );
337 		}
338 		if( this->dx != first->dx ||
339 			this->dy != first->dy ||
340 			first->dx != 1 ||
341 			first->dy != 1 )
342 			jp2k->upsample = TRUE;
343 	}
344 	if( vips_foreign_load_jp2k_set_header( jp2k, load->out ) ) 
345 		return( -1 );
346 	VIPS_SETSTR( load->out->filename, 
347 		vips_connection_filename( VIPS_CONNECTION( jp2k->source ) ) );
348 	return( 0 );
349 }
350 #define PACK( TYPE ) { \
351 	TYPE *tq = (TYPE *) q; \
352 	\
353 	for( x = 0; x < length; x++ ) { \
354 		for( i = 0; i < b; i++ ) \
355 			tq[i] = planes[i][x]; \
356 		\
357 		tq += b; \
358 	} \
359 }
360 #define PACK_UPSAMPLE( TYPE ) { \
361 	TYPE *tq = (TYPE *) q; \
362 	\
363 	for( x = 0; x < length; x++ ) { \
364 		for( i = 0; i < b; i++ ) { \
365 			int dx = image->comps[i].dx; \
366 			int pixel = planes[i][x / dx]; \
367 			\
368 			tq[i] = pixel; \
369 		} \
370 		\
371 		tq += b; \
372 	} \
373 }
374 static void
375 vips_foreign_load_jp2k_pack( gboolean upsample, 
376 	opj_image_t *image, VipsImage *im, 
377 	VipsPel *q, int left, int top, int length )
378 {
379 	int *planes[MAX_BANDS];
380 	int b = image->numcomps;
381 	int x, i;
382 #ifdef DEBUG_VERBOSE
383 	printf( "vips_foreign_load_jp2k_pack: "
384 		"upsample = %d, left = %d, top = %d, length = %d\n", 
385 		upsample, left, top, length ); 
386 	for( i = 0; i < b; i++ ) {
387 		opj_image_comp_t *comp = &image->comps[i];
388 		planes[i] = comp->data + (top / comp->dy) * comp->w + 
389 			(left / comp->dx);
390 	}
391 	if( upsample ) 
392 		switch( im->BandFmt ) {
393 		case VIPS_FORMAT_CHAR:
394 		case VIPS_FORMAT_UCHAR:
395 			PACK_UPSAMPLE( unsigned char );
396 			break;
397 		case VIPS_FORMAT_SHORT:
398 		case VIPS_FORMAT_USHORT:
399 			PACK_UPSAMPLE( unsigned short );
400 			break;
401 		case VIPS_FORMAT_INT:
402 		case VIPS_FORMAT_UINT:
403 			PACK_UPSAMPLE( unsigned int );
404 			break;
405 		default:
406 			g_assert_not_reached();
407 			break;
408 		}
409 	else 
410 		switch( im->BandFmt ) {
411 		case VIPS_FORMAT_CHAR:
412 		case VIPS_FORMAT_UCHAR:
413 			PACK( unsigned char );
414 			break;
415 		case VIPS_FORMAT_SHORT:
416 		case VIPS_FORMAT_USHORT:
417 			PACK( unsigned short );
418 			break;
419 		case VIPS_FORMAT_INT:
420 		case VIPS_FORMAT_UINT:
421 			PACK( unsigned int );
422 			break;
423 		default:
424 			g_assert_not_reached();
425 			break;
426 		}
427 }
428 #define YCC_TO_RGB( TYPE ) { \
429 	TYPE *tq = (TYPE *) q; \
430 	\
431 	for( x = 0; x < length; x++ ) { \
432 		int y = tq[0]; \
433 		int cb = tq[1] - offset; \
434 		int cr = tq[2] - offset; \
435 		\
436 		int r, g, b; \
437 		\
438 		r = y + (int)(1.402 * (float)cr); \
439 		tq[0] = VIPS_CLIP( 0, r, upb ); \
440 		\
441 		g = y - (int)(0.344 * (float)cb + 0.714 * (float)cr); \
442 		tq[1] = VIPS_CLIP( 0, g, upb ); \
443 		\
444 		b = y + (int)(1.772 * (float)cb); \
445 		tq[2] = VIPS_CLIP( 0, b, upb ); \
446 		\
447 		tq += 3; \
448 	} \
449 }
450 static void
451 vips_foreign_load_jp2k_ycc_to_rgb( opj_image_t *image, VipsImage *im, 
452 	VipsPel *q, int length )
453 {
454 	int prec = image->comps[0].prec;
455 	int offset = 1 << (prec - 1);
456 	int upb = (1 << prec) - 1;
457 	int x;
458 	switch( im->BandFmt ) {
459 	case VIPS_FORMAT_CHAR:
460 	case VIPS_FORMAT_UCHAR:
461 		YCC_TO_RGB( unsigned char );
462 		break;
463 	case VIPS_FORMAT_SHORT:
464 	case VIPS_FORMAT_USHORT:
465 		YCC_TO_RGB( unsigned short );
466 		break;
467 	case VIPS_FORMAT_INT:
468 	case VIPS_FORMAT_UINT:
469 		YCC_TO_RGB( unsigned int );
470 		break;
471 	default:
472 		g_assert_not_reached();
473 		break;
474 	}
475 }
476 static int
477 vips_foreign_load_jp2k_generate( VipsRegion *out, 
478 	void *seq, void *a, void *b, gboolean *stop )
479 {
480 	VipsForeignLoad *load = (VipsForeignLoad *) a;
481 	VipsForeignLoadJp2k *jp2k = (VipsForeignLoadJp2k *) load;
482 	VipsRect *r = &out->valid;
483 	int tile_width = VIPS_ROUND_UINT( 
484 		(double) jp2k->info->tdx / jp2k->shrink );
485 	int tile_height = VIPS_ROUND_UINT( 
486 		(double) jp2k->info->tdy / jp2k->shrink );
487 	int tiles_across = jp2k->info->tw;
488 	int x, y, z;
489 #ifdef DEBUG_VERBOSE
490 	printf( "vips_foreign_load_jp2k_generate: "
491 		"left = %d, top = %d, width = %d, height = %d\n", 
492 		r->left, r->top, r->width, r->height ); 
493 	if( jp2k->n_errors )
494 		return( 0 );
495 	y = 0;
496 	while( y < r->height ) {
497 		VipsRect tile, hit;
498 		hit.height = 0;
499 		x = 0;
500 		while( x < r->width ) { 
501 			int tx = (r->left + x) / tile_width;
502 			int ty = (r->top + y) / tile_height;
503 			int xs = tx * tile_width;
504 			int ys = ty * tile_height;
505 			int tile_index = ty * tiles_across + tx;
506 #ifdef DEBUG_VERBOSE
507 			printf( "   fetch tile %d\n", tile_index );
508 			if( !opj_get_decoded_tile( jp2k->codec, 
509 				jp2k->stream, jp2k->image, tile_index ) )
510 				return( -1 );
511 			tile.left = xs;
512 			tile.top = ys;
513 			tile.width = tile_width;
514 			tile.height = tile_height;
515 			vips_rect_intersectrect( &tile, r, &hit );
516 			for( z = 0; z < hit.height; z++ ) {
517 				VipsPel *q = VIPS_REGION_ADDR( out, 
518 					hit.left, hit.top + z );
519 				vips_foreign_load_jp2k_pack( jp2k->upsample, 
520 					jp2k->image, out->im, q,
521 					hit.left - tile.left,
522 					hit.top - tile.top + z,
523 					hit.width ); 
524 				if( jp2k->ycc_to_rgb )
525 					vips_foreign_load_jp2k_ycc_to_rgb( 
526 						jp2k->image, out->im, q, 
527 						hit.width );
528 			}
529 			x += hit.width;
530 		}
531 		y += hit.height;
532 	}
533 	if( load->fail &&
534 		jp2k->n_errors > 0 ) 
535 		return( -1 );
536 	return( 0 );
537 }
538 static int
539 vips_foreign_load_jp2k_load( VipsForeignLoad *load )
540 {
541 	VipsForeignLoadJp2k *jp2k = (VipsForeignLoadJp2k *) load;
542 	int tile_width = jp2k->info->tdx;
543 	int tile_height = jp2k->info->tdy;
544 	int tiles_across = jp2k->info->tw;
545 	VipsImage **t = (VipsImage **) 
546 		vips_object_local_array( VIPS_OBJECT( load ), 3 );
547 #ifdef DEBUG
548 	printf( "vips_foreign_load_jp2k_load:\n" );
549 	t[0] = vips_image_new();
550 	if( vips_foreign_load_jp2k_set_header( jp2k, t[0] ) ) 
551 		return( -1 );
552 	if( vips_image_generate( t[0], 
553 		NULL, vips_foreign_load_jp2k_generate, NULL, jp2k, NULL ) )
554 		return( -1 );
555 	if( vips_tilecache( t[0], &t[1], 
556 		"tile_width", tile_width,
557 		"tile_height", tile_height,
558 		"max_tiles", 3 * tiles_across,
559 		NULL ) ) 
560 		return( -1 );
561 	if( vips_image_write( t[1], load->real ) ) 
562 		return( -1 );
563 	return( 0 );
564 
}

777 int
778 vips__foreign_load_jp2k_decompress( VipsImage *out, 
779 	int width, int height, gboolean ycc_to_rgb, 
780 	void *from, size_t from_length, 
781 	void *to, size_t to_length )
782 {
783 	size_t pel_size = VIPS_IMAGE_SIZEOF_PEL( out );
784 	size_t line_size = pel_size * width;
785 	TileDecompress decompress = { 0 };
786 	opj_dparameters_t parameters;
787 	int i;
788 	gboolean upsample;
789 	VipsPel *q;
790 	int y;
791 #ifdef DEBUG
792 	printf( "vips__foreign_load_jp2k_decompress: width = %d, height = %d, "
793 		"ycc_to_rgb = %d, from_length = %zd, to_length = %zd\n",
794 		width, height, ycc_to_rgb, from_length, to_length );
795 	ycc_to_rgb = ycc_to_rgb && out->Bands == 3;
796 	decompress.codec = opj_create_decompress( OPJ_CODEC_J2K );
797 	opj_set_default_decoder_parameters( &parameters );
798 	opj_setup_decoder( decompress.codec, &parameters );
799 	opj_set_warning_handler( decompress.codec, warning_callback, NULL );
800 	opj_set_error_handler( decompress.codec, error_callback, NULL );
801 	decompress.source = vips_source_new_from_memory( from, from_length );
802 	decompress.stream = vips_foreign_load_jp2k_stream( decompress.source );
803 	if( !opj_read_header( decompress.stream, 
804 		decompress.codec, &decompress.image ) ) {
805 		vips_error( "jp2kload", "%s", ( "header error" ) );
806 		vips__foreign_load_jp2k_decompress_free( &decompress ); 
807 		return( -1 );
808 	}
809 	if( decompress.image->x1 > width || 
810 		decompress.image->y1 > height ||
811 		line_size * height > to_length ) {
812 		vips_error( "jp2kload", "%s", ( "bad dimensions" ) );
813 		vips__foreign_load_jp2k_decompress_free( &decompress ); 
814     		return( -1 );
815 	}
816 	if( !opj_decode( decompress.codec, 
817 		decompress.stream, decompress.image ) ) {
818 		vips_error( "jp2kload", "%s", ( "decode error" ) );
819 		vips__foreign_load_jp2k_decompress_free( &decompress ); 
820 		return( -1 );
821 	}
822 	upsample = FALSE;
823 	for( i = 0; i < decompress.image->numcomps; i++ ) {
824 		opj_image_comp_t *this = &decompress.image->comps[i];
825 		if( this->dx > 1 ||
826 			this->dy > 1 )
827 			upsample = TRUE;
828 	}
829 	q = to;
830 	for( y = 0; y < height; y++ ) {
831 		vips_foreign_load_jp2k_pack( upsample, 
832 			decompress.image, out, q,
833 			0, y, width ); 
834 		if( ycc_to_rgb )
835 			vips_foreign_load_jp2k_ycc_to_rgb( 
836 				decompress.image, out, q, 
837 				width );
838 		q += line_size;
839 	}
840 	vips__foreign_load_jp2k_decompress_free( &decompress ); 
841 	return( 0 );
842 }
