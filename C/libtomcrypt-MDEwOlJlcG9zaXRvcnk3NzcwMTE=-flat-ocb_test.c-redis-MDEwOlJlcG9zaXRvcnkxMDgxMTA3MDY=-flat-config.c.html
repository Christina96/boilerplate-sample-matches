1511  struct rewriteConfigState *rewriteConfigReadOldFile(char *path) {
1512      FILE *fp = fopen(path,"r");

1530                  state->has_tail = 1;

1536              sds aux = sdsnew("# ??? ");

1538              sdsfree(line);
