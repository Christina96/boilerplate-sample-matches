139  {
140  	const struct timespec interval = { 0, 250000000 };
141  	unsigned idx = 0, nr = 1;
142  	(void) data;

147  		     i++, t = (t + FSEL_FILES / nr) % FSEL_FILES) {
148  			if (fsel_cnt[t] == FSEL_CNT_MAX)
149  				continue;
