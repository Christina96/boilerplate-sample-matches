#define CLOOP( TYPE ) { \
	TYPE * restrict left = (TYPE *) in[0]; \
	TYPE * restrict right = (TYPE *) in[1]; \
	TYPE * restrict q = (TYPE *) out; \
	\
	for( x = 0; x < sz; x++ ) { \
		double x1 = left[0]; \
		double y1 = left[1]; \
		double x2 = right[0]; \
		double y2 = right[1]; \
		\
		left += 2; \
		right += 2; \
		\
		q[0] = x1 * x2 - y1 * y2; \
		q[1] = x1 * y2 + x2 * y1; \
		\
		q += 2; \
	} \
}
#define RLOOP( IN, OUT ) { \
	IN * restrict left = (IN *) in[0]; \
	IN * restrict right = (IN *) in[1]; \
	OUT * restrict q = (OUT *) out; \
	\
	for( x = 0; x < sz; x++ ) \
		q[x] = left[x] * right[x]; \
}

seek_type_to_lseek( GSeekType type )
{
	switch( type ) {
	default:
	case G_SEEK_CUR:
		return( SEEK_CUR );
	case G_SEEK_SET:
		return( SEEK_SET );
	case G_SEEK_END:
		return( SEEK_END );
	}
}
