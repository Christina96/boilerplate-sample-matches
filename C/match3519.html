290 static int
291 vips_foreign_load_magick7_parse( VipsForeignLoadMagick7 *magick7, 
292 	Image *image, VipsImage *out )
293 {
294 	VipsObjectClass *class = VIPS_OBJECT_GET_CLASS( magick7 );
295 	const char *key;
296 	Image *p;
297 #ifdef DEBUG
298 	printf( "image->depth = %zd\n", image->depth ); 
299 	printf( "GetImageType() = %d\n", GetImageType( image ) );
300 	vips_magick7_print_image_type( image ); 
301 	printf( "GetPixelChannels() = %zd\n", GetPixelChannels( image ) ); 
302 	printf( "image->columns = %zd\n", image->columns ); 
303 	printf( "image->rows = %zd\n", image->rows ); 
304 	out->Xsize = image->columns;
305 	out->Ysize = image->rows;
306 	magick7->frame_height = image->rows;
307 	out->Bands = magick7_get_bands( image ); 
308 	if( out->Xsize <= 0 ||
309 		out->Ysize <= 0 ||
310 		out->Bands <= 0 ||
311 		out->Xsize >= VIPS_MAX_COORD ||
312 		out->Ysize >= VIPS_MAX_COORD ||
313 		out->Bands >= VIPS_MAX_COORD ) {
314 		vips_error( class->nickname, 
315 			_( "bad image dimensions %d x %d pixels, %d bands" ),
316 			out->Xsize, out->Ysize, out->Bands );
317 		return( -1 );
318 	}
319 	out->BandFmt = -1;
320 	if( image->depth >= 1 && image->depth <= 8 ) 
321 		out->BandFmt = VIPS_FORMAT_UCHAR;
322 	if( image->depth >= 9 && image->depth <= 16 ) 
323 		out->BandFmt = VIPS_FORMAT_USHORT;
324 	if( image->depth == 32 )
325 		out->BandFmt = VIPS_FORMAT_FLOAT;
326 	if( image->depth == 64 )
327 		out->BandFmt = VIPS_FORMAT_DOUBLE;
328 	if( out->BandFmt == -1 ) {
329 		vips_error( class->nickname, 
330 			_( "unsupported bit depth %zd" ), image->depth );
331 		return( -1 );
332 	}
333 	switch( image->colorspace ) {
334 	case GRAYColorspace:
335 		if( out->BandFmt == VIPS_FORMAT_USHORT )
336 			out->Type = VIPS_INTERPRETATION_GREY16;
337 		else
338 			out->Type = VIPS_INTERPRETATION_B_W;
339 		break;
340 	case sRGBColorspace:
341 	case RGBColorspace:
342 		if( out->BandFmt == VIPS_FORMAT_USHORT )
343 			out->Type = VIPS_INTERPRETATION_RGB16;
344 		else
345 			out->Type = VIPS_INTERPRETATION_sRGB;
346 		break;
347 	case CMYKColorspace:
348 		out->Type = VIPS_INTERPRETATION_CMYK;
349 		break;
350 	default:
351 		vips_error( class->nickname, 
352 			_( "unsupported colorspace %s" ), 
353 			magick_ColorspaceType2str( image->colorspace ) );
354 		return( -1 );
355 	}
356 	switch( image->units ) {
357 	case PixelsPerInchResolution:
358 		out->Xres = image->resolution.x / 25.4;
359 		out->Yres = image->resolution.y / 25.4;
360 		vips_image_set_string( out, VIPS_META_RESOLUTION_UNIT, "in" );
361 		break;
362 	case PixelsPerCentimeterResolution:
363 		out->Xres = image->resolution.x / 10.0;
364 		out->Yres = image->resolution.y / 10.0;
365 		vips_image_set_string( out, VIPS_META_RESOLUTION_UNIT, "cm" );
366 		break;
367 	default:
368 		out->Xres = 1.0;
369 		out->Yres = 1.0;
370 		break;
371 	}
372 	out->Coding = VIPS_CODING_NONE;
373 	if( vips_image_pipelinev( out, VIPS_DEMAND_STYLE_SMALLTILE, NULL ) )
374 		return( -1 );
375 	ResetImagePropertyIterator( image );
376 	while( (key = GetNextImageProperty( image )) ) {
377 		char name_text[256];
378 		VipsBuf name = VIPS_BUF_STATIC( name_text );
379 		const char *value;
380 		value = GetImageProperty( image, key, magick7->exception );
381 		if( !value ) {
382 			vips_foreign_load_magick7_error( magick7 );
383 			return( -1 ); 
384 		}
385 		vips_buf_appendf( &name, "magick-%s", key );
386 		vips_image_set_string( out, vips_buf_all( &name ), value );
387 	}
388 	if( magick_set_vips_profile( out, image ) )
389 		return( -1 );
390 	magick7->n_pages = GetImageListLength( GetFirstImageInList( image ) );
391 #ifdef DEBUG
392 	printf( "image has %d pages\n", magick7->n_pages );
393 	magick7->n_frames = 0;
394 	for( p = image; p; (p = GetNextImageInList( p )) ) {
395 		if( p->columns != (unsigned int) out->Xsize ||
396 			p->rows != (unsigned int) out->Ysize ||
397 			magick7_get_bands( p ) != out->Bands ||
398 			p->depth != image->depth ) {
399 #ifdef DEBUG
400 			printf( "frame %d differs\n", magick7->n_frames );
401 			printf( "%zdx%zd, %d bands\n", 
402 				p->columns, p->rows, magick7_get_bands( p ) );
403 			printf( "first frame is %dx%d, %d bands\n", 
404 				out->Xsize, out->Ysize, out->Bands );
405 			break;
406 		}
407 		magick7->n_frames += 1;
408 	}
409 	if( p ) 
410 		magick7->n_frames = 1;
411 #ifdef DEBUG
412 	printf( "will read %d frames\n", magick7->n_frames );
413 	if( magick7->n != -1 )
414 		magick7->n_frames = VIPS_MIN( magick7->n_frames, magick7->n );
415 	if( vips_object_argument_isset( VIPS_OBJECT( magick7 ), "n" ) ) {
416 		vips_image_set_int( out, VIPS_META_PAGE_HEIGHT, out->Ysize );
417 		out->Ysize *= magick7->n_frames;
418 	}
419 	vips_image_set_int( out, VIPS_META_N_PAGES, magick7->n_pages );
420 	return( 0 );
421 }
422 #define UNPACK( TYPE ) { \
423 	TYPE * restrict tq = (TYPE *) q; \
424 	int x; \
425 	int b; \
426 	\
427 	for( x = 0; x < r->width; x++ ) { \
428 		for( b = 0; b < GetPixelChannels( image ); b++ ) { \
429 			PixelChannel channel = \
430 				GetPixelChannelChannel( image, b ); \
431 			\
432 			if( channel != IndexPixelChannel ) \
433 				*tq++ = p[b]; \
434 		} \
435 		\
436 		p += GetPixelChannels( image ); \
437 	} \
438 }
439 static int
440 vips_foreign_load_magick7_fill_region( VipsRegion *or, 
441 	void *seq, void *a, void *b, gboolean *stop )
442 {
443 	VipsForeignLoadMagick7 *magick7 = (VipsForeignLoadMagick7 *) a;
444 	VipsRect *r = &or->valid;
445 	VipsImage *im = or->im;
446 	int y;
447 	for( y = 0; y < r->height; y++ ) {
448 		int top = r->top + y;
449 		int frame = top / magick7->frame_height;
450 		int line = top % magick7->frame_height;
451 		Image *image = magick7->frames[frame];
452 		Quantum * restrict p;
453 		VipsPel * restrict q;
454 		g_mutex_lock( magick7->lock );
455 		p = GetCacheViewAuthenticPixels( magick7->cache_view[frame],
456 			r->left, line, r->width, 1, 
457 			magick7->exception );
458 		g_mutex_unlock( magick7->lock );
459 		if( !p ) 
460 			continue;
461 		q = VIPS_REGION_ADDR( or, r->left, top ); 
462 		switch( im->BandFmt ) {
463 		case VIPS_FORMAT_UCHAR:
464 			UNPACK( unsigned char ); 
465 			break;
466 		case VIPS_FORMAT_USHORT:
467 			UNPACK( unsigned short ); 
468 			break;
469 		case VIPS_FORMAT_FLOAT:
470 			UNPACK( float ); 
471 			break;
472 		case VIPS_FORMAT_DOUBLE:
473 			UNPACK( double ); 
474 			break;
475 		default:
476 			g_assert_not_reached();
477 		}
478 	}
479 	return( 0 );
480 }
481 static int
482 vips_foreign_load_magick7_load( VipsForeignLoadMagick7 *magick7 )
483 {
484 	VipsForeignLoad *load = (VipsForeignLoad *) magick7;
485 	Image *p;
486 	int i;
487 #ifdef DEBUG
488 	printf( "vips_foreign_load_magick7_load: %p\n", magick7 ); 
489 	if( vips_foreign_load_magick7_parse( magick7, 
490 		magick7->image, load->out ) )
491 		return( -1 );
492 	g_assert( !magick7->frames ); 
493 	if( !(magick7->frames = 
494 		VIPS_ARRAY( NULL, magick7->n_frames, Image * )) )
495 		return( -1 );
496 	p = magick7->image;
497 	for( i = 0; i < magick7->n_frames; i++ ) {
498 		magick7->frames[i] = p;
499 		p = GetNextImageInList( p );
500 	}
501 	g_assert( !magick7->cache_view ); 
502 	if( !(magick7->cache_view = VIPS_ARRAY( NULL, 
503 		magick7->n_frames, CacheView * )) )
504 		return( -1 );
505 	for( i = 0; i < magick7->n_frames; i++ ) {
506 		magick7->cache_view[i] = AcquireAuthenticCacheView( 
507 			magick7->frames[i], magick7->exception );
508 	}
509 #ifdef DEBUG
510 	vips_magick7_print_traits( magick7->frames[0] ); 
511 	vips_magick7_print_channel_names( magick7->frames[0] );
512 	if( vips_image_generate( load->out, 
513 		NULL, vips_foreign_load_magick7_fill_region, NULL, 
514 		magick7, NULL ) )
515 		return( -1 );
516 	return( 0 );
517 
}
