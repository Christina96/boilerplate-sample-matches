340      {
341      union
342          {

344          u64b_t  w[SKEIN_512_STATE_WORDS];
345          } cfg;                              \* config block */
346      Skein_Assert(hashBitLen > 0,SKEIN_BAD_HASHLEN);

350  #ifndef SKEIN_NO_PRECOMP

353          default:

355              cfg.w[0] = Skein_Swap64(SKEIN_SCHEMA_VER);  \* set the schema, version */

363      Skein_Start_New_Type(ctx,MSG);              \* T0=0, T1= MSG type */

366  static int Skein_512_Update(Skein_512_Ctxt_t *ctx, const u08b_t *msg, size_t msgByteCnt)
367      {
368      size_t n;
369      Skein_Assert(ctx->h.bCnt <= SKEIN_512_BLOCK_BYTES,SKEIN_FAIL);    \* catch uninitialized context */
370      if (msgByteCnt + ctx->h.bCnt > SKEIN_512_BLOCK_BYTES)
371          {
372          if (ctx->h.bCnt)                              \* finish up any buffered message data */
373              {
374              n = SKEIN_512_BLOCK_BYTES - ctx->h.bCnt;  \* # bytes free in buffer b[] */
375              if (n)
376                  {
377                  Skein_assert(n < msgByteCnt);         \* check on our logic here */
378                  memcpy(&ctx->b[ctx->h.bCnt],msg,n);
379                  msgByteCnt  -= n;
380                  msg         += n;
381                  ctx->h.bCnt += n;
382                  }
383              Skein_assert(ctx->h.bCnt == SKEIN_512_BLOCK_BYTES);
384              Skein_512_Process_Block(ctx,ctx->b,1,SKEIN_512_BLOCK_BYTES);
385              ctx->h.bCnt = 0;
386              }
387          if (msgByteCnt > SKEIN_512_BLOCK_BYTES)
388              {
389              n = (msgByteCnt-1) / SKEIN_512_BLOCK_BYTES;   \* number of full blocks to process */
390              Skein_512_Process_Block(ctx,msg,n,SKEIN_512_BLOCK_BYTES);
391              msgByteCnt -= n * SKEIN_512_BLOCK_BYTES;
392              msg        += n * SKEIN_512_BLOCK_BYTES;
393              }
394          Skein_assert(ctx->h.bCnt == 0);
395          }
396      if (msgByteCnt)
397          {
398          Skein_assert(msgByteCnt + ctx->h.bCnt <= SKEIN_512_BLOCK_BYTES);
399          memcpy(&ctx->b[ctx->h.bCnt],msg,msgByteCnt);
400          ctx->h.bCnt += msgByteCnt;
401          }
402      return SKEIN_SUCCESS;
403      }
404  static int Skein_512_Final(Skein_512_Ctxt_t *ctx, u08b_t *hashVal)
405      {
406      size_t i,n,byteCnt;
407      u64b_t X[SKEIN_512_STATE_WORDS];
408      Skein_Assert(ctx->h.bCnt <= SKEIN_512_BLOCK_BYTES,SKEIN_FAIL);    \* catch uninitialized context */
409      ctx->h.T[1] |= SKEIN_T1_FLAG_FINAL;                 \* tag as the final block */
410      if (ctx->h.bCnt < SKEIN_512_BLOCK_BYTES)            \* zero pad b[] if necessary */
411          memset(&ctx->b[ctx->h.bCnt],0,SKEIN_512_BLOCK_BYTES - ctx->h.bCnt);
412      Skein_512_Process_Block(ctx,ctx->b,1,ctx->h.bCnt);  \* process the final block */
413      byteCnt = (ctx->h.hashBitLen + 7) >> 3;             \* total number of output bytes */
414      memset(ctx->b,0,sizeof(ctx->b));  \* zero out b[], so it can hold the counter */
415      memcpy(X,ctx->X,sizeof(X));       \* keep a local copy of counter mode "key" */
416      for (i=0;i*SKEIN_512_BLOCK_BYTES < byteCnt;i++)
417          {
418          ((u64b_t *)ctx->b)[0]= Skein_Swap64((u64b_t) i); \* build the counter block */
419          Skein_Start_New_Type(ctx,OUT_FINAL);
420          Skein_512_Process_Block(ctx,ctx->b,1,sizeof(u64b_t)); \* run "counter mode" */
421          n = byteCnt - i*SKEIN_512_BLOCK_BYTES;   \* number of output bytes left to go */
422          if (n >= SKEIN_512_BLOCK_BYTES)
423              n  = SKEIN_512_BLOCK_BYTES;
424          Skein_Put64_LSB_First(hashVal+i*SKEIN_512_BLOCK_BYTES,ctx->X,n);   \* "output" the ctr mode bytes */
425          Skein_Show_Final(512,&ctx->h,n,hashVal+i*SKEIN_512_BLOCK_BYTES);
426          memcpy(ctx->X,X,sizeof(X));   \* restore the counter mode key for next time */
427          }
428      return SKEIN_SUCCESS;
429      }

456  {
457    Skein_Assert((state->u.h.T[1] & SKEIN_T1_FLAG_BIT_PAD) == 0 || databitlen == 0, SKEIN_FAIL);

459    if ((databitlen & 7) == 0)  \* partial bytes? */
460    {
461      return Skein_512_Update(&state->u.ctx_512,data,databitlen >> 3);
462    }
463    else
464    {   \* handle partial final byte */
465      size_t bCnt = (databitlen >> 3) + 1;                  \* number of bytes to handle (nonzero here!) */
466      u08b_t b,mask;
467      mask = (u08b_t) (1u << (7 - (databitlen & 7)));       \* partial byte bit mask */

470      Skein_512_Update(&state->u.ctx_512,&b  ,  1   ); \* process the (masked) partial byte */
471      Skein_Set_Bit_Pad_Flag(state->u.h);                    \* set tweak flag for the final call */
