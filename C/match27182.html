234 	VIPS_DEBUG_MSG( "fits2vips_generate: "

272 			for( z = 0; z < MAX_DIMENSIONS; z++ )
273 				inc[z] = 1;

275 			g_mutex_lock( fits->lock );
276 			if( vips_fits_read_subset( fits, 
277 				fpixel, lpixel, inc, q ) ) { 
278 				g_mutex_unlock( fits->lock );
279 				return( -1 );
280 			}

301 int
302 vips__fits_read( const char *filename, VipsImage *out )
303 {
304 	VipsImage *t;
305 	int bands;
306 	VipsInterpretation interpretation;
307 	VIPS_DEBUG_MSG( "fits2vips: reading \"%s\"\n", filename );
308 	t = vips_image_new();
309 	if( vips__fits_read_header( filename, t ) ) {
310 		g_object_unref( t );
311 		return( -1 );
312 	}
313 	bands = t->Bands;
314 	interpretation = t->Type;
315 	g_object_unref( t );
316 	if( bands == 1 ) {
317 		if( fits2vips( filename, out, 0 ) )
318 			return( -1 );
319 	}
320 	else {
321 		VipsImage **x;
322 		VipsImage **y;
323 		int i;
324 		t = vips_image_new();
325 		x = (VipsImage **) vips_object_local_array( VIPS_OBJECT( t ), 
326 			bands );
327 		y = (VipsImage **) vips_object_local_array( VIPS_OBJECT( t ), 
328 			3 );
329 		for( i = 0; i < bands; i++ ) {
330 			x[i] = vips_image_new();
331 			if( fits2vips( filename, x[i], i ) ) {
332 				g_object_unref( t );
333 				return( -1 );
334 			}
335 		}
336 		if( vips_bandjoin( x, &y[0], bands, NULL ) ||
337 			vips_copy( y[0], &y[1], 
338 				"interpretation", interpretation,
339 				NULL ) ||
340 			vips_image_write( y[1], out ) ) {
341 			g_object_unref( t );
342 			return( -1 );
343 		}
344 		g_object_unref( t );
345 	}
346 	return( 0 );
