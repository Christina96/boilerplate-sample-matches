void
vips_CMYK2XYZ_line( VipsColour *colour, VipsPel *out, VipsPel **in, int width )
{
	unsigned char *p = (unsigned char *) in[0];
	float *q = (float *) out;
	int i;
	for( i = 0; i < width; i++ ) {
		float c = p[0] / 255.0;
		float m = p[1] / 255.0;
		float y = p[2] / 255.0;
		float k = p[3] / 255.0;
		float r = 1.0 - (c * (1.0 - k) + k);
		float g = 1.0 - (m * (1.0 - k) + k);
		float b = 1.0 - (y * (1.0 - k) + k);
		q[0] = VIPS_D65_X0 * r;
		q[1] = VIPS_D65_Y0 * g;
		q[2] = VIPS_D65_Z0 * b;
		p += 4;
		q += 3;
	}
}
