550 		int_sum += value;
551 		int_sum = VIPS_CLIP( SHRT_MIN, int_sum, SHRT_MAX ); 
552 	}

554 	if( convi->exp > 0 )

556 	else

558 	int_value = VIPS_CLIP( 0, int_value, 255 ); 
559 	if( VIPS_ABS( true_value - int_value ) > 2 ) {
560 		g_info( "vips_convi_intize: too inaccurate" );
561 		return( -1 ); 
562 	}
563 }
