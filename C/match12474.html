#define POW( Y, X, E ) { \
	double left = (double) (X); \
	double right = (double) (E); \
	\
	(Y) = (left == 0.0) \
		? 0.0 \
		: (right == -1) \
			? 1.0 / left \
			: (right == 0.5) \

}
