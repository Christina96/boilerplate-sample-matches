float
vips_pythagoras( float L1, float a1, float b1, float L2, float a2, float b2 )
{
	float dL = L1 - L2;
	float da = a1 - a2;
	float db = b1 - b2;
	return( sqrt( dL * dL + da * da + db * db ) );
}
void
vips__pythagoras_line( VipsColour *colour, 
	VipsPel *out, VipsPel **in, int width )
{
	float * restrict p1 = (float *) in[0];
	float * restrict p2 = (float *) in[1];
	float * restrict q = (float *) out;
	int x;
	for( x = 0; x < width; x++ ) {
		float dL = p1[0] - p2[0];
		float da = p1[1] - p2[1];
		float db = p1[2] - p2[2];
		q[x] = sqrt( dL * dL + da * da + db * db );
		p1 += 3;
		p2 += 3;
	}
}

static void 
calcul_tables( int range, int *Y2v, float *v2Y )
{
	int i;
	for( i = 0; i < range; i++ ) {
		float f = (float) i / (range - 1);
		float v;
		if( f <= 0.0031308 )
			v = 12.92 * f; 
		else
			v = (1.0 + 0.055) * pow( f, 1.0 / 2.4 ) - 0.055;
		Y2v[i] = VIPS_RINT( (range - 1) * v );
	}
	Y2v[range] = Y2v[range - 1];
	for( i = 0; i < range; i++ ) {
		float f = (float) i / (range - 1);
		if( f <= 0.04045 )
			v2Y[i] = f / 12.92;
		else
			v2Y[i] = pow( (f + 0.055) / (1 + 0.055), 2.4 );
	}
}
