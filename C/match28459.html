375 gboolean
376 vips_colourspace_issupported( const VipsImage *image )
377 {
378 	VipsInterpretation interpretation;
379 	int i;
380 	interpretation = vips_image_guess_interpretation( image );
381 	if( interpretation == VIPS_INTERPRETATION_RGB )
382 		interpretation = VIPS_INTERPRETATION_sRGB;
383 	for( i = 0; i < VIPS_NUMBER( vips_colour_routes ); i++ )
384 		if( vips_colour_routes[i].from == interpretation )
385 			return( TRUE );
386 	return( FALSE );
387 }
388 typedef struct _VipsColourspace {
389 	VipsOperation parent_instance;
390 	VipsImage *in;
391 	VipsImage *out;
392 	VipsInterpretation space;
393 	VipsInterpretation source_space;
394 } VipsColourspace;
395 typedef VipsOperationClass VipsColourspaceClass;
396 G_DEFINE_TYPE( VipsColourspace, vips_colourspace, VIPS_TYPE_OPERATION );
397 static int
398 vips_colourspace_build( VipsObject *object )
399 {
400 	VipsColourspace *colourspace = (VipsColourspace *) object; 
401 	int i, j;
402 	VipsImage *x;
403 	VipsImage **t = (VipsImage **) 
404 		vips_object_local_array( object, 1 );
405 	VipsImage **pipe = (VipsImage **) 
406 		vips_object_local_array( object, MAX_STEPS );
407 	VipsInterpretation interpretation;
408 	if( VIPS_OBJECT_CLASS( vips_colourspace_parent_class )->
409 		build( object ) )
410 		return( -1 );
411 	x = colourspace->in;
412 	if( x->Coding == VIPS_CODING_RAD ) {
413 		if( vips_rad2float( x, &t[0], NULL ) )
414 			return( -1 );
415 		x = t[0]; 
416 	}
417 	if( vips_object_argument_isset( object, "source_space" ) )
418 		interpretation = colourspace->source_space;
419 	else
420 		interpretation = vips_image_guess_interpretation( x );
421 	if( interpretation == VIPS_INTERPRETATION_RGB )
422 		interpretation = VIPS_INTERPRETATION_sRGB;
423 	if( interpretation == colourspace->space ) {
424 		g_object_set( colourspace, "out", vips_image_new(), NULL ); 
425 		return( vips_image_write( colourspace->in, colourspace->out ) );
426 	}
427 	for( i = 0; i < VIPS_NUMBER( vips_colour_routes ); i++ )
428 		if( vips_colour_routes[i].from == interpretation &&
429 			vips_colour_routes[i].to == colourspace->space )
430 			break;
431 	if( i == VIPS_NUMBER( vips_colour_routes ) ) {
432 		vips_error( "vips_colourspace", 
433 			_( "no known route from '%s' to '%s'" ),
434 			vips_enum_nick( VIPS_TYPE_INTERPRETATION,
