static void
vips_LabQ2LabS_line( VipsColour *colour, VipsPel *out, VipsPel **in, int width )
{
	unsigned char * restrict p = (unsigned char *) in[0];
	signed short * restrict q = (signed short *) out;
	int i;
	unsigned char ext;
	signed short l, a, b;
	for( i = 0; i < width; i++ ) {
		l = p[0] << 7;
		a = p[1] << 8;
		b = p[2] << 8;
		ext = p[3];
		p += 4;
		l |= (unsigned char) (ext & 0xc0) >> 1;
		a |= (ext & 0x38) << 2;
		b |= (ext & 0x7) << 5;
		q[0] = l;
		q[1] = a;
		q[2] = b;
		q += 3;
	}
}
