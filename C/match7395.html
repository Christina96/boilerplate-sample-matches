25 vips_stdif_generate( VipsRegion *or, 
26 	void *vseq, void *a, void *b, gboolean *stop )
27 {
28 	VipsRect *r = &or->valid;
29 	VipsRegion *ir = (VipsRegion *) vseq;
30 	VipsImage *in = (VipsImage *) a;
31 	VipsStdif *stdif = (VipsStdif *) b;
32 	int bands = in->Bands; 
33 	int npel = stdif->width * stdif->width;
34 	VipsRect irect;
35 	int y;
36 	int lsk;
37 	int centre;			
38 	irect.left = or->valid.left;
39 	irect.top = or->valid.top;
40 	irect.width = or->valid.width + stdif->width;
41 	irect.height = or->valid.height + stdif->height;
42 	if( vips_region_prepare( ir, &irect ) )
43 		return( -1 );
44 	lsk = VIPS_REGION_LSKIP( ir );
45 	centre = lsk * (stdif->height / 2) + stdif->width / 2;
46 	for( y = 0; y < r->height; y++ ) {
47 		VipsPel *p = VIPS_REGION_ADDR( ir, r->left, r->top + y );
48 		VipsPel *q = VIPS_REGION_ADDR( or, r->left, r->top + y );
49 		double f1 = stdif->a * stdif->m0;
50 		double f2 = 1.0 - stdif->a;
51 		double f3 = stdif->b * stdif->s0;
52 		VipsPel *p1;
53 		int x, i, j, b;
54 		unsigned int sum[MAX_BANDS];
55 		unsigned int sum2[MAX_BANDS];
56 		for( b = 0; b < bands; b++ ) {
57 			memset( sum, 0, bands * sizeof( unsigned int ) );
58 			memset( sum2, 0, bands * sizeof( unsigned int ) );
59 		}
60 		p1 = p;
61 		for( j = 0; j < stdif->height; j++ ) {
62 			i = 0;
63 			for( x = 0; x < stdif->width; x++ ) {
64 				for( b = 0; b < bands; b++ ) { 
65 					int t = p1[i++];
66 					sum[b] += t;
67 					sum2[b] += t * t;
68 				}
69 			}
70 			p1 += lsk;
71 		}
72 		for( x = 0; x < r->width; x++ ) {
73 			for( b = 0; b < bands; b++ ) { 
74 				double mean = (double) sum[b] / npel;
75 				double var = (double) sum2[b] / npel - 
76 					(mean * mean);
77 				double sig = sqrt( var );
78 				double res = f1 + f2 * mean + 
79 					((double) p[centre] - mean) * 
80 					(f3 / (stdif->s0 + stdif->b * sig));
81 				if( res < 0.0 )
82 					*q++ = 0;
83 				else if( res >= 256.0 )
84 					*q++ = 255;
85 				else
86 					*q++ = res + 0.5;
87 				p1 = p;
88 				for( j = 0; j < stdif->height; j++ ) {
89 					int t1 = p1[0];
90 					int t2 = p1[bands * stdif->width];
91 					sum[b] -= t1;
92 					sum2[b] -= t1 * t1;
93 					sum[b] += t2;
94 					sum2[b] += t2 * t2;
95 					p1 += lsk;
96 				}
97 				p += 1;
98 			}
99 		}
100 	}
101 	return( 0 );
