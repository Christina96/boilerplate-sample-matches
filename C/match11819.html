static void
vips_Lab2LabQ_line( VipsColour *colour, VipsPel *out, VipsPel **in, int width )
{
	float * restrict p = (float *) in[0];
	VipsPel * restrict q = out; 
	int i;
	for( i = 0; i < width; i++ ) {
		float fval;
		int lsbs;
		int intv;

		intv = VIPS_ROUND_UINT( 10.23 * p[0] ); 
		intv = VIPS_CLIP( 0, intv, 1023 );
		lsbs = (intv & 0x3) << 6;       		q[0] = intv >> 2; 		
		fval = 8.0 * p[1];              		intv = VIPS_RINT( fval );
		intv = VIPS_CLIP( -1024, intv, 1023 );
		lsbs |= (intv & 0x7) << 3;      		q[1] = intv >> 3;   		
		fval = 8.0 * p[2];              		intv = VIPS_RINT( fval );
		intv = VIPS_CLIP( -1024, intv, 1023 );
		lsbs |= (intv & 0x7);
		q[2] = intv >> 3;
		q[3] = lsbs;                	
		p += 3;
		q += 4;
	}
}
