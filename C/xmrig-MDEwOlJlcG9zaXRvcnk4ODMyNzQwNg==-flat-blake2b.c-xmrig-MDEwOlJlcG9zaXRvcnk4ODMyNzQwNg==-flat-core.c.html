278  void xmrig_ar2_fill_first_blocks(uint8_t *blockhash, const argon2_instance_t *instance) {
279      uint32_t l;
280      uint8_t blockhash_bytes[ARGON2_BLOCK_SIZE];
281      for (l = 0; l < instance->lanes; ++l) {
282          store32(blockhash + ARGON2_PREHASH_DIGEST_LENGTH, 0);
283          store32(blockhash + ARGON2_PREHASH_DIGEST_LENGTH + 4, l);
284          xmrig_ar2_blake2b_long(blockhash_bytes, ARGON2_BLOCK_SIZE, blockhash, ARGON2_PREHASH_SEED_LENGTH);
285          load_block(&instance->memory[l * instance->lane_length + 0], blockhash_bytes);
286          store32(blockhash + ARGON2_PREHASH_DIGEST_LENGTH, 1);
287          xmrig_ar2_blake2b_long(blockhash_bytes, ARGON2_BLOCK_SIZE, blockhash, ARGON2_PREHASH_SEED_LENGTH);
288          load_block(&instance->memory[l * instance->lane_length + 1], blockhash_bytes);
289      }
290      xmrig_ar2_clear_internal_memory(blockhash_bytes, ARGON2_BLOCK_SIZE);
291  }
292  void xmrig_ar2_initial_hash(uint8_t *blockhash, argon2_context *context,
293                    argon2_type type) {
294      blake2b_state BlakeHash;
295      uint8_t value[sizeof(uint32_t)];
296      if (NULL == context || NULL == blockhash) {
297          return;
298      }
299      xmrig_ar2_blake2b_init(&BlakeHash, ARGON2_PREHASH_DIGEST_LENGTH);
300      store32(&value, context->lanes);
301      xmrig_ar2_blake2b_update(&BlakeHash, (const uint8_t *)&value, sizeof(value));
302      store32(&value, context->outlen);
303      xmrig_ar2_blake2b_update(&BlakeHash, (const uint8_t *)&value, sizeof(value));

309      xmrig_ar2_blake2b_update(&BlakeHash, (const uint8_t *)&value, sizeof(value));
310      store32(&value, (uint32_t)type);
311      xmrig_ar2_blake2b_update(&BlakeHash, (const uint8_t *)&value, sizeof(value));
312      store32(&value, context->pwdlen);
313      xmrig_ar2_blake2b_update(&BlakeHash, (const uint8_t *)&value, sizeof(value));

321      }

324      if (context->salt != NULL) {

326                         context->saltlen);
327      }
328      store32(&value, context->secretlen);
329      xmrig_ar2_blake2b_update(&BlakeHash, (const uint8_t *)&value, sizeof(value));
330      if (context->secret != NULL) {

335          }
336      }

338      xmrig_ar2_blake2b_update(&BlakeHash, (const uint8_t *)&value, sizeof(value));
339      if (context->ad != NULL) {
340          xmrig_ar2_blake2b_update(&BlakeHash, (const uint8_t *)context->ad, context->adlen);
341      }
342      xmrig_ar2_blake2b_final(&BlakeHash, blockhash, ARGON2_PREHASH_DIGEST_LENGTH);
343  }
