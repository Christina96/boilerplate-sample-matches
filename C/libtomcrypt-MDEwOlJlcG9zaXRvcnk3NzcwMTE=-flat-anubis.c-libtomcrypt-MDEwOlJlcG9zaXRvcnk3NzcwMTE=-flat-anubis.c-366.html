889           inter[i]  = T0[(kappa[j--] >> 24) & 0xff]; if (j < 0) j = N - 1;
890           inter[i] ^= T1[(kappa[j--] >> 16) & 0xff]; if (j < 0) j = N - 1;
891           inter[i] ^= T2[(kappa[j--] >>  8) & 0xff]; if (j < 0) j = N - 1;
892           inter[i] ^= T3[(kappa[j  ]      ) & 0xff];
893        }
894        kappa[0] = inter[0] ^ rc[r];
895        for (i = 1; i < N; i++) {
896           kappa[i] = inter[i];
897        }
898     }
899     for (i = 0; i < 4; i++) {
900        skey->anubis.roundKeyDec[0][i] = skey->anubis.roundKeyEnc[R][i];
901        skey->anubis.roundKeyDec[R][i] = skey->anubis.roundKeyEnc[0][i];
902     }
903     for (r = 1; r < R; r++) {
904        for (i = 0; i < 4; i++) {
905           v = skey->anubis.roundKeyEnc[R - r][i];
906           skey->anubis.roundKeyDec[r][i] =
907              T0[T4[(v >> 24) & 0xff] & 0xff] ^
908              T1[T4[(v >> 16) & 0xff] & 0xff] ^
909              T2[T4[(v >>  8) & 0xff] & 0xff] ^
910              T3[T4[(v      ) & 0xff] & 0xff];
911        }
912     }
913     return CRYPT_OK;
914  }

924  static void anubis_crypt(const unsigned char *plaintext, unsigned char *ciphertext,
925                           const ulong32 roundKey[18 + 1][4], int R) {
926     int i, pos, r;
927     ulong32 state[4];
928     ulong32 inter[4];
929      for (i = 0, pos = 0; i < 4; i++, pos += 4) {
930        state[i] =
931           (((ulong32)plaintext[pos    ]) << 24) ^
932           (((ulong32)plaintext[pos + 1]) << 16) ^
933           (((ulong32)plaintext[pos + 2]) <<  8) ^
934           (((ulong32)plaintext[pos + 3])      ) ^
935           roundKey[0][i];
936      }
937      for (r = 1; r < R; r++) {
938        inter[0] =
939           T0[(state[0] >> 24) & 0xff] ^
940           T1[(state[1] >> 24) & 0xff] ^
941           T2[(state[2] >> 24) & 0xff] ^
942           T3[(state[3] >> 24) & 0xff] ^
943           roundKey[r][0];
944        inter[1] =
945           T0[(state[0] >> 16) & 0xff] ^
946           T1[(state[1] >> 16) & 0xff] ^
947           T2[(state[2] >> 16) & 0xff] ^
948           T3[(state[3] >> 16) & 0xff] ^
949           roundKey[r][1];
950        inter[2] =
951           T0[(state[0] >>  8) & 0xff] ^
952           T1[(state[1] >>  8) & 0xff] ^
953           T2[(state[2] >>  8) & 0xff] ^
954           T3[(state[3] >>  8) & 0xff] ^
955           roundKey[r][2];
956        inter[3] =
957           T0[(state[0]      ) & 0xff] ^
958           T1[(state[1]      ) & 0xff] ^
959           T2[(state[2]      ) & 0xff] ^
960           T3[(state[3]      ) & 0xff] ^
961           roundKey[r][3];
962        state[0] = inter[0];
963        state[1] = inter[1];
964        state[2] = inter[2];
965        state[3] = inter[3];
966      }
967     inter[0] =
968        (T0[(state[0] >> 24) & 0xff] & 0xff000000U) ^
969        (T1[(state[1] >> 24) & 0xff] & 0x00ff0000U) ^
970        (T2[(state[2] >> 24) & 0xff] & 0x0000ff00U) ^
971        (T3[(state[3] >> 24) & 0xff] & 0x000000ffU) ^
972        roundKey[R][0];
973     inter[1] =
974        (T0[(state[0] >> 16) & 0xff] & 0xff000000U) ^
975        (T1[(state[1] >> 16) & 0xff] & 0x00ff0000U) ^
976        (T2[(state[2] >> 16) & 0xff] & 0x0000ff00U) ^
977        (T3[(state[3] >> 16) & 0xff] & 0x000000ffU) ^
978        roundKey[R][1];
979     inter[2] =
980        (T0[(state[0] >>  8) & 0xff] & 0xff000000U) ^
981        (T1[(state[1] >>  8) & 0xff] & 0x00ff0000U) ^
982        (T2[(state[2] >>  8) & 0xff] & 0x0000ff00U) ^
983        (T3[(state[3] >>  8) & 0xff] & 0x000000ffU) ^
984        roundKey[R][2];
985     inter[3] =
986        (T0[(state[0]      ) & 0xff] & 0xff000000U) ^
987        (T1[(state[1]      ) & 0xff] & 0x00ff0000U) ^
988        (T2[(state[2]      ) & 0xff] & 0x0000ff00U) ^
989        (T3[(state[3]      ) & 0xff] & 0x000000ffU) ^
990        roundKey[R][3];
991      for (i = 0, pos = 0; i < 4; i++, pos += 4) {
992          ulong32 w = inter[i];
993          ciphertext[pos    ] = (unsigned char)(w >> 24);
994          ciphertext[pos + 1] = (unsigned char)(w >> 16);
995          ciphertext[pos + 2] = (unsigned char)(w >>  8);
996          ciphertext[pos + 3] = (unsigned char)(w      );
997      }
998  }
