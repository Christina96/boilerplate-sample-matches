51 static int
52 vips_copy_build( VipsObject *object )
53 {
54 	VipsObjectClass *class = VIPS_OBJECT_GET_CLASS( object );
55 	VipsConversion *conversion = VIPS_CONVERSION( object );
56 	VipsCopy *copy = (VipsCopy *) object;
57 	guint64 pel_size_before;
58 	guint64 pel_size_after;
59 	VipsImage copy_of_fields;
60 	int i;
61 	if( VIPS_OBJECT_CLASS( vips_copy_parent_class )->build( object ) )
62 		return( -1 );
63 	if( vips_image_pio_input( copy->in ) )
64 		return( -1 );
65 	if( copy->swap ) 
66 		g_warning( "%s", 
67 			_( "copy swap is deprecated, use byteswap instead" ) );
68 	if( vips_image_pipelinev( conversion->out, 
69 		VIPS_DEMAND_STYLE_THINSTRIP, copy->in, NULL ) )
70 		return( -1 );
71 	copy_of_fields = *conversion->out;
72 	for( i = 0; i < VIPS_NUMBER( vips_copy_names ); i++ ) {
73 		const char *name = vips_copy_names[i];
74 		GParamSpec *pspec;
75 		VipsArgumentClass *argument_class;
76 		VipsArgumentInstance *argument_instance;
77 		if( vips_object_get_argument( object, name,
78 			&pspec, &argument_class, &argument_instance ) )
79 			return( -1 );
80 		if( argument_instance->assigned ) {
81 			GType type = G_PARAM_SPEC_VALUE_TYPE( pspec );
82 			GValue value = { 0, };
83 			g_value_init( &value, type );
84 			g_object_get_property( G_OBJECT( object ), 
85 				name, &value );
86 #ifdef VIPS_DEBUG
87 {
88 			char *str;
89 			str = g_strdup_value_contents( &value );
90 			printf( "vips_copy_build: %s = %s\n", name, str );
91 			g_free( str );
92 }
93 #endif 
94 			g_object_set_property( G_OBJECT( conversion->out ), 
95 				name, &value );
96 			g_value_unset( &value );
97 		}
98 	}
99 	pel_size_before = VIPS_IMAGE_SIZEOF_PEL( &copy_of_fields );
100 	pel_size_after = VIPS_IMAGE_SIZEOF_PEL( conversion->out );
101 	if( pel_size_after != pel_size_before ) {
102 		vips_error( class->nickname, 
103 			"%s", _( "must not change pel size" ) ); 
104 		return( -1 );
105 
	}
106 	if( vips_image_generate( conversion->out,
107 		vips_start_one, vips_copy_gen, vips_stop_one, 
108 		copy->in, copy ) )
109 		return( -1 );
110 	return( 0 );
111 }
