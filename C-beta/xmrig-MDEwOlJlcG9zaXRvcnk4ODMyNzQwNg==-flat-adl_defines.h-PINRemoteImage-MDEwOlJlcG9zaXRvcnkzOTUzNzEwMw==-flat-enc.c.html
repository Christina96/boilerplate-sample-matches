479    int sum = 0;

530                             const VP8Matrix* const mtx) {

534      const int j = kZigzag[n];

538        const uint32_t Q = mtx->q_[j];
539        const uint32_t iQ = mtx->iq_[j];
540        const uint32_t B = mtx->bias_[j];

547      } else {
548        out[n] = 0;
549        in[j] = 0;
550      }
551    }
