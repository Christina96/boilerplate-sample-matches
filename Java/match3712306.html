34 public class RelationUnknown extends ElasticsearchException implements ResourceUnknownException, TableScopeException {
35     private final RelationName relationName;
36     public static RelationUnknown of(String relation, List<String> candidates) {
37         switch (candidates.size()) {
38             case 0:
39                 return new RelationUnknown(relation);
40             case 1: {
41                 var name = RelationName.fromIndexName(relation);
42                 var msg = "Relation '" + relation + "' unknown. Maybe you meant '" + Identifiers.quoteIfNeeded(candidates.get(0)) + "'";
43                 return new RelationUnknown(name, msg);
44             }
45             default: {
46                 var name = RelationName.fromIndexName(relation);
47                 var msg = "Relation '" + relation + "' unknown. Maybe you meant one of: "
48                           + String.join(", ", Lists2.map(candidates, Identifiers::quoteIfNeeded));
49                 return new RelationUnknown(name, msg);
50             }
51         }
52     }
53     public RelationUnknown(String tableName, Throwable e) {
54         super(String.format(Locale.ENGLISH, "Relation '%s' unknown", tableName), e);
55         this.relationName = RelationName.fromIndexName(tableName);
56     }
57     public RelationUnknown(String tableName) {
58         this(RelationName.fromIndexName(tableName), String.format(Locale.ENGLISH, "Relation '%s' unknown", tableName));
59     }
60     public RelationUnknown(RelationName relationName) {
61         this(relationName, String.format(Locale.ENGLISH, "Relation '%s' unknown", relationName));
62     }
63     private RelationUnknown(RelationName relationName, String errorMessage) {
64         super(errorMessage);
65         this.relationName = relationName;
66     }
67     public RelationUnknown(StreamInput in) throws IOException {
68         super(in);

70     }
