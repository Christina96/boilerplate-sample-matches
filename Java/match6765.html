395   private <T> void assertNextPermutation(
396       List<T> expectedPermutation, Iterator<List<T>> permutations) {
397     assertTrue("Expected another permutation, but there was none.", permutations.hasNext());
398     assertEquals(expectedPermutation, permutations.next());
399   }
400   private <T> void assertNoMorePermutations(Iterator<List<T>> permutations) {
401     assertFalse("Expected no more permutations, but there was one.", permutations.hasNext());
402     try {
403       permutations.next();
404       fail("Expected NoSuchElementException.");
405     } catch (NoSuchElementException expected) {
406     }
407   }
408   private <T> void assertPermutationsCount(int expected, Collection<List<T>> permutationSet) {
409     assertEquals(expected, permutationSet.size());
410     Iterator<List<T>> permutations = permutationSet.iterator();
411     for (int i = 0; i < expected; i++) {
412       assertTrue(permutations.hasNext());
413       permutations.next();
414     }
415     assertNoMorePermutations(permutations);

}
