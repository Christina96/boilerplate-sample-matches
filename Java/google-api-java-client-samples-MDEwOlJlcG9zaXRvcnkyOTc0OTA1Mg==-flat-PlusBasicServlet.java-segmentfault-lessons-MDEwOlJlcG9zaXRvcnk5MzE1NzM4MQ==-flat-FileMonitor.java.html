16          updateLastModifiedSnapshot(monitoredFile);
17          pollingWatchingService.scheduleAtFixedRate(() -> {
18              long currentLastModified = monitoredFile.lastModified();

20              if (currentLastModified > previousLastModified) {
21                  eventPublisher.publish(monitoredFile);
22                  updateLastModifiedSnapshot(monitoredFile);
23              }
24          }, 0L, 5L, TimeUnit.SECONDS);
25      }
26      private void updateLastModifiedSnapshot(File monitoredFile) {
27          lastModifiedSnapshot.put(monitoredFile, monitoredFile.lastModified());
28      }
29      public void addListeners(FileChangedListener listener, FileChangedListener... otherListeners) {
30          eventPublisher.addFileChangedListener(listener);
31          Stream.of(otherListeners).forEach(eventPublisher::addFileChangedListener);
32      }
