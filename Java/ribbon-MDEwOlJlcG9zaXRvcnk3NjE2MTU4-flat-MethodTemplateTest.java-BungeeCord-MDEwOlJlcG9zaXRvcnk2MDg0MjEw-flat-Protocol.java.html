601      static final class DirectionData
602      {
603          private final TIntObjectMap protocols = new TIntObjectHashMap<>();
604          private final Protocol protocolPhase;
605          @Getter
606          private final ProtocolConstants.Direction direction;
607          public DirectionData(Protocol protocolPhase, ProtocolConstants.Direction direction)
608          {
609              this.protocolPhase = protocolPhase;
610              this.direction = direction;
611              for ( int protocol : ProtocolConstants.SUPPORTED_VERSION_IDS )
612              {
613                  protocols.put( protocol, new ProtocolData( protocol ) );
614              }
615          }
616          private ProtocolData getProtocolData(int version)
617          {
618              ProtocolData protocol = protocols.get( version );
619              if ( protocol == null && ( protocolPhase != Protocol.GAME ) )
620              {
621                  protocol = Iterables.getFirst( protocols.valueCollection(), null );
622              }
623              return protocol;
624          }
625          public final DefinedPacket createPacket(int id, int version)
626          {
627              ProtocolData protocolData = getProtocolData( version );
628              if ( protocolData == null )
629              {
630                  throw new BadPacketException( "Unsupported protocol version " + version );
631              }
632              if ( id > MAX_PACKET_ID || id < 0 )
633              {
634                  throw new BadPacketException( "Packet with id " + id + " outside of range" );
635              }
636              Supplier constructor = protocolData.packetConstructors[id];
637              return ( constructor == null ) ? null : constructor.get();
638          }
639          private void registerPacket(Class packetClass, Supplier constructor, ProtocolMapping... mappings)
640          {
641              int mappingIndex = 0;
642              ProtocolMapping mapping = mappings[mappingIndex];
643              for ( int protocol : ProtocolConstants.SUPPORTED_VERSION_IDS )
644              {
645                  if ( protocol < mapping.protocolVersion )
646                  {
647                      continue;
648                  }
649                  if ( mapping.protocolVersion < protocol && mappingIndex + 1 < mappings.length )
650                  {
651                      ProtocolMapping nextMapping = mappings[mappingIndex + 1];
652                      if ( nextMapping.protocolVersion == protocol )
653                      {
654                          Preconditions.checkState( nextMapping.packetID != mapping.packetID, "Duplicate packet mapping (%s, %s)", mapping.protocolVersion, nextMapping.protocolVersion );
655                          mapping = nextMapping;
656                          mappingIndex++;
657                      }
658                  }
659                  if ( mapping.packetID < 0 )
660                  {
661                      break;
662                  }
663                  ProtocolData data = protocols.get( protocol );
664                  data.packetMap.put( packetClass, mapping.packetID );
665                  data.packetConstructors[mapping.packetID] = constructor;
666              }
667          }
668          final int getId(Class packet, int version)
669          {
670              ProtocolData protocolData = getProtocolData( version );
671              if ( protocolData == null )
672              {
673                  throw new BadPacketException( "Unsupported protocol version" );
674              }
675              Preconditions.checkArgument( protocolData.packetMap.containsKey( packet ), "Cannot get ID for packet %s in phase %s with direction %s", packet, protocolPhase, direction );
676              return protocolData.packetMap.get( packet );
677          }
