60     @Nullable
61     public String validate(UserDefinedFunctionMetadata meta) {
62         try {
63             resolvePolyglotFunctionValue(meta.name(), meta.definition());
64             return null;
65         } catch (IllegalArgumentException | IOException | PolyglotException t) {
66             return String.format(Locale.ENGLISH, "Invalid JavaScript in function '%s.%s(%s)' AS '%s': %s",
67                 meta.schema(),
68                 meta.name(),
69                 meta.argumentTypes().stream().map(DataType::getName).collect(Collectors.joining(", ")),
70                 meta.definition(),
71                 t.getMessage()
72             );
73         }
74     }
75     static Value resolvePolyglotFunctionValue(String functionName, String script) throws IOException {
76         var context = Context.newBuilder("js")
77             .engine(ENGINE)
78             .allowHostAccess(HOST_ACCESS)
79             .build();
80         var source = Source.newBuilder("js", script, functionName).build();
81         context.eval(source);
82         var polyglotFunctionValue = context.getBindings("js").getMember(functionName);
83         if (polyglotFunctionValue == null) {
84             throw new IllegalArgumentException(
85                 "The name of the function signature '" + functionName + "' doesn't match " +
86                 "the function name in the function definition.");
87         }
88         return polyglotFunctionValue;
89     }
