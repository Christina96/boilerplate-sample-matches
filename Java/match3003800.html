53     public static FilterPath[] compile(Set<String> filters) {
54         if (filters == null || filters.isEmpty()) {
55             return null;
56         }
57         List<FilterPath> paths = new ArrayList<>();
58         for (String filter : filters) {
59             if (filter != null) {
60                 filter = filter.trim();
61                 if (filter.length() > 0) {
62                     paths.add(parse(filter, filter));
63                 }
64             }
65         }
66         return paths.toArray(new FilterPath[paths.size()]);
67     }
68     private static FilterPath parse(final String filter, final String segment) {
69         int end = segment.length();
70         for (int i = 0; i < end; ) {
71             char c = segment.charAt(i);
72             if (c == '.') {
73                 String current = segment.substring(0, i).replaceAll("\\\\.", ".");
74                 return new FilterPath(filter, current, parse(filter, segment.substring(i + 1)));
75             }
76             ++i;
77             if ((c == '\\') && (i < end) && (segment.charAt(i) == '.')) {
78                 ++i;
79             }
80         }
81         return new FilterPath(filter, segment.replaceAll("\\\\.", "."), EMPTY);
82     }

57     public ClusterState execute(ClusterState currentState, RenameTableRequest request) throws Exception {
58         RelationName source = request.sourceTableIdent();
59         RelationName target = request.targetTableIdent();
60         boolean isPartitioned = request.isPartitioned();
61         Metadata currentMetadata = currentState.getMetadata();
62         Metadata.Builder newMetadata = Metadata.builder(currentMetadata);
63         if (isPartitioned) {
64             IndexTemplateMetadata indexTemplateMetadata = DDLClusterStateHelpers.templateMetadata(currentMetadata, source);
65             if (indexTemplateMetadata == null) {
66                 throw new IndexTemplateMissingException("Template for partitioned table is missing");
67             }
68             renameTemplate(newMetadata, indexTemplateMetadata, target);
69         }
70         RoutingTable.Builder newRoutingTable = RoutingTable.builder(currentState.routingTable());
71         ClusterBlocks.Builder blocksBuilder = ClusterBlocks.builder().blocks(currentState.blocks());
72         logger.info("renaming table '{}' to '{}'", source.fqn(), target.fqn());
73         try {
74             Index[] sourceIndices = indexNameExpressionResolver.concreteIndices(
75                 currentState, STRICT_INDICES_OPTIONS, source.indexNameOrAlias());
76             for (Index sourceIndex : sourceIndices) {
77                 IndexMetadata sourceIndexMetadata = currentMetadata.getIndexSafe(sourceIndex);
78                 String sourceIndexName = sourceIndex.getName();
79                 newMetadata.remove(sourceIndexName);
80                 newRoutingTable.remove(sourceIndexName);
81                 blocksBuilder.removeIndexBlocks(sourceIndexName);
82                 IndexMetadata targetMd;
83                 if (isPartitioned) {
84                     PartitionName partitionName = PartitionName.fromIndexOrTemplate(sourceIndexName);
85                     String targetIndexName = IndexParts.toIndexName(target, partitionName.ident());
86                     targetMd = IndexMetadata.builder(sourceIndexMetadata)
87                         .removeAllAliases()
88                         .putAlias(AliasMetadata.builder(target.indexNameOrAlias()).build())
89                         .index(targetIndexName)
90                         .build();
91                 } else {
92                     targetMd = IndexMetadata.builder(sourceIndexMetadata)
93                         .index(target.indexNameOrAlias())
94                         .build();
95                 }
96                 newMetadata.put(targetMd, true);
97                 newRoutingTable.addAsFromCloseToOpen(targetMd);
98                 blocksBuilder.addBlocks(targetMd);
99             }
100         } catch (IndexNotFoundException e) {
101             if (isPartitioned == false) {
102                 throw e;
103             }
104         }
105         ClusterState clusterStateAfterRename = ClusterState.builder(currentState)
106             .metadata(newMetadata)
107             .routingTable(newRoutingTable.build())
108             .blocks(blocksBuilder)
109             .build();
110         return allocationService.reroute(
111             ddlClusterStateService.onRenameTable(clusterStateAfterRename, source, target, request.isPartitioned()),
112             "rename-table"
113         );
114     }
