192     static CompletableFuture<ResolveIndicesAndTemplatesContext> resolveIndexNames(String repositoryName,
193                                                                                   Set<BoundRestoreSnapshot.RestoreTableInfo> restoreTables,
194                                                                                   boolean ignoreUnavailable,
195                                                                                   TransportGetSnapshotsAction transportGetSnapshotsAction) {
196         ResolveIndicesAndTemplatesContext context = new ResolveIndicesAndTemplatesContext();
197         ArrayList<BoundRestoreSnapshot.RestoreTableInfo> toResolveFromSnapshot = new ArrayList<>();
198         for (var table : restoreTables) {
199             if (table.hasPartitionInfo()) {
200                 context.addIndex(table.partitionName().asIndexName());
201                 context.addTemplate(table.partitionTemplate());
202             } else if (ignoreUnavailable) {
203                 context.addIndex(table.tableIdent().indexNameOrAlias());
204                 String templateName = table.partitionTemplate();
205                 context.addIndex(templateName + "*");
206                 context.addTemplate(templateName);
207             } else {
208                 toResolveFromSnapshot.add(table);
209             }
210         }
211         if (toResolveFromSnapshot.isEmpty()) {
212             return CompletableFuture.completedFuture(context);
213         } else {
214             FutureActionListener<GetSnapshotsResponse, ResolveIndicesAndTemplatesContext> listener =
215                 new FutureActionListener<>(
216                     response -> {
217                         resolveTablesFromSnapshots(toResolveFromSnapshot, response.getSnapshots(), context);
218                         return context;
219                     }
220                 );
221             transportGetSnapshotsAction.execute(new GetSnapshotsRequest(repositoryName), listener);
222             return listener;
223         }

233     @VisibleForTesting
234     static void resolveTableFromSnapshots(BoundRestoreSnapshot.RestoreTableInfo table,
235                                           List<SnapshotInfo> snapshots,
236                                           ResolveIndicesAndTemplatesContext context) throws RelationUnknown {
237         String name = table.tableIdent().indexNameOrAlias();
238         for (SnapshotInfo snapshot : snapshots) {
239             for (String index : snapshot.indices()) {
240                 if (name.equals(index)) {
241                     context.addIndex(index);
242                     return;
243                 } else if (isIndexPartitionOfTable(index, table.tableIdent())) {
244                     String templateName = table.partitionTemplate();
245                     context.addIndex(templateName + "*");
246                     context.addTemplate(templateName);
247                     return;
248                 }
249             }
250         }
251         context.addTemplate(table.partitionTemplate());
252     }
