12      public static void main(String[] args) throws IOException {
13          try (ServerSocketChannel serverSocketChannel = ServerSocketChannel.open();) {
14              serverSocketChannel.socket().bind(new InetSocketAddress(8080));
15              serverSocketChannel.configureBlocking(false);
16              System.out.println("当前服务器地址：" + serverSocketChannel.socket().getLocalSocketAddress());
17              Selector selector = Selector.open();
18              serverSocketChannel.register(selector, SelectionKey.OP_ACCEPT);
19              while (true) {
20                  int n = selector.select();
21                  if (n == 0) {
22                      continue;
23                  }
24                  Set keys = selector.selectedKeys();
25                  Iterator iterator = keys.iterator();
26                  while (iterator.hasNext()) {
27                      SelectionKey key = iterator.next();
28                      if (key.isAcceptable()) {
29                          ServerSocketChannel ssc = (ServerSocketChannel) key.channel();
30                          SocketChannel socketChannel = ssc.accept();
31                          if (socketChannel == null) {
32                              continue;
33                          }
34                          System.out.printf("接受客户端[%s] 的连接...%n", socketChannel.getRemoteAddress());
35                          ByteBuffer buffer = ByteBuffer.allocate(8);
36                          buffer.putLong(System.currentTimeMillis());
37                          buffer.flip();
38                          while (buffer.hasRemaining()) {
39                              socketChannel.write(buffer);
40                          }
41                          socketChannel.close();
42                          System.out.println("当前服务器时间已发送到客户端");
43                      }
44                      iterator.remove(); 
45                  }
