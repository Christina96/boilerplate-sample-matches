33 public class NGramTokenizerFactory extends AbstractTokenizerFactory {
34     private final int minGram;
35     private final int maxGram;
36     private final CharMatcher matcher;
37     static final Map<String, CharMatcher> MATCHERS;
38     static {
39         Map<String, CharMatcher> matchers = new HashMap<>();
40         matchers.put("letter", CharMatcher.Basic.LETTER);
41         matchers.put("digit", CharMatcher.Basic.DIGIT);
42         matchers.put("whitespace", CharMatcher.Basic.WHITESPACE);
43         matchers.put("punctuation", CharMatcher.Basic.PUNCTUATION);
44         matchers.put("symbol", CharMatcher.Basic.SYMBOL);
45         for (Field field : Character.class.getFields()) {
46             if (!field.getName().startsWith("DIRECTIONALITY")
47                     && Modifier.isPublic(field.getModifiers())
48                     && Modifier.isStatic(field.getModifiers())
49                     && field.getType() == byte.class) {
50                 try {
51                     matchers.put(field.getName().toLowerCase(Locale.ROOT), CharMatcher.ByUnicodeCategory.of(field.getByte(null)));
52                 } catch (Exception e) {
53                     continue;
54                 }
55             }
56         }
57         MATCHERS = unmodifiableMap(matchers);
58     }
59     static CharMatcher parseTokenChars(List<String> characterClasses) {
60         if (characterClasses == null || characterClasses.isEmpty()) {
61             return null;
62         }
63         CharMatcher.Builder builder = new CharMatcher.Builder();
64         for (String characterClass : characterClasses) {
65             characterClass = characterClass.toLowerCase(Locale.ROOT).trim();
66             CharMatcher matcher = MATCHERS.get(characterClass);
67             if (matcher == null) {
68                 throw new IllegalArgumentException("Unknown token type: '" + characterClass + "', must be one of " + MATCHERS.keySet());
69             }
70             builder.or(matcher);
71         }
72         return builder.build();
73     }
74     NGramTokenizerFactory(IndexSettings indexSettings, Environment environment, String name, Settings settings) {
75         super(indexSettings, name, settings);
76         int maxAllowedNgramDiff = indexSettings.getMaxNgramDiff();
77         this.minGram = settings.getAsInt("min_gram", NGramTokenizer.DEFAULT_MIN_NGRAM_SIZE);
78         this.maxGram = settings.getAsInt("max_gram", NGramTokenizer.DEFAULT_MAX_NGRAM_SIZE);
79         int ngramDiff = maxGram - minGram;
80         if (ngramDiff > maxAllowedNgramDiff) {
81             deprecationLogger.deprecated("Deprecated big difference between max_gram and min_gram in NGram Tokenizer,"
82                 + "expected difference must be less than or equal to: [" + maxAllowedNgramDiff + "]");
83         }
84         this.matcher = parseTokenChars(settings.getAsList("token_chars"));
85     }
86     @Override
87     public Tokenizer create() {
88         if (matcher == null) {
89             return new NGramTokenizer(minGram, maxGram);
90         } else {
91             return new NGramTokenizer(minGram, maxGram) {
92                 @Override
93                 protected boolean isTokenChar(int chr) {
94                     return matcher.isTokenChar(chr);
95                 }
96             };
97         }
98     }
