36     @Override
37     public TemporalAccessor parse(String input) {
38         try {
39             if (input.contains(".")) {
40                 String[] inputs = SPLIT_BY_DOT_PATTERN.split(input, 2);
41                 Long seconds = Long.valueOf(inputs[0]);
42                 if (inputs[1].length() == 0) {
43                     return Instant.ofEpochSecond(seconds, 0).atZone(ZoneOffset.UTC);
44                 }
45                 if (inputs[1].contains("e")) {
46                     return Instant.ofEpochSecond(Double.valueOf(input).longValue()).atZone(ZoneOffset.UTC);
47                 }
48                 if (inputs[1].length() > 9) {
49                     throw new DateTimeParseException("too much granularity after dot [" + input + "]", input, 0);
50                 }
51                 Long nanos = new BigDecimal(inputs[1]).movePointRight(9 - inputs[1].length()).longValueExact();
52                 if (seconds < 0) {
53                     nanos = nanos * -1;
54                 }
55                 return Instant.ofEpochSecond(seconds, nanos).atZone(ZoneOffset.UTC);
56             } else {
57                 return Instant.ofEpochSecond(Long.valueOf(input)).atZone(ZoneOffset.UTC);
58             }
59         } catch (NumberFormatException e) {
60             throw new DateTimeParseException("invalid number [" + input + "]", input, 0, e);
61         }
62     }
