125     public ExecutionPlan build(PlannerContext plannerContext,
126                                Set<PlanHint> hints,
127                                ProjectionBuilder projectionBuilder,
128                                int limit,
129                                int offset,
130                                @Nullable OrderBy order,
131                                @Nullable Integer pageSizeHint,
132                                Row params,
133                                SubQueryResults subQueryResults) {
134         /*
135          * Benchmarks reveal that if rows are filtered out distributed execution gives better performance.
136          * Therefore if `filterNeeded` is true (there is joinCondition or a filtering after the join operation)
137          * then it's a good indication that distributed execution will be faster.
138          *
139          * We may at some point add some kind of session-settings to override this behaviour
140          * or otherwise come up with a better heuristic.
141          */
142         Integer childPageSizeHint = !isFiltered && limit != TopN.NO_LIMIT
143             ? limitAndOffset(limit, offset)
144             : null;
145         ExecutionPlan left = lhs.build(
146             plannerContext, hints, projectionBuilder, NO_LIMIT, 0, null, childPageSizeHint, params, subQueryResults);
147         ExecutionPlan right = rhs.build(
148             plannerContext, hints, projectionBuilder, NO_LIMIT, 0, null, childPageSizeHint, params, subQueryResults);
149         PositionalOrderBy orderByFromLeft = left.resultDescription().orderBy();
150         boolean hasDocTables = baseTables.stream().anyMatch(r -> r instanceof DocTableRelation);
151         boolean isDistributed = hasDocTables && isFiltered && !joinType.isOuter();
152         LogicalPlan leftLogicalPlan = lhs;
153         LogicalPlan rightLogicalPlan = rhs;
154         isDistributed = isDistributed &&
155                         (!left.resultDescription().nodeIds().isEmpty() && !right.resultDescription().nodeIds().isEmpty());
156         boolean blockNlPossible = !isDistributed && isBlockNlPossible(left, right);
157         JoinType joinType = this.joinType;
158         if (!orderByWasPushedDown && joinType.supportsInversion() &&
159             (isDistributed && lhs.numExpectedRows() < rhs.numExpectedRows() && orderByFromLeft == null) ||
160             (blockNlPossible && lhs.numExpectedRows() > rhs.numExpectedRows())) {
161             ExecutionPlan tmpExecutionPlan = left;
162             left = right;
163             right = tmpExecutionPlan;
164             leftLogicalPlan = rhs;
165             rightLogicalPlan = lhs;
166             joinType = joinType.invert();
167         }
168         Tuple<Collection<String>, List<MergePhase>> joinExecutionNodesAndMergePhases =
169             configureExecution(left, right, plannerContext, isDistributed);
170         List<Symbol> joinOutputs = Lists2.concat(leftLogicalPlan.outputs(), rightLogicalPlan.outputs());
171         SubQueryAndParamBinder paramBinder = new SubQueryAndParamBinder(params, subQueryResults);
172         Symbol joinInput = null;
173         if (joinCondition != null) {
174             joinInput = InputColumns.create(paramBinder.apply(joinCondition), joinOutputs);
175         }
176         NestedLoopPhase nlPhase = new NestedLoopPhase(
177             plannerContext.jobId(),
178             plannerContext.nextExecutionPhaseId(),
179             isDistributed ? "distributed-nested-loop" : "nested-loop",
180             Collections.singletonList(JoinOperations.createJoinProjection(outputs, joinOutputs)),
181             joinExecutionNodesAndMergePhases.v2().get(0),
182             joinExecutionNodesAndMergePhases.v2().get(1),
183             leftLogicalPlan.outputs().size(),
184             rightLogicalPlan.outputs().size(),
185             joinExecutionNodesAndMergePhases.v1(),
186             joinType,
187             joinInput,
188             Symbols.typeView(leftLogicalPlan.outputs()),
189             leftLogicalPlan.estimatedRowSize(),
190             leftLogicalPlan.numExpectedRows(),
191             blockNlPossible
192         );
193         return new Join(
194             nlPhase,
195             left,
196             right,
197             TopN.NO_LIMIT,
198             0,
199             TopN.NO_LIMIT,
200             outputs.size(),
201             orderByFromLeft
202         );

261         LinkedHashSet<Symbol> usedFromLeft = new LinkedHashSet<>();
262         LinkedHashSet<Symbol> usedFromRight = new LinkedHashSet<>();
263         for (Symbol usedColumn : usedColumns) {
264             SymbolVisitors.intersection(usedColumn, lhs.outputs(), usedFromLeft::add);
265             SymbolVisitors.intersection(usedColumn, rhs.outputs(), usedFromRight::add);
266         }
267         if (joinCondition != null) {
268             SymbolVisitors.intersection(joinCondition, lhs.outputs(), usedFromLeft::add);
269             SymbolVisitors.intersection(joinCondition, rhs.outputs(), usedFromRight::add);
270         }
271         FetchRewrite lhsFetchRewrite = lhs.rewriteToFetch(tableStats, usedFromLeft);
272         FetchRewrite rhsFetchRewrite = rhs.rewriteToFetch(tableStats, usedFromRight);
273         if (lhsFetchRewrite == null && rhsFetchRewrite == null) {
274             return null;
275         }
276         LinkedHashMap<Symbol, Symbol> allReplacedOutputs = new LinkedHashMap<>();
277         setReplacedOutputs(lhs, lhsFetchRewrite, allReplacedOutputs);
278         setReplacedOutputs(rhs, rhsFetchRewrite, allReplacedOutputs);

282                 lhsFetchRewrite == null ? lhs : lhsFetchRewrite.newPlan(),
283                 rhsFetchRewrite == null ? rhs : rhsFetchRewrite.newPlan(),

302     private Tuple<Collection<String>, List<MergePhase>> configureExecution(ExecutionPlan left,
303                                                                            ExecutionPlan right,
304                                                                            PlannerContext plannerContext,
305                                                                            boolean isDistributed) {
306         Collection<String> nlExecutionNodes = Set.of(plannerContext.handlerNode());
307         ResultDescription leftResultDesc = left.resultDescription();
308         ResultDescription rightResultDesc = right.resultDescription();
309         MergePhase leftMerge = null;
310         MergePhase rightMerge = null;
311         if (leftResultDesc.nodeIds().size() == 1
312             && leftResultDesc.nodeIds().equals(rightResultDesc.nodeIds())
313             && !rightResultDesc.hasRemainingLimitOrOffset()) {
314             nlExecutionNodes = leftResultDesc.nodeIds();
315             left.setDistributionInfo(DistributionInfo.DEFAULT_SAME_NODE);
316             right.setDistributionInfo(DistributionInfo.DEFAULT_SAME_NODE);
317         } else if (isDistributed && !leftResultDesc.hasRemainingLimitOrOffset()) {
318             nlExecutionNodes = leftResultDesc.nodeIds();
319             left.setDistributionInfo(DistributionInfo.DEFAULT_SAME_NODE);
320             right.setDistributionInfo(DistributionInfo.DEFAULT_BROADCAST);
321             rightMerge = JoinOperations.buildMergePhaseForJoin(plannerContext, rightResultDesc, nlExecutionNodes);
322         } else {
323             left.setDistributionInfo(DistributionInfo.DEFAULT_BROADCAST);
324             right.setDistributionInfo(DistributionInfo.DEFAULT_BROADCAST);
325             if (JoinOperations.isMergePhaseNeeded(nlExecutionNodes, leftResultDesc, false)) {
326                 leftMerge = JoinOperations.buildMergePhaseForJoin(plannerContext, leftResultDesc, nlExecutionNodes);
327             }
328             if (JoinOperations.isMergePhaseNeeded(nlExecutionNodes, rightResultDesc, false)) {
329                 rightMerge = JoinOperations.buildMergePhaseForJoin(plannerContext, rightResultDesc, nlExecutionNodes);
330             }
331         }
332         return new Tuple<>(nlExecutionNodes, Arrays.asList(leftMerge, rightMerge));
333     }
