22 package io.crate.metadata.cluster;
23 import io.crate.execution.ddl.index.SwapAndDropIndexRequest;
24 import org.elasticsearch.cluster.ClusterState;
25 import org.elasticsearch.cluster.block.ClusterBlocks;
26 import org.elasticsearch.cluster.metadata.IndexMetadata;
27 import org.elasticsearch.cluster.metadata.Metadata;
28 import org.elasticsearch.cluster.routing.RoutingTable;
29 import org.elasticsearch.cluster.routing.allocation.AllocationService;
30 public class SwapAndDropIndexExecutor extends DDLClusterStateTaskExecutor<SwapAndDropIndexRequest> {
31     private final AllocationService allocationService;
32     public SwapAndDropIndexExecutor(AllocationService allocationService) {
33         this.allocationService = allocationService;
34     }
35     @Override
36     public ClusterState execute(ClusterState currentState, SwapAndDropIndexRequest request) throws Exception {
37         final Metadata metadata = currentState.getMetadata();
38         final ClusterBlocks.Builder blocksBuilder = ClusterBlocks.builder().blocks(currentState.blocks());
39         final Metadata.Builder mdBuilder = Metadata.builder(metadata);
40         final RoutingTable.Builder routingBuilder = RoutingTable.builder(currentState.routingTable());
41         String sourceIndexName = request.source();
42         String targetIndexName = request.target();
43         mdBuilder.remove(sourceIndexName);
44         mdBuilder.remove(targetIndexName);
45         routingBuilder.remove(sourceIndexName);
46         routingBuilder.remove(targetIndexName);
47         blocksBuilder.removeIndexBlocks(sourceIndexName);
48         blocksBuilder.removeIndexBlocks(targetIndexName);
49         IndexMetadata sourceIndex = metadata.index(sourceIndexName);
50         if (sourceIndex == null) {
51             throw new IllegalArgumentException("Source index must exist: " + sourceIndexName);
52         }
53         IndexMetadata newIndexMetadata = IndexMetadata.builder(sourceIndex).index(targetIndexName).build();
54         mdBuilder.put(newIndexMetadata, true);
55         routingBuilder.addAsFromCloseToOpen(newIndexMetadata);
56         blocksBuilder.addBlocks(newIndexMetadata);
57         return allocationService.reroute(
58             ClusterState.builder(currentState)
59                 .metadata(mdBuilder)
60                 .routingTable(routingBuilder.build())
61                 .blocks(blocksBuilder)
62                 .build(),
63             "swap and drop index"
64         );
65     }
