22              return Collections.emptyList();
23          }
24          List allDeclaredFields = new LinkedList<>();
25          Field[] declaredFields = type.getDeclaredFields();
26          allDeclaredFields.addAll(asList(declaredFields));

29              if (superClass == null || Object.class.equals(superClass)) {
30                  break;
31              }
32              allDeclaredFields.addAll(getAllDeclaredFields(superClass));
33              superClass = superClass.getSuperclass();
34          }
35          return allDeclaredFields;
36      }
