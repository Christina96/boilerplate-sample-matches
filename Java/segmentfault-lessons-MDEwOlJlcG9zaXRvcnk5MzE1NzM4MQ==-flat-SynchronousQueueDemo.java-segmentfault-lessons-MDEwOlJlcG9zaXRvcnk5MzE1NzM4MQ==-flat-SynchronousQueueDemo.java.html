1  package com.segmentfault.deep.in.java.concurrency;
2  import java.util.concurrent.*;
3  import java.util.concurrent.atomic.AtomicInteger;
4  public class SynchronousQueueDemo {
5      public static void main(String[] args) throws InterruptedException {
6          BlockingQueue sQueue = new SynchronousQueue<>(true);
7          ExecutorService executorService = Executors.newFixedThreadPool(2);
8          for (AtomicInteger i = new AtomicInteger(1); i.get() < 100; i.incrementAndGet()) {
9              executorService.execute(() -> { 
10                  try {
11                      sQueue.put(i.get());
12                  } catch (Exception e) {
13                      e.printStackTrace();
14                  }
15              });
16              executorService.execute(() -> { 
17                  try {
18                      System.out.println(sQueue.take());
19                  } catch (InterruptedException e) {
20                      e.printStackTrace();
21                  }
22              });
23          }
24          executorService.awaitTermination(10, TimeUnit.MICROSECONDS);
25          executorService.shutdown();
26      }
27      private static void synchronousQueuevsArrayBlockingQueue() throws InterruptedException {
28          BlockingQueue sQueue = new SynchronousQueue();
