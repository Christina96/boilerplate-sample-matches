451     interface ConcurrentlyMutatedList<E> extends List<E> {
452       Set<List<E>> getAllStates();
453     }
454     private static ConcurrentlyMutatedList<Integer> newConcurrentlyMutatedList(
455         final Collection<Integer> initialContents,
456         final Iterable<ListFrobber> actionsToPerformConcurrently) {
457       InvocationHandler invocationHandler =
458           new InvocationHandler() {
459             final CopyOnWriteArrayList<Integer> delegate =
460                 new CopyOnWriteArrayList<>(initialContents);
461             final Method getAllStatesMethod =
462                 getOnlyElement(asList(ConcurrentlyMutatedList.class.getDeclaredMethods()));
463             final Iterator<ListFrobber> remainingActions = actionsToPerformConcurrently.iterator();
464             final Set<List<Integer>> allStates = newHashSet();
465             @Override
466             public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
467               return method.equals(getAllStatesMethod)
468                   ? getAllStates()
469                   : invokeListMethod(method, args);
470             }
471             private Set<List<Integer>> getAllStates() {
472               return allStates;
473             }
474             private Object invokeListMethod(Method method, Object[] args) throws Throwable {
475               try {
476                 Object returnValue = method.invoke(delegate, args);
477                 mutateDelegate();
478                 return returnValue;
479               } catch (InvocationTargetException e) {
480                 throw e.getCause();
481               } catch (IllegalAccessException e) {
482                 throw new AssertionError(e);
483               }
484             }
485             private void mutateDelegate() 
{
486               allStates.add(ImmutableList.copyOf(delegate));
487               remainingActions.next().perform(delegate);
488               allStates.add(ImmutableList.copyOf(delegate));
489             }
490           };
491       @SuppressWarnings("unchecked")
492       ConcurrentlyMutatedList<Integer> list =
493           (ConcurrentlyMutatedList<Integer>)
494               newProxyInstance(
495                   ImmutableListTest.CreationTests.class.getClassLoader(),
496                   new Class[] {ConcurrentlyMutatedList.class},
497                   invocationHandler);
498       return list;
499     }
