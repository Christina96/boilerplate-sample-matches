301     private Tuple<Collection<String>, List<MergePhase>> configureExecution(ExecutionPlan left,
302                                                                            ExecutionPlan right,
303                                                                            PlannerContext plannerContext,
304                                                                            boolean isDistributed) {
305         Collection<String> nlExecutionNodes = Set.of(plannerContext.handlerNode());
306         ResultDescription leftResultDesc = left.resultDescription();
307         ResultDescription rightResultDesc = right.resultDescription();
308         MergePhase leftMerge = null;
309         MergePhase rightMerge = null;
310         if (leftResultDesc.nodeIds().size() == 1
311             && leftResultDesc.nodeIds().equals(rightResultDesc.nodeIds())
312             && !rightResultDesc.hasRemainingLimitOrOffset()) {
313             nlExecutionNodes = leftResultDesc.nodeIds();
314             left.setDistributionInfo(DistributionInfo.DEFAULT_SAME_NODE);
315             right.setDistributionInfo(DistributionInfo.DEFAULT_SAME_NODE);
316         } else if (isDistributed && !leftResultDesc.hasRemainingLimitOrOffset()) {
317             nlExecutionNodes = leftResultDesc.nodeIds();
318             left.setDistributionInfo(DistributionInfo.DEFAULT_SAME_NODE);
319             right.setDistributionInfo(DistributionInfo.DEFAULT_BROADCAST);
320             rightMerge = JoinOperations.buildMergePhaseForJoin(plannerContext, rightResultDesc, nlExecutionNodes);
321         } else {
322             left.setDistributionInfo(DistributionInfo.DEFAULT_BROADCAST);
323             right.setDistributionInfo(DistributionInfo.DEFAULT_BROADCAST);
324             if (JoinOperations.isMergePhaseNeeded(nlExecutionNodes, leftResultDesc, false)) {
325                 leftMerge = JoinOperations.buildMergePhaseForJoin(plannerContext, leftResultDesc, nlExecutionNodes);
326             }
327             if (JoinOperations.isMergePhaseNeeded(nlExecutionNodes, rightResultDesc, false)) {
328                 rightMerge = JoinOperations.buildMergePhaseForJoin(plannerContext, rightResultDesc, nlExecutionNodes);
329             }
330         }
331         return new Tuple<>(nlExecutionNodes, Arrays.asList(leftMerge, rightMerge));
