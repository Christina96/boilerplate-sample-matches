269          }
270          if (!interfaceClass.isInterface()) {
271              throw new IllegalStateException("The interface class " + interfaceClass + " is not a interface!");
272          }
273          if (methods != null && !methods.isEmpty()) {
274              for (MethodConfig methodBean : methods) {
275                  String methodName = methodBean.getName();
276                  if (methodName == null || methodName.length() == 0) {
277                      throw new IllegalStateException(" name attribute is required! Please check: ");
279                  }
280                  java.lang.reflect.Method hasMethod = null;
281                  for (java.lang.reflect.Method method : interfaceClass.getMethods()) {
282                      if (method.getName().equals(methodName)) {
283                          if (methodBean.getArgumentTypes() != null

285                              hasMethod = method;
286                              break;
287                          }
288                          if (methodBean.getArgumentTypes() != null) {
289                              continue;
