47     private static TableHealth map(ClusterIndexHealth indexHealth) {
48         var indexParts = new IndexParts(indexHealth.getIndex());
49         String partitionIdent = null;
50         if (indexParts.isPartitioned()) {

53         int missingPrimaryShards = Math.max(

55             indexHealth.getNumberOfShards() - indexHealth.getActivePrimaryShards()
56         );
57         int underreplicatedShards = Math.max(
58             0,
59             indexHealth.getUnassignedShards() + indexHealth.getInitializingShards() - missingPrimaryShards
60         );
61         return new TableHealth(
62             indexParts.getTable(),
63             indexParts.getSchema(),
64             partitionIdent,
65             TableHealth.Health.valueOf(indexHealth.getStatus().name()),
