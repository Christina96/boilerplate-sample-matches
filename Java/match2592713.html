40     private static final RowType RETURN_TYPE = new RowType(

44     public static void register(TableFunctionModule module) {
45         module.register(
46             Signature.table(
47                 FUNCTION_NAME,

49             ),
50             PgGetKeywordsFunction::new
51         );
52     }
53     private final Signature signature;
54     private final Signature boundSignature;
55     public PgGetKeywordsFunction(Signature signature, Signature boundSignature) {
56         this.signature = signature;
57         this.boundSignature = boundSignature;
58     }
59     @Override
60     public Iterable<Row> evaluate(TransactionContext txnCtx, NodeContext nodeCtx, Input<List<Object>>[] args) {
61         return () -> Identifiers.KEYWORDS.stream()
62             .map(new Function<Identifiers.Keyword, Row>() {
63                 final Object[] columns = new Object[3];
64                 final RowN row = new RowN(columns);
65                 @Override
66                 public Row apply(Identifiers.Keyword keyword) {
67                     columns[0] = keyword.getWord().toLowerCase(Locale.ENGLISH);
68                     if (keyword.isReserved()) {
69                         columns[1] = "R";
70                         columns[2] = "reserved";
71                     } else {
72                         columns[1] = "U";
73                         columns[2] = "unreserved";
74                     }
75                     return row;
76                 }
77             }).iterator();
78     }

86     }

93         return false;
