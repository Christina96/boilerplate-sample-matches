226  					}
227  				}
228  				else
229  				{
230  					ess.getLogger().warning("Invalid command cost for: " + command);

526  	{

528  		final File userdir = new File(ess.getDataFolder(), "userdata");
529  		if (!userdir.exists())

531  			return;
532  		}
