          for (Entry<Range<Integer>, Integer> entry : rangeMap.asMapOfRanges().entrySet()) {

              expected.put(entry.getKey().intersection(subRange), entry.getValue());

          for (int i = MIN_BOUND; i <= MAX_BOUND; i++) {

          for (Range<Integer> query : RANGES) {
