40     private final Pattern<Order> pattern;
41     public MoveOrderBeneathFetchOrEval() {
42         this.fetchCapture = new Capture<>();
43         this.pattern = typeOf(Order.class)
44             .with(source(), typeOf(Eval.class).capturedAs(fetchCapture));
45     }
46     @Override
47     public Pattern<Order> pattern() {
48         return pattern;
49     }
50     @Override
51     public LogicalPlan apply(Order plan,
52                              Captures captures,
53                              TableStats tableStats,
54                              TransactionContext txnCtx,
55                              NodeContext nodeCtx) {
56         Eval eval = captures.get(fetchCapture);
57         List<Symbol> outputsOfSourceOfFetch = eval.source().outputs();
58         List<Symbol> orderBySymbols = plan.orderBy().orderBySymbols();
59         if (outputsOfSourceOfFetch.containsAll(orderBySymbols)
60             || outputsOfSourceOfFetch.containsAll(extractColumns(orderBySymbols))) {
61             return transpose(plan, eval);
62         }
63         return null;
