49     private List<DiscoveryNode> filterNodes(String where) throws NoSuchFieldException, IllegalAccessException {
50         TableInfo tableInfo = SysNodesTableInfo.create();
51         TableRelation tableRelation = new TableRelation(tableInfo);
52         Map<RelationName, AnalyzedRelation> tableSources = Map.of(tableInfo.ident(), tableRelation);
53         SqlExpressions sqlExpressions = new SqlExpressions(tableSources, tableRelation);
54         Symbol query = sqlExpressions.normalize(sqlExpressions.asSymbol(where));
55         List<DiscoveryNode> nodes = new ArrayList<>(NodeStatsCollectSource.filterNodes(
56             discoveryNodes,
57             query,
58             sqlExpressions.nodeCtx));
59         nodes.sort(Comparator.comparing(DiscoveryNode::getId));
60         return nodes;
61     }
