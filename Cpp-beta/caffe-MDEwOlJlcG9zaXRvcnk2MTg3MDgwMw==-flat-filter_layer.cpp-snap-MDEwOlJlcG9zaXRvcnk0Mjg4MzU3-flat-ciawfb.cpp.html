46  void FilterLayer::Forward_cpu(const vector*>& bottom,
47        const vector*>& top) {
48    int new_tops_num = indices_to_forward_.size();
49    for (int t = 0; t < top.size(); ++t) {
50      const Dtype* bottom_data = bottom[t]->cpu_data();
51      Dtype* top_data = top[t]->mutable_cpu_data();
52      int dim = bottom[t]->count() / bottom[t]->shape(0);
53      for (int n = 0; n < new_tops_num; ++n) {
54        int data_offset_top = n * dim;

56        caffe_copy(dim, bottom_data + data_offset_bottom,
57            top_data + data_offset_top);
58      }
