198    } else {

209    int num = stats_batch_size_;

211    caffe_mul(data_stats_count, top_data, top_diff, bottom_diff);
212    caffe_cpu_gemv(CblasNoTrans, channels_ * num, spatial_dim, 1.,
213        bottom_diff, spatial_sum_multiplier_.cpu_data(), 0.,
214        num_by_chans_.mutable_cpu_data());
