4  bool Wordrec::near_point(EDGEPT *point, EDGEPT *line_pt_0, EDGEPT *line_pt_1, EDGEPT **near_pt) {
5    TPOINT p;
6    float slope;
7    float intercept;
8    float x0 = line_pt_0->pos.x;
9    float x1 = line_pt_1->pos.x;
10    float y0 = line_pt_0->pos.y;
11    float y1 = line_pt_1->pos.y;
12    if (x0 == x1) {
13      p.x = static_cast(x0);
14      p.y = point->pos.y;
15    } else {
16      slope = (y0 - y1) / (x0 - x1);
17      intercept = y1 - x1 * slope;
18      p.x = static_cast((point->pos.x + (point->pos.y - intercept) * slope) /
19                                 (slope * slope + 1));
20      p.y = static_cast(slope * p.x + intercept);
21    }
22    if (is_on_line(p, line_pt_0->pos, line_pt_1->pos) && (!same_point(p, line_pt_0->pos)) &&
23        (!same_point(p, line_pt_1->pos))) {
24      *near_pt = make_edgept(p.x, p.y, line_pt_1, line_pt_0);
25      return true;
26    } else { \* Intersection not on line */
27      *near_pt = closest(point, line_pt_0, line_pt_1);
28      return false;
29    }
30  }
