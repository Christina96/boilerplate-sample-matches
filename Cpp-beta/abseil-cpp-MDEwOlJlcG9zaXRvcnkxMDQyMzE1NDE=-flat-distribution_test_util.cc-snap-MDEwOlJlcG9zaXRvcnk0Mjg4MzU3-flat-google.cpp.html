18  DistributionMoments ComputeDistributionMoments(
19      absl::Span data_points) {
20    DistributionMoments result;
21    for (double x : data_points) {
22      result.n++;
23      result.mean += x;
24    }
25    result.mean /= static_cast(result.n);
26    for (double x : data_points) {
27      double v = x - result.mean;
28      result.variance += v * v;
29      result.skewness += v * v * v;
30      result.kurtosis += v * v * v * v;
31    }
32    result.variance /= static_cast(result.n - 1);
33    result.skewness /= static_cast(result.n);
34    result.skewness /= std::pow(result.variance, 1.5);
35    result.kurtosis /= static_cast(result.n);
36    result.kurtosis /= std::pow(result.variance, 2.0);
37    return result;
38  }
