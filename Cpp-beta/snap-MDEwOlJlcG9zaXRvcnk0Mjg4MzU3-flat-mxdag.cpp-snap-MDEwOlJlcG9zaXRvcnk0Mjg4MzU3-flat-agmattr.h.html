28      if (newPos2 < DstPos) { oldOkEdges++; }
29      if (newPos1 < DstPos) { newOkEdges++; } }

31      const int DstPos = NIdPosH.GetDat(NI.GetInNId(e));
32      if (DstPos < newPos2) { oldOkEdges++; }
33      if (DstPos < newPos1) { newOkEdges++; } }
