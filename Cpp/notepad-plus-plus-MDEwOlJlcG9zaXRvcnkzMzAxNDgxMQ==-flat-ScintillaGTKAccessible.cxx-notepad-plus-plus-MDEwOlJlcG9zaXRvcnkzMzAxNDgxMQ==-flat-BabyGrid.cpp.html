543  void ScintillaGTKAccessible::PasteText(int charPosition) {
544  	if (sci->pdoc->IsReadOnly())
545  		return;
546  	struct Helper : GObjectWatcher {
547  		ScintillaGTKAccessible *scia;
548  		Sci::Position bytePosition;
549  		void Destroyed() override {
550  			scia = nullptr;
551  		}
552  		Helper(ScintillaGTKAccessible *scia_, Sci::Position bytePos_) :
553  			GObjectWatcher(G_OBJECT(scia_->sci->sci)),
554  			scia(scia_),
555  			bytePosition(bytePos_) {
556  		}
557  		void TextReceived(GtkClipboard *, const gchar *text) {
558  			if (text) {
559  				size_t len = strlen(text);
560  				std::string convertedText;
561  				if (len > 0 && scia->sci->convertPastes) {
562  					convertedText = Document::TransformLineEnds(text, len, scia->sci->pdoc->eolMode);
563  					len = convertedText.length();
564  					text = convertedText.c_str();
565  				}
566  				scia->InsertStringUTF8(bytePosition, text, static_cast(len));
567  			}
568  		}
569  		static void TextReceivedCallback(GtkClipboard *clipboard, const gchar *text, gpointer data) {
570  			Helper *helper = static_cast(data);
571  			try {
572  				if (helper->scia != nullptr) {
573  					helper->TextReceived(clipboard, text);
574  				}
575  			} catch (...) {}
576  			delete helper;
577  		}
578  	};
579  	Helper *helper = new Helper(this, ByteOffsetFromCharacterOffset(charPosition));
580  	GtkWidget *widget = gtk_accessible_get_widget(accessible);
581  	GtkClipboard *clipboard = gtk_widget_get_clipboard(widget, GDK_SELECTION_CLIPBOARD);

595  void ScintillaGTKAccessible::UpdateCursor() {
596  	Sci::Position pos = sci->WndProc(Message::GetCurrentPos, 0, 0);
597  	if (old_pos != pos) {
598  		int charPosition = CharacterOffsetFromByteOffset(pos);
599  		g_signal_emit_by_name(accessible, "text-caret-moved", charPosition);
600  		old_pos = pos;
601  	}
602  	size_t n_selections = sci->sel.Count();
603  	size_t prev_n_selections = old_sels.size();
604  	bool selection_changed = n_selections != prev_n_selections;
605  	old_sels.resize(n_selections);
606  	for (size_t i = 0; i < n_selections; i++) {
607  		SelectionRange &sel = sci->sel.Range(i);
608  		if (i < prev_n_selections && ! selection_changed) {
609  			SelectionRange &old_sel = old_sels[i];
610  			selection_changed = ((! old_sel.Empty() || ! sel.Empty()) && ! (old_sel == sel));
611  		}
612  		old_sels[i] = sel;
613  	}
614  	if (selection_changed)
615  		g_signal_emit_by_name(accessible, "text-selection-changed");
616  }
617  void ScintillaGTKAccessible::ChangeDocument(Document *oldDoc, Document *newDoc) {
618  	if (!Enabled()) {
619  		return;
620  	}
621  	if (oldDoc == newDoc) {
622  		return;
623  	}
624  	if (oldDoc) {
625  		int charLength = oldDoc->CountCharacters(0, oldDoc->Length());
626  		g_signal_emit_by_name(accessible, "text-changed::delete", 0, charLength);
627  	}
628  	if (newDoc) {
629  		PLATFORM_ASSERT(newDoc == sci->pdoc);
630  		int charLength = newDoc->CountCharacters(0, newDoc->Length());
631  		g_signal_emit_by_name(accessible, "text-changed::insert", 0, charLength);
632  		if ((oldDoc ? oldDoc->IsReadOnly() : false) != newDoc->IsReadOnly()) {
633  			NotifyReadOnly();
634  		}
635  		old_pos = -1;
636  		old_sels.clear();
637  		UpdateCursor();
638  	}
639  }

653  void ScintillaGTKAccessible::Notify(GtkWidget *, gint, NotificationData *nt) {
654  	if (!Enabled())
655  		return;
656  	switch (nt->nmhdr.code) {
657  		case Notification::Modified: {
658  			if (FlagSet(nt->modificationType, ModificationFlags::InsertText)) {
659  				int startChar = CharacterOffsetFromByteOffset(nt->position);
660  				int lengthChar = sci->pdoc->CountCharacters(nt->position, nt->position + nt->length);
661  				g_signal_emit_by_name(accessible, "text-changed::insert", startChar, lengthChar);
662  				UpdateCursor();
663  			}
664  			if (FlagSet(nt->modificationType, ModificationFlags::BeforeDelete)) {
665  				int startChar = CharacterOffsetFromByteOffset(nt->position);
666  				int lengthChar = sci->pdoc->CountCharacters(nt->position, nt->position + nt->length);
667  				g_signal_emit_by_name(accessible, "text-changed::delete", startChar, lengthChar);
668  			}
669  			if (FlagSet(nt->modificationType, ModificationFlags::DeleteText)) {
670  				UpdateCursor();
671  			}
672  			if (FlagSet(nt->modificationType, ModificationFlags::ChangeStyle)) {
673  				g_signal_emit_by_name(accessible, "text-attributes-changed");
674  			}
675  		} break;
676  		case Notification::UpdateUI: {
677  			if (FlagSet(nt->updated, Update::Selection)) {
678  				UpdateCursor();
679  			}
680  		} break;
681  		default:
682  			break;
683  	}
684  }
