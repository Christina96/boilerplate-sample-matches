{

  const ByteVector buffer = Utils::readHeader(stream, bufferSize(), true, &headerOffset);
  if(buffer.isEmpty())

  AdapterFile file(stream);
  for(unsigned int i = 0; i < buffer.size() - 1; ++i) {
