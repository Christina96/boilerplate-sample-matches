12  void ManipulateNodesEdges() {
13    int NNodes = 10000;
14    int NEdges = 100000;
15    const char *FName = "demo.graph.dat";
16    PUndirNet Graph;
17    PUndirNet Graph1;
18    PUndirNet Graph2;
19    int i;
20    int n;
21    int NCount;
22    int ECount1;
23    int ECount2;
24    int x,y;
25    bool t;
26    Graph = TUndirNet::New();
27    t = Graph->Empty();
28    for (i = 0; i < NNodes; i++) {
29      Graph->AddNode(i);
30    }
31    t = Graph->Empty();
32    n = Graph->GetNodes();
33    NCount = NEdges;
34    while (NCount > 0) {
35      x = (long) (drand48() * NNodes);
36      y = (long) (drand48() * NNodes);
37      if (x != y  &&  !Graph->IsEdge(x,y)) {
38        n = Graph->AddEdge(x, y);
39        NCount--;
40      }
41    }
42    PrintGStats("ManipulateNodesEdges:Graph",Graph);
43    NCount = 0;
44    for (TUndirNet::TNodeI NI = Graph->BegNI(); NI < Graph->EndNI(); NI++) {
45      NCount++;
46    }
47    ECount1 = 0;
48    for (TUndirNet::TNodeI NI = Graph->BegNI(); NI < Graph->EndNI(); NI++) {
49      for (int e = 0; e < NI.GetOutDeg(); e++) {
50        ECount1++;
51      }
52    }
53    ECount1 /= 2;
54    ECount2 = 0;
55    for (TUndirNet::TEdgeI EI = Graph->BegEI(); EI < Graph->EndEI(); EI++) {
56      ECount2++;
57    }
58    printf("graph ManipulateNodesEdges:Graph, nodes %d, edges1 %d, edges2 %d\n",
59        NCount, ECount1, ECount2);
60    Graph1 = TUndirNet::New();
61    *Graph1 = *Graph;
62    PrintGStats("ManipulateNodesEdges:Graph1",Graph1);
63    {
64      TFOut FOut(FName);
65      Graph->Save(FOut);
66      FOut.Flush();
67    }
68    {
69      TFIn FIn(FName);
70      Graph2 = TUndirNet::Load(FIn);
71    }
72    PrintGStats("ManipulateNodesEdges:Graph2",Graph2);
73    for (i = 0; i < NNodes; i++) {
74      n = Graph->GetRndNId();
75      Graph->DelNode(n);
76    }
77    PrintGStats("ManipulateNodesEdges:Graph",Graph);
78    Graph1->Clr();
79    PrintGStats("ManipulateNodesEdges:Graph1",Graph1);
