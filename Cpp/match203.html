69         static_cast<CXmlParameterSerializingContext &>(serializingContext);
70     const CComponentLibrary *pComponentLibrary = parameterBuildContext.getComponentLibrary();
71     std::string strComponentType;
72     xmlElement.getAttribute("Type", strComponentType);
73     _pComponentType = pComponentLibrary->getComponentType(strComponentType);
74     if (!_pComponentType) {
75         serializingContext.setError("Unable to create Component " + xmlElement.getPath() +
76                                     ". ComponentType " + strComponentType + " not found!");
77         return false;
78     }
79     if (_pComponentType == getParent()) {
80         serializingContext.setError("Recursive definition of " + _pComponentType->getName() +
81                                     " due to " + xmlElement.getPath() +
82                                     " referring to one of its own type.");
83         return false;
84     }
85     return base::fromXml(xmlElement, serializingContext);
