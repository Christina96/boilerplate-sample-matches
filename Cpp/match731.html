                       "  <member>\n"
                       "    <name>d</name> \n"
                       "    <value><double>43.7</double></value>\n"
                       "  </member>\n"
                       "  <member>\n"
                       "    <name>str</name> \n"
                       "    <value> <string>two</string></value>\n"
                       "  </member>\n"
                       "</struct></value>";

  }

    int ii;
    for (ii = 0; ii < NELMTS; ++ii) {
      char buf[40];
      sprintf(buf, "%d", ii);
      Event[std::string(buf)] = buf;
    }
    Event.clear();
    for (ii = 0; ii < NELMTS; ++ii) {
      char buf[40];
      sprintf(buf, "%d", ii);
      if (ii != NELMTS / 2)
        Event[std::string(buf)] = ii;

          sprintf(bufj, "%d", jj);

        }

      char buf[40];
      sprintf(buf, "%d", ii);
      if (ii != NELMTS / 2)
        EXPECT_EQ(Event[std::string(buf)], XmlRpcValue(ii));
      else
        EXPECT_EQ(Event[std::string(buf)].size(), NELMTS);
    }
  }
}
