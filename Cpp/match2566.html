9 Dtype SGDSolver<Dtype>::GetLearningRate() {
10   Dtype rate;
11   const string& lr_policy = this->param_.lr_policy();
12   if (lr_policy == "fixed") {
13     rate = this->param_.base_lr();
14   } else if (lr_policy == "step") {
15     CHECK_GT(this->param_.stepsize(), 0);
16     this->current_step_ = this->iter_ / this->param_.stepsize();
17     CHECK_GE(this->param_.gamma(), 0);
18     rate = this->param_.base_lr() *
19         pow(this->param_.gamma(), this->current_step_);
20   } else if (lr_policy == "exp") {
21     CHECK_GE(this->param_.gamma(), 0);
22     rate = this->param_.base_lr() * pow(this->param_.gamma(), this->iter_);
23   } else if (lr_policy == "inv") {
24     CHECK_GE(this->param_.gamma(), 0);
25     rate = this->param_.base_lr() *
26         pow(Dtype(1) + this->param_.gamma() * this->iter_,
27             - this->param_.power());
28   } else if (lr_policy == "multistep") {
29     if (this->current_step_ < this->param_.stepvalue_size() &&
30           this->iter_ >= this->param_.stepvalue(this->current_step_)) {
31       this->current_step_++;
32       LOG(INFO) << "MultiStep Status: Iteration " <<
33       this->iter_ << ", step = " << this->current_step_;
34     }
35     CHECK_GE(this->param_.gamma(), 0);
36     rate = this->param_.base_lr() *
37         pow(this->param_.gamma(), this->current_step_);
38   } else if (lr_policy == "poly") {
39     rate = this->param_.base_lr() * pow(Dtype(1.) -
40         (Dtype(this->iter_) / Dtype(this->param_.max_iter())),
41         this->param_.power());
42   } else if (lr_policy == "sigmoid") {
43     CHECK_GE(this->param_.gamma(), 0);
44     CHECK_GT(this->param_.stepsize(), 0);
45     rate = this->param_.base_lr() * (Dtype(1.) /
46         (Dtype(1.) + exp(-this->param_.gamma() * (Dtype(this->iter_) -
47           Dtype(this->param_.stepsize())))));
48   } else {
49     LOG(FATAL) << "Unknown learning rate policy: " << lr_policy;
50   }
51   return rate;
52 }

67 void SGDSolver<Dtype>::ClipGradients() {
68   const Dtype clip_gradients = this->param_.clip_gradients();
69   if (clip_gradients < 0) { return; }
70   const vector<Blob<Dtype>*>& net_params = this->net_->learnable_params();
71   Dtype sumsq_diff = 0;
72   for (int i = 0; i < net_params.size(); ++i) {
73     sumsq_diff += net_params[i]->sumsq_diff();
74   }
75   const Dtype l2norm_diff = std::sqrt(sumsq_diff);
76   if (l2norm_diff > clip_gradients) {
77     Dtype scale_factor = clip_gradients / l2norm_diff;
78     LOG(INFO) << "Gradient clipping: scaling down gradients (L2 norm "
79         << l2norm_diff << " > " << clip_gradients << ") "
80         << "by scale factor " << scale_factor;
81     for (int i = 0; i < net_params.size(); ++i) {
82       net_params[i]->scale_diff(scale_factor);
83     }
84   }
