    if (imageManagerInstance == NULL )

            currentId = imageManagerRegistry.begin()->first;
            imageManagerInstance = imageManagerRegistry.begin()->second;
        }

        {
            yCError(RD_IMG) << "No ImageManager registered, returning null";
        }
    }
    return imageManagerInstance;
}
rd::ImageManager *rd::ImageManager::getImageManager(const std::string & id)
{
    if (imageManagerInstance == NULL )
    {
        if( imageManagerRegistry.find(id) != imageManagerRegistry.end() )
        {
            currentId = id;
            imageManagerInstance = imageManagerRegistry[id];
        }
        else
        {
            yCError(RD_IMG) << "ImageManager" << id << "not found (maybe it was not registered?)";
        }
    }
    return imageManagerInstance;
}
bool rd::ImageManager::destroyImageManager()
{
    if (imageManagerInstance != NULL)
    {
        imageManagerInstance->stop();
        delete imageManagerInstance;
        imageManagerInstance = NULL;
        std::map<std::string, ImageManager *>::iterator it = imageManagerRegistry.find(currentId);
        if (it != imageManagerRegistry.end())
            imageManagerRegistry.erase(it);
    }
    for ( std::map<std::string, ImageManager *>::iterator it = imageManagerRegistry.begin();
          it != imageManagerRegistry.end(); ++it)
    {
        ImageManager * currentManager = it->second;
        if (currentManager != NULL)
        {
            currentManager->stop();
            delete currentManager;
            currentManager = NULL;
         }
         imageManagerRegistry.erase(it);
    }
    return true;
}

bool rd::SDLScreenManager::show()
{
    mutex.lock();
    if (window == NULL)
    {
        if (fullscreen)
        {
            window = SDL_CreateWindow("Robot Devastation",
                                      SDL_WINDOWPOS_UNDEFINED, SDL_WINDOWPOS_UNDEFINED,
                                      200, 100, //-- Arbitrary initial size
                                      SDL_WINDOW_FULLSCREEN | SDL_WINDOW_OPENGL);
        }
        else
        {
            window = SDL_CreateWindow("Robot Devastation",
                                      SDL_WINDOWPOS_UNDEFINED, SDL_WINDOWPOS_UNDEFINED,
                                      200, 100, //-- Arbitrary initial size
                                      0);          }
        if (window==NULL)
        {
            yCError(RD_UI) << "Unable to set video mode:" << SDL_GetError();
            return false;
        }
    }
    int h, w;
    SDL_GetWindowSize(window, &w, &h);
    if (screen->getWidth() != w || screen->getHeight() != h)
        SDL_SetWindowSize(window, screen->getWidth(), screen->getHeight());
    SDL_Surface * screen_surface = SDL_GetWindowSurface(window);
    if (screen_surface == NULL)
    {
        yCError(RD_UI) << "Could not get SDL screen surface!";
        return false;
    }
    if (!screen->drawScreen((void *)screen_surface))
    {
        yCError(RD_UI) << "Could not draw screen!";
        return false;
    }
    SDL_UpdateWindowSurface(window); //Refresh the screen
    SDL_Delay(20); //Wait a bit :)
    SDL_FreeSurface(screen_surface);
    mutex.unlock();
    return true;
}

bool rd::SDLScreenManager::initSDL()
{
    if(SDL_Init(SDL_INIT_VIDEO) < 0)
    {
        yCError(RD_UI) << "SDL could not initialize:" << SDL_GetError();
        return false;
    }
    if (TTF_Init() == -1)
    {
        yCError(RD_UI) << "SDL_ttf could not initialize:" << TTF_GetError();
        cleanupSDL();
        return false;
    }
    if(!(IMG_Init(IMG_INIT_PNG)&IMG_INIT_PNG))
    {
        yCError(RD_UI) << "SDL_image could not initialize:" << IMG_GetError();
        cleanupSDL();
        return false;
    }

    return true;
}
