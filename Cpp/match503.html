void BxpOptimData::compute_pu(const ExtendedSystem& sys, const IntervalVector& box) {
	double pu=1;
	if (sys.active_ctrs(box).empty()) return;
	IntervalVector e=sys.active_ctrs_eval(box);
	for (int j=1; j<e.size();j++) {
		double pui=1;
		if (e[j].diam()>0)
			pui= -e[j].lb()/e[j].diam();
		pu=pu*pui;
	}
}
