			CPPUNIT_ASSERT(m3[i][j]==m[i][0]*m2[0][j]+m[i][1]*m2[1][j]+m[i][2]*m2[2][j]);
	}

	CPPUNIT_ASSERT(m3==(IntervalMatrix(m)*=m2));
}
