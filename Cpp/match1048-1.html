422 bool ASF::File::save()
423 {
424   if(readOnly()) {
425     debug("ASF::File::save() -- File is read only.");
426     return false;
427   }
428   if(!isValid()) {
429     debug("ASF::File::save() -- Trying to save invalid file.");
430     return false;
431   }
432   if(!d->contentDescriptionObject) {
433     d->contentDescriptionObject = new FilePrivate::ContentDescriptionObject();
434     d->objects.append(d->contentDescriptionObject);
435   }
436   if(!d->extendedContentDescriptionObject) {
437     d->extendedContentDescriptionObject = new FilePrivate::ExtendedContentDescriptionObject();
438     d->objects.append(d->extendedContentDescriptionObject);
439   }
440   if(!d->headerExtensionObject) {
441     d->headerExtensionObject = new FilePrivate::HeaderExtensionObject();
442     d->objects.append(d->headerExtensionObject);
443   }
444   if(!d->metadataObject) {
445     d->metadataObject = new FilePrivate::MetadataObject();
446     d->headerExtensionObject->objects.append(d->metadataObject);
447   }
448   if(!d->metadataLibraryObject) {
449     d->metadataLibraryObject = new FilePrivate::MetadataLibraryObject();
450     d->headerExtensionObject->objects.append(d->metadataLibraryObject);
451   }
452   d->extendedContentDescriptionObject->attributeData.clear();
453   d->metadataObject->attributeData.clear();
454   d->metadataLibraryObject->attributeData.clear();
455   const AttributeListMap allAttributes = d->tag->attributeListMap();
456   for(AttributeListMap::ConstIterator it = allAttributes.begin(); it != allAttributes.end(); ++it) {
457     const String &name = it->first;
458     const AttributeList &attributes = it->second;
459     bool inExtendedContentDescriptionObject = false;
460     bool inMetadataObject = false;
461     for(AttributeList::ConstIterator jt = attributes.begin(); jt != attributes.end(); ++jt) {
462       const Attribute &attribute = *jt;
463       const bool largeValue = (attribute.dataSize() > 65535);
464       const bool guid       = (attribute.type() == Attribute::GuidType);
465       if(!inExtendedContentDescriptionObject && !guid && !largeValue && attribute.language() == 0 && attribute.stream() == 0) {
466         d->extendedContentDescriptionObject->attributeData.append(attribute.render(name));
467         inExtendedContentDescriptionObject = true;
468       }
469       else if(!inMetadataObject && !guid && !largeValue && attribute.language() == 0 && attribute.stream() != 0) {
470         d->metadataObject->attributeData.append(attribute.render(name, 1));
471         inMetadataObject = true;
472       }
473       else {
474         d->metadataLibraryObject->attributeData.append(attribute.render(name, 2));

480     data.append((*it)->render(this));

483   writeBlock(ByteVector::fromLongLong(data.size() + 30, false));
484   writeBlock(ByteVector::fromUInt(d->objects.size(), false));
485   writeBlock(ByteVector("\x01\x02", 2));
486   insert(data, 30, static_cast<unsigned long>(d->headerSize - 30));
487   d->headerSize = data.size() + 30;

490 void ASF::File::read()
491 {
492   if(!isValid())
493     return;
494   if(readBlock(16) != headerGuid) {
495     debug("ASF::File::read(): Not an ASF file.");
496     setValid(false);
497     return;
498   }
499   d->tag = new ASF::Tag();
500   d->properties = new ASF::Properties();
501   bool ok;
502   d->headerSize = readQWORD(this, &ok);
503   if(!ok) {
504     setValid(false);
505     return;
506   }
507   int numObjects = readDWORD(this, &ok);
508   if(!ok) {
509     setValid(false);
510     return;
511   }
512   seek(2, Current);
513   FilePrivate::FilePropertiesObject   *filePropertiesObject   = 0;
514   FilePrivate::StreamPropertiesObject *streamPropertiesObject = 0;
515   for(int i = 0; i < numObjects; i++) {
516     const ByteVector guid = readBlock(16);
517     if(guid.size() != 16) {
518       setValid(false);
519       break;
520     }
521     long size = static_cast<long>(readQWORD(this, &ok));
522     if(!ok) {
523       setValid(false);
524       break;
525     }
526     FilePrivate::BaseObject *obj;
527     if(guid == filePropertiesGuid) {
528       filePropertiesObject = new FilePrivate::FilePropertiesObject();
529       obj = filePropertiesObject;
530     }
531     else if(guid == streamPropertiesGuid) {
532       streamPropertiesObject = new FilePrivate::StreamPropertiesObject();
533       obj = streamPropertiesObject;
534     }
535     else if(guid == contentDescriptionGuid) {
536       d->contentDescriptionObject = new FilePrivate::ContentDescriptionObject();
537       obj = d->contentDescriptionObject;
538     }
539     else if(guid == extendedContentDescriptionGuid) {
540       d->extendedContentDescriptionObject = new FilePrivate::ExtendedContentDescriptionObject();
541       obj = d->extendedContentDescriptionObject;
542     }
543     else if(guid == headerExtensionGuid) {
544       d->headerExtensionObject = new FilePrivate::HeaderExtensionObject();
545       obj = d->headerExtensionObject;
546     }
547     else if(guid == codecListGuid) {
548       obj = new FilePrivate::CodecListObject();
549     }
550     else {
551       if(guid == contentEncryptionGuid ||
552          guid == extendedContentEncryptionGuid ||
553          guid == advancedContentEncryptionGuid) {
554         d->properties->setEncrypted(true);
555       }
556       obj = new FilePrivate::UnknownObject(guid);
557     }
558     obj->parse(this, size);
559     d->objects.append(obj);
560   }
561   if(!filePropertiesObject || !streamPropertiesObject) {
562     debug("ASF::File::read(): Missing mandatory header objects.");
563     setValid(false);
564     return;
565   }
