50   ContrastiveLossLayer<Dtype> layer(layer_param);
51   layer.SetUp(this->blob_bottom_vec_, this->blob_top_vec_);
52   layer.Forward(this->blob_bottom_vec_, this->blob_top_vec_);
53   const Dtype margin = layer_param.contrastive_loss_param().margin();
54   const int num = this->blob_bottom_data_i_->num();
55   const int channels = this->blob_bottom_data_i_->channels();
56   Dtype loss(0);
57   for (int i = 0; i < num; ++i) {
58     Dtype dist_sq(0);
59     for (int j = 0; j < channels; ++j) {
60       Dtype diff = this->blob_bottom_data_i_->cpu_data()[i*channels+j] -
61           this->blob_bottom_data_j_->cpu_data()[i*channels+j];
62       dist_sq += diff*diff;
63     }
64     if (this->blob_bottom_y_->cpu_data()[i]) {        loss += dist_sq;
65     } else {
66       Dtype dist = std::max<Dtype>(margin - sqrt(dist_sq), 0.0);
67       loss += dist*dist;
68     }
69   }
70   loss /= static_cast<Dtype>(num) * Dtype(2);
71   EXPECT_NEAR(this->blob_top_loss_->cpu_data()[0], loss, 1e-6);
