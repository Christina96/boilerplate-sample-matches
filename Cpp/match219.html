65 StrategyChoiceModule::set(ExecuteContext& ctx)
66 {
67   auto prefix = ctx.args.get<Name>("prefix");
68   auto strategy = ctx.args.get<Name>("strategy");
69   ctx.controller.start<ndn::nfd::StrategyChoiceSetCommand>(
70     ControlParameters().setName(prefix).setStrategy(strategy),
71     [&] (const ControlParameters& resp) {
72       ctx.out << "strategy-set ";
73       text::ItemAttributes ia;
74       ctx.out << ia("prefix") << resp.getName()
75               << ia("strategy") << resp.getStrategy() << '\n';
76     },
77     [&] (const ControlResponse& resp) {
78       if (resp.getCode() == 404) {
79         ctx.exitCode = 7;
80         ctx.err << "Unknown strategy: " << strategy << '\n';
81         return;
82       }
83       ctx.makeCommandFailureHandler("setting strategy")(resp);     },
84     ctx.makeCommandOptions());
85   ctx.face.processEvents();
