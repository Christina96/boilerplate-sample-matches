345      } else
346      if (WId1<WId2){

357      double WordWgt1=SpV1->GetWgt(RestWIdN1);

361      double WordWgt2=SpV2->GetWgt(RestWIdN2);
362      SqDifSum+=TMath::Sqr(WordWgt2);
363    }
364    double Sim=-sqrt(SqDifSum);
