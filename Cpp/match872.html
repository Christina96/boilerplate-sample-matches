					{-1,0},{-2,0},{-3,0}};

	CPPUNIT_ASSERT(almost_eq(res,z,ERROR));
	CPPUNIT_ASSERT(res.is_superset(z));
}

		}

	for (int i=0; i<nn; i++) {
		if (c[i]!=mz[i]) {
			if (debug) cout << "i=" << i << c[i] << "!=" << mz[i] << endl;

		}

}
