97 bool CXmlDocSource::isInstanceDocumentValid()
98 {
99 #ifdef LIBXML_SCHEMAS_ENABLED
100     string schemaUri = getSchemaUri();
101     xmlDocPtr pSchemaDoc = xmlReadFile(schemaUri.c_str(), nullptr, XML_PARSE_NONET);
102     if (!pSchemaDoc) {
103         return false;
104     }
105     xmlSchemaParserCtxtPtr pParserCtxt = xmlSchemaNewDocParserCtxt(pSchemaDoc);
106     if (!pParserCtxt) {
107         xmlFreeDoc(pSchemaDoc);
108         return false;
109     }
110     xmlSchemaPtr pSchema = xmlSchemaParse(pParserCtxt);
111     if (!pSchema) {
112         xmlSchemaFreeParserCtxt(pParserCtxt);
113         xmlFreeDoc(pSchemaDoc);
114         return false;
115     }
116     xmlSchemaValidCtxtPtr pValidationCtxt = xmlSchemaNewValidCtxt(pSchema);
117     if (!pValidationCtxt) {
118         xmlSchemaFree(pSchema);
119         xmlSchemaFreeParserCtxt(pParserCtxt);
120         xmlFreeDoc(pSchemaDoc);
121         return false;
122     }
123     bool isDocValid = xmlSchemaValidateDoc(pValidationCtxt, _pDoc) == 0;
124     xmlSchemaFreeValidCtxt(pValidationCtxt);
125     xmlSchemaFree(pSchema);
126     xmlSchemaFreeParserCtxt(pParserCtxt);
127     xmlFreeDoc(pSchemaDoc);
128     return isDocValid;
129 #else
130     return true;
131 #endif

142 _xmlDoc *CXmlDocSource::mkXmlDoc(const string &source, bool fromFile, bool xincludes,
143                                  CXmlSerializingContext &serializingContext)
144 {
145     _xmlDoc *doc = nullptr;
146     if (fromFile) {
147         doc = xmlReadFile(source.c_str(), nullptr, 0);
148     } else {
149         doc = xmlReadMemory(source.c_str(), (int)source.size(), "", nullptr, 0);
150     }
151     if (doc == nullptr) {
152         string errorMsg = "libxml failed to read";
153         if (fromFile) {
154             errorMsg += " \"" + source + "\"";
155         }
156         serializingContext.appendLineToError(errorMsg);
157         return nullptr;
158     }
159     if (xincludes and (xmlXIncludeProcess(doc) < 0)) {
160         serializingContext.appendLineToError("libxml failed to resolve XIncludes");
161         xmlFreeDoc(doc);
162         doc = nullptr;
163     }
164     return doc;
