11  unsigned long long streamToInt(const std::string &stream)
12  {
13      if(!stream.size())
14          return 0;
15      double streamval = 1.0;
16      std::vector units = {"B", "KB", "MB", "GB", "TB", "PB", "EB"};
17      size_t index = units.size();
18      do
19      {
20          index--;
21          if(endsWith(stream, units[index]))
22          {
23              streamval = std::pow(1024, index) * to_number(stream.substr(0, stream.size() - units[index].size()), 0.0);

26      }
