101         !getWheelNames(controller_nh, "right_wheel", right_wheel_names))

104     }

141     ROS_INFO_STREAM_NAMED(name_, "Right wheel radius will be multiplied by "
142                           << right_wheel_radius_multiplier_ << ".");
143 
    int velocity_rolling_window_size = 10;
144     controller_nh.param("velocity_rolling_window_size", velocity_rolling_window_size, velocity_rolling_window_size);
145     ROS_INFO_STREAM_NAMED(name_, "Velocity rolling window size of "
146                           << velocity_rolling_window_size << ".");
147     odometry_.setVelocityRollingWindowSize(velocity_rolling_window_size);
148     controller_nh.param("cmd_vel_timeout", cmd_vel_timeout_, cmd_vel_timeout_);
149     ROS_INFO_STREAM_NAMED(name_, "Velocity commands will be considered old if they are older than "
150                           << cmd_vel_timeout_ << "s.");
151     controller_nh.param("allow_multiple_cmd_vel_publishers", allow_multiple_cmd_vel_publishers_, allow_multiple_cmd_vel_publishers_);
152     ROS_INFO_STREAM_NAMED(name_, "Allow mutiple cmd_vel publishers is "
153                           << (allow_multiple_cmd_vel_publishers_?"enabled":"disabled"));
154     controller_nh.param("base_frame_id", base_frame_id_, base_frame_id_);
155     ROS_INFO_STREAM_NAMED(name_, "Base frame_id set to " << base_frame_id_);
156     controller_nh.param("odom_frame_id", odom_frame_id_, odom_frame_id_);
157     ROS_INFO_STREAM_NAMED(name_, "Odometry frame_id set to " << odom_frame_id_);
158     controller_nh.param("enable_odom_tf", enable_odom_tf_, enable_odom_tf_);
159     ROS_INFO_STREAM_NAMED(name_, "Publishing to tf is " << (enable_odom_tf_?"enabled":"disabled"));
160     controller_nh.param("linear/x/has_velocity_limits"    , limiter_lin_.has_velocity_limits    , limiter_lin_.has_velocity_limits    );

168     controller_nh.param("linear/x/min_jerk"               , limiter_lin_.min_jerk               , -limiter_lin_.max_jerk              );

172     controller_nh.param("angular/z/max_velocity"           , limiter_ang_.max_velocity           ,  limiter_ang_.max_velocity          );

200       cmd_vel_pub_.reset(new realtime_tools::RealtimePublisher<geometry_msgs::TwistStamped>(controller_nh, "cmd_vel_out", 100));
201     }
202     if (publish_wheel_joint_controller_state_)

250     dyn_reconf_server_->updateConfig(config);

252     dyn_reconf_server_->setCallback(
253         boost::bind(&DiffDriveController::reconfCallback, this, boost::placeholders::_1, boost::placeholders::_2));
254     return true;

256   void DiffDriveController::update(const ros::Time& time, const ros::Duration& period)
257   {
258     updateDynamicParams();
259     const double ws  = wheel_separation_multiplier_   * wheel_separation_;
260     const double lwr = left_wheel_radius_multiplier_  * wheel_radius_;
261     const double rwr = right_wheel_radius_multiplier_ * wheel_radius_;
262     odometry_.setWheelParams(ws, lwr, rwr);
263     if (open_loop_)
264     {
265       odometry_.updateOpenLoop(last0_cmd_.lin, last0_cmd_.ang, time);
266     }
267     else
268     {
269       double left_pos  = 0.0;
270       double right_pos = 0.0;
271       for (size_t i = 0; i < wheel_joints_size_; ++i)
272       {
273         const double lp = left_wheel_joints_[i].getPosition();
274         const double rp = right_wheel_joints_[i].getPosition();
275         if (std::isnan(lp) || std::isnan(rp))
276           return;
277         left_pos  += lp;
278         right_pos += rp;
279       }
280       left_pos  /= wheel_joints_size_;
281 
      right_pos /= wheel_joints_size_;
282       odometry_.update(left_pos, right_pos, time);
283     }
284     if (last_state_publish_time_ + publish_period_ < time)
285     {
286       last_state_publish_time_ += publish_period_;
287       const geometry_msgs::Quaternion orientation(
288             tf::createQuaternionMsgFromYaw(odometry_.getHeading()));
289       if (odom_pub_->trylock())
290       {
291         odom_pub_->msg_.header.stamp = time;
292         odom_pub_->msg_.pose.pose.position.x = odometry_.getX();
293         odom_pub_->msg_.pose.pose.position.y = odometry_.getY();
294         odom_pub_->msg_.pose.pose.orientation = orientation;
295         odom_pub_->msg_.twist.twist.linear.x  = odometry_.getLinear();
296         odom_pub_->msg_.twist.twist.angular.z = odometry_.getAngular();
297         odom_pub_->unlockAndPublish();
298       }
299       if (enable_odom_tf_ && tf_odom_pub_->trylock())
300       {
301         geometry_msgs::TransformStamped& odom_frame = tf_odom_pub_->msg_.transforms[0];
302         odom_frame.header.stamp = time;
303         odom_frame.transform.translation.x = odometry_.getX();
304         odom_frame.transform.translation.y = odometry_.getY();
305         odom_frame.transform.rotation = orientation;
306         tf_odom_pub_->unlockAndPublish();
307       }
308     }
309     Commands curr_cmd = *(command_.readFromRT());
310     const double dt = (time - curr_cmd.stamp).toSec();
311     if (dt > cmd_vel_timeout_)
312     {
313       curr_cmd.lin = 0.0;
314       curr_cmd.ang = 0.0;
315     }
316     const double cmd_dt(period.toSec());
317     limiter_lin_.limit(curr_cmd.lin, last0_cmd_.lin, last1_cmd_.lin, cmd_dt);
318     limiter_ang_.limit(curr_cmd.ang, last0_cmd_.ang, last1_cmd_.ang, cmd_dt);
319     last1_cmd_ = last0_cmd_;
320     last0_cmd_ = curr_cmd;
321     if (publish_cmd_ && cmd_vel_pub_ && cmd_vel_pub_->trylock())
322     {
323       cmd_vel_pub_->msg_.header.stamp = time;
324       cmd_vel_pub_->msg_.twist.linear.x = curr_cmd.lin;
325       cmd_vel_pub_->msg_.twist.angular.z = curr_cmd.ang;
326       cmd_vel_pub_->unlockAndPublish();
327     }
328     const double vel_left  = (curr_cmd.lin - curr_cmd.ang * ws / 2.0)/lwr;
329     const double vel_right = (curr_cmd.lin + curr_cmd.ang * ws / 2.0)/rwr;
330     for (size_t i = 0; i < wheel_joints_size_; ++i)
331     {
332       left_wheel_joints_[i].setCommand(vel_left);
333       right_wheel_joints_[i].setCommand(vel_right);
334     }
335     publishWheelData(time, period, curr_cmd, ws, lwr, rwr);
336     time_previous_ = time;
337   }

358   void DiffDriveController::cmdVelCallback(const geometry_msgs::Twist& command)
359   {
360     if (isRunning())
361     {
362       if (!allow_multiple_cmd_vel_publishers_ && sub_command_.getNumPublishers() > 1)
363       {
364         ROS_ERROR_STREAM_THROTTLE_NAMED(1.0, name_, "Detected " << sub_command_.getNumPublishers()
365             << " publishers. Only 1 publisher is allowed. Going to brake.");
366         brake();
367         return;
368       }
369       if(!std::isfinite(command.angular.z) || !std::isfinite(command.linear.x))
370       {
371         ROS_WARN_THROTTLE(1.0, "Received NaN in velocity command. Ignoring.");
372         return;
373       }
374       command_struct_.ang   = command.angular.z;
375       command_struct_.lin   = command.linear.x;
376       command_struct_.stamp = ros::Time::now();
377       command_.writeFromNonRT (command_struct_);
378       ROS_DEBUG_STREAM_NAMED(name_,
379                              "Added values to command. "
380                              << "Ang: "   << command_struct_.ang << ", "
381                              << "Lin: "   << command_struct_.lin << ", "
382                              << "Stamp: " << command_struct_.stamp);
383     }
384     else
385     {
386       ROS_ERROR_NAMED(name_, "Can't accept new commands. Controller is not running.");
387     }
388   }
389   bool DiffDriveController::getWheelNames(ros::NodeHandle& controller_nh,
390                               const std::string& wheel_param,
391                               std::vector<std::string>& wheel_names)
392   {
393       XmlRpc::XmlRpcValue wheel_list;
394       if (!controller_nh.getParam(wheel_param, wheel_list))
395       {
396         ROS_ERROR_STREAM_NAMED(name_,
397             "Couldn't retrieve wheel param '" << wheel_param << "'.");
398         return false;
399       }
400       if (wheel_list.getType() == XmlRpc::XmlRpcValue::TypeArray)
401       {
402         if (wheel_list.size() == 0)
403         {
404           ROS_ERROR_STREAM_NAMED(name_,
405               "Wheel param '" << wheel_param << "' is an empty list");
406           return false;
407         }
408         for (int i = 0; i < wheel_list.size(); ++i)
409         {
410           if (wheel_list[i].getType() != XmlRpc::XmlRpcValue::TypeString)
411           {
412             ROS_ERROR_STREAM_NAMED(name_,
413                 "Wheel param '" << wheel_param << "' #" << i <<
414                 " isn't a string.");
415             return false;
416           }
417         }
418         wheel_names.resize(wheel_list.size());
419         for (int i = 0; i < wheel_list.size(); ++i)
420         {
421           wheel_names[i] = static_cast<std::string>(wheel_list[i]);
422         }
423       }
424       else if (wheel_list.getType() == XmlRpc::XmlRpcValue::TypeString)
425       {
426         wheel_names.push_back(wheel_list);
427       }
428       else
429       {
430         ROS_ERROR_STREAM_NAMED(name_,
431             "Wheel param '" << wheel_param <<
432             "' is neither a list of strings nor a string.");
433         return false;
434       }
435       return true;
436 
  }
437   bool DiffDriveController::setOdomParamsFromUrdf(ros::NodeHandle& root_nh,
438                              const std::string& left_wheel_name,
439                              const std::string& right_wheel_name,
440                              bool lookup_wheel_separation,
441                              bool lookup_wheel_radius)
442   {
443     if (!(lookup_wheel_separation || lookup_wheel_radius))
444     {
445       return true;
446     }
447     const std::string model_param_name = "robot_description";
448     bool res = root_nh.hasParam(model_param_name);
449     std::string robot_model_str="";
450     if (!res || !root_nh.getParam(model_param_name,robot_model_str))
451     {
452       ROS_ERROR_NAMED(name_, "Robot description couldn't be retrieved from param server.");
453       return false;
454     }
455     urdf::ModelInterfaceSharedPtr model(urdf::parseURDF(robot_model_str));
456     urdf::JointConstSharedPtr left_wheel_joint(model->getJoint(left_wheel_name));
457     urdf::JointConstSharedPtr right_wheel_joint(model->getJoint(right_wheel_name));
458     if (!left_wheel_joint)
459     {
460       ROS_ERROR_STREAM_NAMED(name_, left_wheel_name
461                              << " couldn't be retrieved from model description");
462       return false;
463     }
464     if (!right_wheel_joint)
465     {
466       ROS_ERROR_STREAM_NAMED(name_, right_wheel_name
467                              << " couldn't be retrieved from model description");
468       return false;
469     }
470     if (lookup_wheel_separation)
471     {
472       ROS_INFO_STREAM("left wheel to origin: " << left_wheel_joint->parent_to_joint_origin_transform.position.x << ","
473                       << left_wheel_joint->parent_to_joint_origin_transform.position.y << ", "
474                       << left_wheel_joint->parent_to_joint_origin_transform.position.z);
475       ROS_INFO_STREAM("right wheel to origin: " << right_wheel_joint->parent_to_joint_origin_transform.position.x << ","
476                       << right_wheel_joint->parent_to_joint_origin_transform.position.y << ", "
477                       << right_wheel_joint->parent_to_joint_origin_transform.position.z);
478 
      wheel_separation_ = euclideanOfVectors(left_wheel_joint->parent_to_joint_origin_transform.position,
479                                              right_wheel_joint->parent_to_joint_origin_transform.position);
480     }
481     if (lookup_wheel_radius)
482     {
483       if (!getWheelRadius(model->getLink(left_wheel_joint->child_link_name), wheel_radius_))
484       {
485         ROS_ERROR_STREAM_NAMED(name_, "Couldn't retrieve " << left_wheel_name << " wheel radius");
486         return false;
487       }
488     }
489     return true;
490   }

539     dynamic_params.enable_odom_tf = config.enable_odom_tf;
540     dynamic_params_.writeFromNonRT(dynamic_params);
541     ROS_INFO_STREAM_NAMED(name_, "Dynamic Reconfigure:\n" << dynamic_params);
542   }
543   void DiffDriveController::updateDynamicParams()
544   {
545     const DynamicParams dynamic_params = *(dynamic_params_.readFromRT());
546     left_wheel_radius_multiplier_  = dynamic_params.left_wheel_radius_multiplier;
547     right_wheel_radius_multiplier_ = dynamic_params.right_wheel_radius_multiplier;
548     wheel_separation_multiplier_   = dynamic_params.wheel_separation_multiplier;
549     publish_period_ = ros::Duration(1.0 / dynamic_params.publish_rate);
550     enable_odom_tf_ = dynamic_params.enable_odom_tf;
551   }
552   void DiffDriveController::publishWheelData(const ros::Time& time, const ros::Duration& period, Commands& curr_cmd,
553           double wheel_separation, double left_wheel_radius, double right_wheel_radius)
554   {
555     if (publish_wheel_joint_controller_state_ && controller_state_pub_->trylock())
556     {
557       const double cmd_dt(period.toSec());
558       const double vel_left_desired  = (curr_cmd.lin - curr_cmd.ang * wheel_separation / 2.0) / left_wheel_radius;
559       const double vel_right_desired = (curr_cmd.lin + curr_cmd.ang * wheel_separation / 2.0) / right_wheel_radius;
560       controller_state_pub_->msg_.header.stamp = time;
561       for (size_t i = 0; i < wheel_joints_size_; ++i)
562       {
563         const double control_duration = (time - time_previous_).toSec();
564         const double left_wheel_acc = (left_wheel_joints_[i].getVelocity() - vel_left_previous_[i]) / control_duration;
565         const double right_wheel_acc = (right_wheel_joints_[i].getVelocity() - vel_right_previous_[i]) / control_duration;
566         controller_state_pub_->msg_.actual.positions[i]     = left_wheel_joints_[i].getPosition();
567         controller_state_pub_->msg_.actual.velocities[i]    = left_wheel_joints_[i].getVelocity();
568         controller_state_pub_->msg_.actual.accelerations[i] = left_wheel_acc;
569         controller_state_pub_->msg_.actual.effort[i]        = left_wheel_joints_[i].getEffort();
570         controller_state_pub_->msg_.actual.positions[i + wheel_joints_size_]     = right_wheel_joints_[i].getPosition();
571         controller_state_pub_->msg_.actual.velocities[i + wheel_joints_size_]    = right_wheel_joints_[i].getVelocity();
572         controller_state_pub_->msg_.actual.accelerations[i + wheel_joints_size_] = right_wheel_acc;
573         controller_state_pub_->msg_.actual.effort[i+ wheel_joints_size_]         = right_wheel_joints_[i].getEffort();
574         controller_state_pub_->msg_.desired.positions[i]    += vel_left_desired * cmd_dt;
575         controller_state_pub_->msg_.desired.velocities[i]    = vel_left_desired;
576         controller_state_pub_->msg_.desired.accelerations[i] = (vel_left_desired - vel_left_desired_previous_) * cmd_dt;
577         controller_state_pub_->msg_.desired.effort[i]        = std::numeric_limits<double>::quiet_NaN();
578         controller_state_pub_->msg_.desired.positions[i + wheel_joints_size_]    += vel_right_desired * cmd_dt;
579         controller_state_pub_->msg_.desired.velocities[i + wheel_joints_size_]    = vel_right_desired;
580         controller_state_pub_->msg_.desired.accelerations[i + wheel_joints_size_] = (vel_right_desired - vel_right_desired_previous_) * cmd_dt;
581         controller_state_pub_->msg_.desired.effort[i+ wheel_joints_size_]         = std::numeric_limits<double>::quiet_NaN();
582         controller_state_pub_->msg_.error.positions[i]     = controller_state_pub_->msg_.desired.positions[i] -
583                                                                               controller_state_pub_->msg_.actual.positions[i];
584         controller_state_pub_->msg_.error.velocities[i]    = controller_state_pub_->msg_.desired.velocities[i] -
585                                                                               controller_state_pub_->msg_.actual.velocities[i];
586         controller_state_pub_->msg_.error.accelerations[i] = controller_state_pub_->msg_.desired.accelerations[i] -
587                                                                               controller_state_pub_->msg_.actual.accelerations[i];
588         controller_state_pub_->msg_.error.effort[i]        = controller_state_pub_->msg_.desired.effort[i] -
589                                                                               controller_state_pub_->msg_.actual.effort[i];
590         controller_state_pub_->msg_.error.positions[i + wheel_joints_size_]     = controller_state_pub_->msg_.desired.positions[i + wheel_joints_size_] -
591                                                                                                    controller_state_pub_->msg_.actual.positions[i + wheel_joints_size_];
592         controller_state_pub_->msg_.error.velocities[i + wheel_joints_size_]    = controller_state_pub_->msg_.desired.velocities[i + wheel_joints_size_] -
593                                                                                                    controller_state_pub_->msg_.actual.velocities[i + wheel_joints_size_];
594         controller_state_pub_->msg_.error.accelerations[i + wheel_joints_size_] = controller_state_pub_->msg_.desired.accelerations[i + wheel_joints_size_] -
595                                                                                                    controller_state_pub_->msg_.actual.accelerations[i + wheel_joints_size_];
596         controller_state_pub_->msg_.error.effort[i+ wheel_joints_size_]         = controller_state_pub_->msg_.desired.effort[i + wheel_joints_size_] -
597                                                                                                    controller_state_pub_->msg_.actual.effort[i + wheel_joints_size_];
598         vel_left_previous_[i] = left_wheel_joints_[i].getVelocity();
599         vel_right_previous_[i] = right_wheel_joints_[i].getVelocity();
600         vel_left_desired_previous_ = vel_left_desired;
601         vel_right_desired_previous_ = vel_right_desired;
602       }
603       controller_state_pub_->unlockAndPublish();
604     }
605   }
