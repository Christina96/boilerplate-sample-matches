74 EthernetTransport::sendPacket(const ndn::Block& block)
75 {
76   ndn::EncodingBuffer buffer(block);
77   if (block.size() < ethernet::MIN_DATA_LEN) {
78     static const uint8_t padding[ethernet::MIN_DATA_LEN] = {};
79     buffer.appendBytes(ndn::make_span(padding).subspan(block.size()));
80   }

82   buffer.prependBytes({reinterpret_cast<const uint8_t*>(&ethertype), ethernet::TYPE_LEN});
83   buffer.prependBytes(m_srcAddress);
84   buffer.prependBytes(m_destAddress);
85   int sent = pcap_inject(m_pcap, buffer.data(), buffer.size());
86   if (sent < 0)

88   else if (static_cast<size_t>(sent) < buffer.size())
89     handleError("Failed to send the full frame: size=" + to_string(buffer.size()) +
90                 " sent=" + to_string(sent));

100 void
101 EthernetTransport::handleRead(const boost::system::error_code& error)
102 {
103   if (error) {
104     if (error != boost::asio::error::operation_aborted &&
105         getState() != TransportState::CLOSING &&
106         getState() != TransportState::FAILED &&
107         getState() != TransportState::CLOSED) {
108       handleError("Receive operation failed: " + error.message());
109     }
110     return;
111   }
112   span<const uint8_t> pkt;
113   std::string err;
114   std::tie(pkt, err) = m_pcap.readNextPacket();
115   if (pkt.empty()) {
116     NFD_LOG_FACE_WARN("Read error: " << err);
117   }
118   else {
119     const ether_header* eh;
120     std::tie(eh, err) = ethernet::checkFrameHeader(pkt, m_srcAddress,
121                                                    m_destAddress.isMulticast() ? m_destAddress : m_srcAddress);
122     if (eh == nullptr) {
123       NFD_LOG_FACE_WARN(err);
124     }
125     else {
126       ethernet::Address sender(eh->ether_shost);
127       pkt = pkt.subspan(ethernet::HDR_LEN);
128       receivePayload(pkt, sender);
129     }
130   }
131 #ifdef _DEBUG
132   size_t nDropped = m_pcap.getNDropped();
133   if (nDropped - m_nDropped > 0)
134     NFD_LOG_FACE_DEBUG("Detected " << nDropped - m_nDropped << " dropped frame(s)");
135   m_nDropped = nDropped;
136 #endif
137   asyncRead();
